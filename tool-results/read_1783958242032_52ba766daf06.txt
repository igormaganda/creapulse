    71→    71→function isFilled(value: unknown): boolean {
    72→    72→  if (value === null || value === undefined) return false
    73→    73→  if (typeof value === 'string') return value.trim().length > 0
    74→    74→  if (Array.isArray(value)) return value.length > 0
    75→    75→  if (typeof value === 'object') {
    76→    76→    return Object.values(value).some((v) =>
    77→    77→      typeof v === 'string' && v.trim().length > 0 ||
    78→    78→      typeof v === 'number' && v !== 0 ||
    79→    79→      Array.isArray(v) && v.length > 0
    80→    80→    )
    81→    81→  }
    82→    82→  return false
    83→    83→}
    84→    84→
    85→    85→function computeWordCount(value: unknown): number {
    86→    86→  if (typeof value === 'string') return value.trim().split(/\s+/).filter(Boolean).length
    87→    87→  return 0
    88→    88→}

   125→   125→/** Auto-create a snapshot if last one is older than 5 minutes (debounce) */
   126→   126→async function maybeAutoSnapshot(
   127→   127→  userId: string,
   128→   128→  journeyId: string,
   129→   129→  tenantId: string,
   130→   130→  bpSections: Record<string, unknown>,
   131→   131→  bpProjectContext: Record<string, unknown> | null,
   132→   132→  trigger: 'auto' | 'ai-generate' = 'auto',
   133→   133→) {
   134→   134→  try {
   135→   135→    const lastSnapshot = await db.bpSnapshot.findFirst({
   136→   136→      where: { userId },
   137→   137→      orderBy: { createdAt: 'desc' },
   138→   138→      select: { createdAt: true, version: true },
   139→   139→    })
   140→   140→    const fiveMinAgo = new Date(Date.now() - 5 * 60_000)
   141→   141→    if (lastSnapshot && lastSnapshot.createdAt > fiveMinAgo) return
   142→   142→
   143→   143→    const sectionCount = Object.values(bpSections).filter(isFilled).length
   144→   144→    const wordCount = Object.values(bpSections).reduce(
   145→   145→      (sum: number, v) => sum + computeWordCount(v), 0,
   146→   146→    )
   147→   147→    const version: number = (lastSnapshot?.version ?? 0) + 1
   148→   148→    const now = new Date()
   149→   149→    const label = `V${version} — ${now.toLocaleDateString('fr-FR', { day: 'numeric', month: 'short', year: 'numeric' })} ${now.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' })}`
   150→   150→
   151→   151→    await db.bpSnapshot.create({
   152→   152→      data: {
   153→   153→        userId,
   154→   154→        tenantId,
   155→   155→        creatorJourneyId: journeyId,
   156→   156→        bpSections: bpSections as Prisma.InputJsonValue,
   157→   157→        bpProjectContext: (bpProjectContext ?? {}) as Prisma.InputJsonValue,
   158→   158→        version,
   159→   159→        label,
   160→   160→        trigger,
   161→   161→        sectionCount,
   162→   162→        wordCount,
   163→   163→      },
   164→   164→    })
   165→   165→  } catch (err) {
   166→   166→    // Non-critical: log but don't fail the parent operation
   167→   167→    console.error('[AutoSnapshot] Failed to create auto-snapshot:', err)
   168→   168→  }
   169→   169→}

   173→   173→async function getAuth(request: NextRequest) {
   174→   174→  const cookieToken = request.cookies.get('session')?.value
   175→   175→  const headerToken = getTokenFromHeader(request)
   176→   176→  const token = cookieToken || headerToken
   177→   177→  if (!token) throw Object.assign(new Error('No session token found'), { code: 'UNAUTHORIZED' })
   178→   178→  return verifyToken(token)
   179→   179→}

   182→   182→
   183→   183→function parseJsonFromLLM(text: string): Record<string, string> | null {
   184→   184→  // Try direct JSON parse first
   185→   185→  try {
   186→   186→    const parsed = JSON.parse(text)
   187→   187→    if (typeof parsed === 'object' && parsed !== null) return parsed
   188→   188→  } catch {}
   189→   189→
   190→   190→  // Try matching JSON block
   191→   191→  const jsonMatch = text.match(/\{[\s\S]*\}/)
   192→   192→  if (jsonMatch) {
   193→   193→    try {
   194→   194→      const parsed = JSON.parse(jsonMatch[0])
   195→   195→      if (typeof parsed === 'object' && parsed !== null) return parsed
   196→   196→    } catch {}
   197→   197→  }
   198→   198→
   199→   199→  return null
   200→   200→}

   443→   443→    return aiErrorResponse("La réponse IA n'a pas pu être interprétée. Veuillez réessayer.")
   444→   444→  }
   445→   445→
   446→   446→  // 6. Merge into existing bpSections (don't overwrite filled sections)
   447→   447→  const existingSections = (journey?.bpSections as Record<string, unknown>) ?? {}
   448→   448→  const existingMeta = (journey?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   449→   449→  const mergedSections = { ...existingSections }
   450→   450→  const updatedMeta = { ...existingMeta }
   451→   451→
   452→   452→  for (const [key, value] of Object.entries(generatedSections)) {
   453→   453→    const existing = mergedSections[key]
   454→   454→    const isEmpty = existing === null || existing === undefined ||
   455→   455→      (typeof existing === 'string' && existing.trim() === '') ||
   456→   456→      (typeof existing === 'object' && !Array.isArray(existing) && Object.keys(existing as object).length === 0)
   457→   457→
   458→   458→    if (isEmpty && typeof value === 'string' && value.trim()) {
   459→   459→      mergedSections[key] = value
   460→   460→      // Mark as generated from parcours
   461→   461→      updatedMeta[key] = buildSectionMetaForSync(existingMeta, key, value, 'parcours')
   462→   462→    }
   463→   463→  }
   464→   464→
   465→   465→  // 7. Recalculate bpScore
   466→   466→  const filledCount = Object.values(mergedSections).filter(isFilled).length
   467→   467→  const bpScore = Math.min(100, Math.round((filledCount / 24) * 100))
   468→   468→
   469→   469→  // 8. Update CreatorJourney (including bpSectionMeta)
   470→   470→  const savedJourney = await db.creatorJourney.upsert({
   471→   471→    where: { userId },
   472→   472→    create: {
   473→   473→      userId,
   474→   474→      bpSections: mergedSections as Prisma.InputJsonValue,
   475→   475→      bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   476→   476→      bpStatus: 'DRAFT',
   477→   477→      bpScore,
   478→   478→      bpGeneratedAt: new Date(),
   479→   479→    },
   480→   480→    update: {
   481→   481→      bpSections: mergedSections as Prisma.InputJsonValue,
   482→   482→      bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   483→   483→      bpStatus: 'DRAFT',
   484→   484→      bpScore,
   485→   485→      bpGeneratedAt: new Date(),
   486→   486→    },
   487→   487→  })
   488→   488→
   489→   489→  // Auto-snapshot after AI generation
   490→   490→  const userForTenant = await db.user.findUnique({ where: { id: userId }, select: { tenantId: true } })
   491→   491→  if (userForTenant?.tenantId) {
   492→   492→    maybeAutoSnapshot(userId, savedJourney.id, userForTenant.tenantId, mergedSections, {
   493→   493→      projectTitle: journey?.projectTitle,
   494→   494→      projectSector: journey?.projectSector,
   495→   495→    }, 'ai-generate')
   496→   496→  }
   497→   497→
   498→   498→  // 9. Count what was generated vs skipped
   499→   499→  const generatedKeys = Object.keys(generatedSections).filter(
   500→   500→    (k) => typeof generatedSections[k] === 'string' && generatedSections[k].trim(),
   501→   501→  )
   502→   502→  const skippedKeys = generatedKeys.filter(
   503→   503→    (k) => mergedSections[k] !== generatedSections[k],
   504→   504→  )
   505→   505→
   506→   506→  return success(
   507→   507→    {
   508→   508→      sections: mergedSections,
   509→   509→      generated: generatedKeys,
   510→   510→      skipped: skippedKeys,
   511→   511→      bpStatus: 'DRAFT',
   512→   512→      bpScore,
   513→   513→    },
   514→   514→    `Business plan généré à partir du parcours (${generatedKeys.length} sections)`,
   515→   515→  )
   516→   516→}
