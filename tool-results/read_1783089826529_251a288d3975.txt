   247→
   248→function isSectionFilled(key: string, value: unknown): boolean {
   249→  if (value === null || value === undefined) return false
   250→  if (typeof value === 'string') return value.trim().length > 0
   251→  if (Array.isArray(value)) return value.length > 0
   252→  if (typeof value === 'object') {
   253→    const obj = value as Record<string, unknown>
   254→    return Object.values(obj).some(
   255→      (v) => typeof v === 'string' && v.trim().length > 0 || typeof v === 'number' && v !== 0,
   256→    )
   257→  }
   258→  return false
   259→}

   288→export function BusinessPlanModule() {
   289→  const [sections, setSections] = useState<BpSections>(DEFAULT_SECTIONS)
   290→  const [activeTab, setActiveTab] = useState('presentation')
   291→  const [activeSection, setActiveSection] = useState<string | null>(null)
   292→  const [isLoading, setIsLoading] = useState(true)
   293→  const [isSaving, setIsSaving] = useState(false)
   294→  const [isGeneratingParcours, setIsGeneratingParcours] = useState(false)
   295→  const [isSyncing, setIsSyncing] = useState(false)
   296→  const [aiLoading, setAiLoading] = useState<string | null>(null)
   297→  const [previewOpen, setPreviewOpen] = useState(false)
   298→  const [exportPdfOpen, setExportPdfOpen] = useState(false)
   299→  const [sidebarCollapsed, setSidebarCollapsed] = useState(false)
   300→  const [projectContext, setProjectContext] = useState<Record<string, string | null> | null>(null)
   301→  const [pipelineStatus, setPipelineStatus] = useState<PipelineStatus>(DEFAULT_PIPELINE)
   302→
   303→
   304→  // ─── Load data on mount ──────────────────
   305→  useEffect(() => {
   306→    async function load() {
   307→      // Try localStorage first
   308→      const saved = localStorage.getItem('creapulse-bp')
   309→      if (saved) {
   310→        try {
   311→          const parsed = JSON.parse(saved) as BpSections
   312→          setSections(parsed)
   313→        } catch { /* ignore */ }
   314→      }
   315→
   316→      // Then try API
   317→      try {
   318→        const res = await fetch('/api/business-plan')
   319→        if (res.ok) {
   320→          const json = await res.json()
   321→          if (json.success && json.data?.sections) {
   322→            const apiSections = json.data.sections as BpSections
   323→            // Merge with defaults for any missing keys
   324→            const merged = { ...DEFAULT_SECTIONS, ...apiSections }
   325→            setSections(merged)
   326→            localStorage.setItem('creapulse-bp', JSON.stringify(merged))
   327→          }
   328→          if (json.data?.projectContext) {
   329→            setProjectContext(json.data.projectContext)
   330→          }
   331→        }
   332→      } catch { /* ignore */ }
   333→
   334→      setIsLoading(false)
   335→    }
   336→    load()
   337→  }, [])
   338→
   339→  // ─── Auto-save to localStorage ──────────
   340→  useEffect(() => {
   341→    if (!isLoading) {
   342→      localStorage.setItem('creapulse-bp', JSON.stringify(sections))
   343→    }
   344→  }, [sections, isLoading])
   345→
   346→  // ─── Completion tracking ────────────────
   347→  const completionInfo = useMemo(() => {
   348→    let filled = 0
   349→    ALL_SECTIONS.forEach((s) => {
   350→      if (isSectionFilled(s.id, sections[s.id])) filled++
   351→    })
   352→    return { filled, total: ALL_SECTIONS.length, percent: Math.round((filled / ALL_SECTIONS.length) * 100) }
   353→  }, [sections])
   354→
   355→  // ─── Update section helper ──────────────
   356→  const updateSection = useCallback((key: string, value: unknown) => {
   357→    setSections((prev) => ({ ...prev, [key]: value }))
   358→  }, [])
   359→
   360→  // ─── Save to API ────────────────────────
   361→  const handleSave = useCallback(async () => {
   362→    setIsSaving(true)
   363→    try {
   364→      const res = await fetch('/api/business-plan', {
   365→        method: 'PUT',
   366→        headers: { 'Content-Type': 'application/json' },
   367→        body: JSON.stringify({ sections }),
   368→      })
   369→      const json = await res.json()
   370→      if (json.success) {
   371→        toast.success(`Business plan sauvegardé — ${completionInfo.filled}/${completionInfo.total} sections remplies (${completionInfo.percent}%)`)
   372→      } else {
   373→        toast.error(json.error?.message || 'Erreur lors de la sauvegarde')
   374→      }
   375→    } catch {
   376→      toast.error('Erreur de connexion au serveur')
   377→    } finally {
   378→      setIsSaving(false)
   379→    }
   380→  }, [sections, completionInfo])
   381→
   382→  // ─── AI Suggestion ──────────────────────
   383→  const handleAiSuggest = useCallback(async (sectionDef: SectionDef) => {
   384→    setAiLoading(sectionDef.id)
   385→    try {
   386→      const contextStr = projectContext
   387→        ? `${projectContext.projectTitle || ''} ${projectContext.projectSector || ''} ${projectContext.targetAudience || ''}`
   388→        : ''
   389→
   390→      const res = await fetch('/api/business-plan', {
   391→        method: 'POST',
   392→        headers: { 'Content-Type': 'application/json' },
   393→        body: JSON.stringify({
   394→          action: 'ai-suggest',
   395→          sectionId: sectionDef.id,
   396→          sectionTitle: sectionDef.title,
   397→          existingContent: typeof sections[sectionDef.id] === 'string' ? (sections[sectionDef.id] as string) : undefined,
   398→          projectContext: contextStr || undefined,
   399→        }),
   400→      })
   401→
   402→      const json = await res.json()
   403→      if (json.success && json.data?.suggestion) {
   404→        if (sectionDef.type === 'swot') {
   405→          updateSection(sectionDef.id, {
   406→            ...(sections[sectionDef.id] as SwotData || { strengths: '', weaknesses: '', opportunities: '', threats: '' }),
   407→            strengths: json.data.suggestion,
   408→          })
   409→        } else {
   410→          updateSection(sectionDef.id, json.data.suggestion)
   411→        }
   412→        toast.success('Suggestion IA appliquée ! Modifiez le contenu si nécessaire.')
   413→      } else {
   414→        toast.error(json.error?.message || 'Erreur lors de la génération IA')
   415→      }
   416→    } catch {
   417→      toast.error('Erreur de connexion au serveur')
   418→    } finally {
   419→      setAiLoading(null)
   420→    }
   421→  }, [sections, projectContext, updateSection])
   422→
   423→  // ─── Generate from Parcours ─────────────
   424→  const handleGenerateFromParcours = useCallback(async () => {
   425→    setIsGeneratingParcours(true)
   426→    try {
   427→      const res = await fetch('/api/business-plan', {
   428→        method: 'POST',
   429→        headers: { 'Content-Type': 'application/json' },
   430→        body: JSON.stringify({ action: 'generate-from-parcours' }),
   431→      })
   432→      const json = await res.json()
   433→      if (json.success && json.data?.sections) {
   434→        setSections(prev => {
   435→          const merged = { ...DEFAULT_SECTIONS, ...prev, ...json.data.sections }
   436→          localStorage.setItem('creapulse-bp', JSON.stringify(merged))
   437→          return merged
   438→        })
   439→        setPipelineStatus(prev => ({ ...prev, parcours: true }))
   440→        toast.success('Business Plan généré depuis votre parcours entrepreneurial')
   441→      } else {
   442→        toast.error(json.error?.message || 'Erreur lors de la génération depuis le parcours')
   443→      }
   444→    } catch {
   445→      toast.error('Erreur de connexion au serveur')
   446→    } finally {
   447→      setIsGeneratingParcours(false)
   448→    }
   449→  }, [])
   450→
   451→  // ─── Sync Simulators ─────────────────────
   452→  const handleSyncSimulators = useCallback(async () => {
   453→    setIsSyncing(true)
   454→    try {
   455→      const res = await fetch('/api/business-plan', {
   456→        method: 'POST',
   457→        headers: { 'Content-Type': 'application/json' },
   458→        body: JSON.stringify({ action: 'sync-simulators' }),
   459→      })
   460→      const json = await res.json()
   461→      if (json.success && json.data) {
   462→        if (json.data.sections) {
   463→          setSections(prev => {
   464→            const merged = { ...DEFAULT_SECTIONS, ...prev, ...json.data.sections }
   465→            localStorage.setItem('creapulse-bp', JSON.stringify(merged))
   466→            return merged
   467→          })
   468→        }
   469→        if (json.data.syncedSources) {
   470→          setPipelineStatus(prev => ({
   471→            ...prev,
   472→            ...(json.data.syncedSources.marche && { marche: true }),
   473→            ...(json.data.syncedSources.financier && { financier: true }),
   474→            ...(json.data.syncedSources.juridique && { juridique: true }),
   475→            ...(json.data.syncedSources.creasim && { creasim: true }),
   476→          }))
   477→        }
   478→        const summary = json.data.summary || 'Données synchronisées'
