    69→    69→// ─── Helpers ─────────────────────────────────
    70→    70→
    71→    71→function isFilled(value: unknown): boolean {
    72→    72→  if (value === null || value === undefined) return false
    73→    73→  if (typeof value === 'string') return value.trim().length > 0
    74→    74→  if (Array.isArray(value)) return value.length > 0
    75→    75→  if (typeof value === 'object') {
    76→    76→    return Object.values(value).some((v) =>
    77→    77→      typeof v === 'string' && v.trim().length > 0 ||
    78→    78→      typeof v === 'number' && v !== 0 ||
    79→    79→      Array.isArray(v) && v.length > 0
    80→    80→    )
    81→    81→  }
    82→    82→  return false
    83→    83→}
    84→    84→
    85→    85→function computeWordCount(value: unknown): number {
    86→    86→  if (typeof value === 'string') return value.trim().split(/\s+/).filter(Boolean).length
    87→    87→  return 0
    88→    88→}
    89→    89→
    90→    90→const SYNC_SOURCES: DataSource[] = ['marche', 'financier', 'juridique', 'creasim', 'parcours']
    91→    91→
    92→    92→function buildSectionMetaForManual(
    93→    93→  existingMeta: BpSectionMeta,
    94→    94→  sectionId: string,
    95→    95→  content: unknown,
    96→    96→): SectionMetaEntry {
    97→    97→  const prev = existingMeta[sectionId]
    98→    98→  return {
    99→    99→    source: (prev && SYNC_SOURCES.includes(prev.source)) ? prev.source : 'manual',
   100→   100→    lastModified: new Date().toISOString(),
   101→   101→    manuallyEditedAt: new Date().toISOString(),
   102→   102→    syncedAt: prev?.syncedAt ?? null,
   103→   103→    wordCount: computeWordCount(content),
   104→   104→    version: (prev?.version ?? 0) + 1,
   105→   105→  }
   106→   106→}

   125→   125→/** Auto-create a snapshot if last one is older than 5 minutes (debounce) */
   126→   126→async function maybeAutoSnapshot(
   127→   127→  userId: string,
   128→   128→  journeyId: string,
   129→   129→  tenantId: string,
   130→   130→  bpSections: Record<string, unknown>,
   131→   131→  bpProjectContext: Record<string, unknown> | null,
   132→   132→  trigger: 'auto' | 'ai-generate' = 'auto',
   133→   133→) {
   134→   134→  try {
   135→   135→    const lastSnapshot = await db.bpSnapshot.findFirst({
   136→   136→      where: { userId },
   137→   137→      orderBy: { createdAt: 'desc' },
   138→   138→      select: { createdAt: true, version: true },
   139→   139→    })
   140→   140→    const fiveMinAgo = new Date(Date.now() - 5 * 60_000)
   141→   141→    if (lastSnapshot && lastSnapshot.createdAt > fiveMinAgo) return
   142→   142→
   143→   143→    const sectionCount = Object.values(bpSections).filter(isFilled).length
   144→   144→    const wordCount = Object.values(bpSections).reduce(
   145→   145→      (sum: number, v) => sum + computeWordCount(v), 0,
   146→   146→    )
   147→   147→    const version: number = (lastSnapshot?.version ?? 0) + 1
   148→   148→    const now = new Date()
   149→   149→    const label = `V${version} — ${now.toLocaleDateString('fr-FR', { day: 'numeric', month: 'short', year: 'numeric' })} ${now.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' })}`
   150→   150→
   151→   151→    await db.bpSnapshot.create({
   152→   152→      data: {
   153→   153→        userId,
   154→   154→        tenantId,
   155→   155→        creatorJourneyId: journeyId,
   156→   156→        bpSections: bpSections as Prisma.InputJsonValue,
   157→   157→        bpProjectContext: (bpProjectContext ?? {}) as Prisma.InputJsonValue,
   158→   158→        version,
   159→   159→        label,
   160→   160→        trigger,
   161→   161→        sectionCount,
   162→   162→        wordCount,
   163→   163→      },
   164→   164→    })
   165→   165→  } catch (err) {
   166→   166→    // Non-critical: log but don't fail the parent operation
   167→   167→    console.error('[AutoSnapshot] Failed to create auto-snapshot:', err)
   168→   168→  }
   169→   169→}

   171→   171→// ─── Helper: Auth from request ───────────────
   172→   172→
   173→   173→async function getAuth(request: NextRequest) {
   174→   174→  const cookieToken = request.cookies.get('session')?.value
   175→   175→  const headerToken = getTokenFromHeader(request)
   176→   176→  const token = cookieToken || headerToken
   177→   177→  if (!token) throw Object.assign(new Error('No session token found'), { code: 'UNAUTHORIZED' })
   178→   178→  return verifyToken(token)
   179→   179→}

   181→   181→// ─── Helper: Parse JSON from LLM response ───
   182→   182→
   183→   183→function parseJsonFromLLM(text: string): Record<string, string> | null {
   184→   184→  // Try direct JSON parse first
   185→   185→  try {
   186→   186→    const parsed = JSON.parse(text)
   187→   187→    if (typeof parsed === 'object' && parsed !== null) return parsed
   188→   188→  } catch {}
   189→   189→
   190→   190→  // Try matching JSON block
   191→   191→  const jsonMatch = text.match(/\{[\s\S]*\}/)
   192→   192→  if (jsonMatch) {
   193→   193→    try {
   194→   194→      const parsed = JSON.parse(jsonMatch[0])
   195→   195→      if (typeof parsed === 'object' && parsed !== null) return parsed
   196→   196→    } catch {}
   197→   197→  }
   198→   198→
   199→   199→  return null
   200→   200→}

   207→   207→
   208→   208→    const journey = await db.creatorJourney.findUnique({
   209→   209→      where: { userId: payload.userId },
   210→   210→      select: {
   211→   211→        bpSections: true,
   212→   212→        bpStatus: true,
   213→   213→        bpScore: true,
   214→   214→        bpGeneratedAt: true,
   215→   215→        bpValidatedAt: true,
   216→   216→        bpValidatedBy: true,
   217→   217→        projectTitle: true,
   218→   218→        projectSector: true,
   219→   219→        projectStage: true,

   221→   221→        valueProposition: true,
   222→   222→      },

   224→   224→
   225→   225→    if (!journey) {
   226→   226→      return success(
   227→   227→        {
   228→   228→          sections: {},
   229→   229→          bpStatus: 'NOT_STARTED',
   230→   230→          bpScore: null,
   231→   231→          projectContext: null,

   233→   233→        'Aucun business plan enregistré',
   234→   234→      )
   235→   235→    }
   236→   236→
   237→   237→    return success(
   238→   238→      {
   239→   239→        sections: journey.bpSections ?? {},
   240→   240→        bpStatus: journey.bpStatus,
   241→   241→        bpScore: journey.bpScore,
   242→   242→        bpGeneratedAt: journey.bpGeneratedAt,
   243→   243→        bpValidatedAt: journey.bpValidatedAt,
   244→   244→        projectContext: {
   245→   245→          projectTitle: journey.projectTitle,
   246→   246→          projectSector: journey.projectSector,
   247→   247→          projectStage: journey.projectStage,
   248→   248→          targetAudience: journey.targetAudience,
   249→   249→          valueProposition: journey.valueProposition,
   250→   250→        },
   251→   251→      },
   252→   252→      'Business plan chargé',
   253→   253→    )
   254→   254→  } catch (err) {
   255→   255→    if (err && typeof err === 'object' && 'code' in err) {
   256→   256→      const authErr = err as { code: string }
   257→   257→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   258→   258→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   259→   259→      }
   260→   260→    }
   261→   261→    return handleApiError(err)
   262→   262→  }
   263→   263→}

   267→   267→export async function PUT(request: NextRequest) {
   268→   268→  try {
   269→   269→    const payload = await getAuth(request)
   270→   270→    const body = await request.json()
   271→   271→
   272→   272→    // ── Single-section save: action=save-section ──
   273→   273→    if (body.action === 'save-section') {
   274→   274→      const { sectionId, content } = body
   275→   275→      if (!sectionId || content === undefined) {
   276→   276→        return Errors.validation([{ field: 'sectionId/content', message: 'sectionId et content sont requis' }])
   277→   277→      }
   278→   278→
   279→   279→      const existingJourney = await db.creatorJourney.findUnique({
   280→   280→        where: { userId: payload.userId },
   281→   281→        select: {
   282→   282→          id: true, bpSections: true, bpSectionMeta: true,
   283→   283→          bpStatus: true, bpScore: true, userId: true,
   284→   284→          user: { select: { tenantId: true } },
   285→   285→        },
   286→   286→      })
   287→   287→
   288→   288→      const existingSections = (existingJourney?.bpSections as Record<string, unknown>) ?? {}
   289→   289→      const existingMeta = (existingJourney?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   290→   290→
   291→   291→      // Update the section
   292→   292→      existingSections[sectionId] = content
   293→   293→
   294→   294→      // Update metadata for this section only
   295→   295→      existingMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)
   296→   296→
   297→   297→      // Recalculate bpScore
   298→   298→      const filledSections = Object.values(existingSections).filter(isFilled).length
   299→   299→      const bpScore = Math.min(100, Math.round((filledSections / 24) * 100))
   300→   300→
   301→   301→      let status = existingJourney?.bpStatus ?? 'IN_PROGRESS'
   302→   302→      if (filledSections === 0) status = 'NOT_STARTED'
   303→   303→      else if (filledSections === 24) status = 'DRAFT'
   304→   304→      else status = 'IN_PROGRESS'
   305→   305→
   306→   306→      const journey = await db.creatorJourney.upsert({
   307→   307→        where: { userId: payload.userId },
   308→   308→        create: {
   309→   309→          userId: payload.userId,
   310→   310→          bpSections: existingSections as Prisma.InputJsonValue,
   311→   311→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   312→   312→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   313→   313→          bpScore,
   314→   314→        },
   315→   315→        update: {
   316→   316→          bpSections: existingSections as Prisma.InputJsonValue,
   317→   317→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   318→   318→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   319→   319→          bpScore,
   320→   320→        },
   321→   321→      })
   322→   322→
   323→   323→      return success(
   324→   324→        {
   325→   325→          id: journey.id,
   326→   326→          sectionId,
   327→   327→          meta: existingMeta[sectionId],
   328→   328→          bpScore,
   329→   329→          filledSections,
   330→   330→          totalSections: 24,
   331→   331→        },
   332→   332→        'Section sauvegardée avec succès',
   333→   333→      )
   334→   334→    }
   335→   335→
   336→   336→    // ── Bulk save (existing logic) ──
   337→   337→    const parsed = saveBpSchema.safeParse(body)
   338→   338→
   339→   339→    if (!parsed.success) {
   340→   340→      return Errors.validation(
   341→   341→        parsed.error.issues.map((i) => ({
   342→   342→          field: i.path.join('.'),
   343→   343→          message: i.message,
   344→   344→        })),
   345→   345→      )
   346→   346→    }
   347→   347→
   348→   348→    const { sections, bpStatus } = parsed.data
   349→   349→
   350→   350→    // Read existing bpSectionMeta for metadata merge
   351→   351→    const previousJourney = await db.creatorJourney.findUnique({
   352→   352→      where: { userId: payload.userId },
   353→   353→      select: {
   354→   354→        bpSectionMeta: true, bpStatus: true, bpSections: true, userId: true,
   355→   355→        user: { select: { tenantId: true, firstName: true, lastName: true } },
   356→   356→      },
   357→   357→    })
   358→   358→    const existingMeta = (previousJourney?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   359→   359→    const updatedMeta: BpSectionMeta = { ...existingMeta }
   360→   360→    const previousStatus = previousJourney?.bpStatus ?? null
   361→   361→    const userTenantId = previousJourney?.user?.tenantId
   362→   362→
   363→   363→    // Calculate completion & update metadata for all provided sections
   364→   364→    const filledSections = sections
   365→   365→      ? Object.values(sections).filter(isFilled).length
   366→   366→      : 0
   367→   367→    const totalSections = 24
   368→   368→    const bpScore = Math.min(100, Math.round((filledSections / totalSections) * 100))
   369→   369→
   370→   370→    // Update bpSectionMeta for each non-empty section
   371→   371→    if (sections) {
   372→   372→      for (const [sectionId, content] of Object.entries(sections)) {
   373→   373→        if (isFilled(content)) {
   374→   374→          updatedMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)
   375→   375→        }
   376→   376→      }
   377→   377→    }
   378→   378→
   379→   379→    // Determine status — preserve SUBMITTED when explicitly requested
   380→   380→    let status: string = bpStatus ?? 'IN_PROGRESS'
   381→   381→    if (status !== 'SUBMITTED') {
   382→   382→      if (filledSections === 0) status = 'NOT_STARTED'
   383→   383→      else if (filledSections === totalSections) status = 'DRAFT'
   384→   384→      else status = 'IN_PROGRESS'
   385→   385→    }
   386→   386→
   387→   387→    // Upsert CreatorJourney bp fields (now includes bpSectionMeta)
   388→   388→    const journey = await db.creatorJourney.upsert({
   389→   389→      where: { userId: payload.userId },
   390→   390→      create: {
   391→   391→        userId: payload.userId,
   392→   392→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   393→   393→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   394→   394→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   395→   395→        bpScore,
   396→   396→      },
   397→   397→      update: {
   398→   398→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   399→   399→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   400→   400→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   401→   401→        bpScore,
   402→   402→        bpGeneratedAt: new Date(),
   403→   403→      },
   404→   404→    })
   405→   405→
   406→   406→    // ── Auto-snapshot: >3 sections changed, debounced to 5 min ──
   407→   407→    if (sections && Object.keys(sections).length > 3 && userTenantId) {
   408→   408→      const projectContext = previousJourney ? {
   409→   409→        projectTitle: (previousJourney as any).projectTitle ?? null,
   410→   410→        projectSector: (previousJourney as any).projectSector ?? null,
   411→   411→      } : null
   412→   412→      maybeAutoSnapshot(payload.userId, journey.id, userTenantId, sections ?? {}, projectContext, 'auto')
   413→   413→    }
   414→   414→
   415→   415→    // ── Fire-and-forget notifications on status change ──
   416→   416→    if (previousStatus && previousStatus !== status) {
   417→   417→      const statusLabel = status === 'NOT_STARTED' ? 'Non démarré'
   418→   418→        : status === 'IN_PROGRESS' ? 'En cours'
   419→   419→        : status === 'DRAFT' ? 'Brouillon'
   420→   420→        : status === 'SUBMITTED' ? 'Soumis'
   421→   421→        : status
   422→   422→
   423→   423→      createNotification({
   424→   424→        userId: payload.userId,
   425→   425→        title: 'Business plan mis à jour',
   426→   426→        content: `Votre business plan a été mis à jour — statut : ${statusLabel}`,
   427→   427→        type: 'INFO',
   428→   428→        link: '/bureau/business-plan',
   429→   429→      }).catch(() => {})
   430→   430→
   431→   431→      // If SUBMITTED, notify all COUNSELOR users in the same tenant
   432→   432→      if (status === 'SUBMITTED' && userTenantId) {
   433→   433→        const beneficiaryName = [previousJourney?.user?.firstName, previousJourney?.user?.lastName].filter(Boolean).join(' ') || 'Un bénéficiaire'
   434→   434→
   435→   435→        // Fetch counselors + send notifications (fire-and-forget)
   436→   436→        db.user.findMany({
   437→   437→          where: { tenantId: userTenantId, role: 'COUNSELOR', isActive: true },
   438→   438→          select: { id: true, email: true, firstName: true },
   439→   439→        }).then((counselors) => {
   440→   440→          const notificationPromises = counselors.map((c) =>
   441→   441→            createNotification({
   442→   442→              userId: c.id,
   443→   443→              title: 'Business plan à reviewer',
   444→   444→              content: `${beneficiaryName} a soumis son business plan pour review`,
   445→   445→              type: 'ACTION_REQUIRED',
   446→   446→              link: '/admin-centre/business-plan',
   447→   447→            })
   448→   448→          )
   449→   449→          return Promise.all(notificationPromises)
   450→   450→        }).catch(() => {})
   451→   451→
   452→   452→        // Send email to counselors (fire-and-forget)
   453→   453→        import('@/lib/email').then(({ sendBpSubmittedEmail }) => {
   454→   454→          db.user.findMany({
   455→   455→            where: { tenantId: userTenantId, role: 'COUNSELOR', isActive: true },
   456→   456→            select: { email: true, firstName: true },
   457→   457→          }).then((counselors) => {
   458→   458→            const beneficiaryFirstName = previousJourney?.user?.firstName || ''
   459→   459→            counselors.forEach((c) => {
   460→   460→              sendBpSubmittedEmail(beneficiaryFirstName, c.email, c.firstName || '').catch(() => {})
   461→   461→            })
   462→   462→          }).catch(() => {})
   463→   463→        }).catch(() => {})
   464→   464→
   465→   465→        log.info('BP submitted', { userId: payload.userId, beneficiaryName })
   466→   466→      }
   467→   467→    }

   492→   492→export async function POST(request: NextRequest) {
   493→   493→  try {
   494→   494→    const payload = await getAuth(request)
   495→   495→
   496→   496→    const body = await request.json()
   497→   497→    const parsed = bpActionSchema.safeParse(body)
   498→   498→
   499→   499→    if (!parsed.success) {
   500→   500→      return Errors.validation(
   501→   501→        parsed.error.issues.map((i) => ({
   502→   502→          field: i.path.join('.'),
   503→   503→          message: i.message,
   504→   504→        })),
   505→   505→      )
   506→   506→    }
   507→   507→
   508→   508→    switch (parsed.data.action) {
   509→   509→      case 'ai-suggest':
   510→   510→        return handleAiSuggest(parsed.data, payload.userId)
   511→   511→      case 'generate-from-parcours':
   512→   512→        return handleGenerateFromParcours(payload.userId)
   513→   513→      case 'sync-simulators':
   514→   514→        return handleSyncSimulators(payload.userId)
   515→   515→    }
   516→   516→  } catch (err) {
   517→   517→    if (err && typeof err === 'object' && 'code' in err) {
   518→   518→      const authErr = err as { code: string }
   519→   519→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   520→   520→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   521→   521→      }
   522→   522→    }
   523→   523→    return handleApiError(err)
   524→   524→  }
   525→   525→}

   856→   856→async function handleSyncSimulators(userId: string) {
   857→   857→  // 1. Fetch existing BP sections and metadata
   858→   858→  const journey = await db.creatorJourney.findUnique({
   859→   859→    where: { userId },
   860→   860→    select: { bpSections: true, bpSectionMeta: true },
   861→   861→  })
   862→   862→
   863→   863→  const existingSections = (journey?.bpSections as Record<string, unknown>) ?? {}
   864→   864→  const mergedSections = { ...existingSections }
   865→   865→  const sectionMeta = (journey?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   866→   866→
   867→   867→  // Track what was synced vs skipped (with source key for meta)
   868→   868→  const synced: { section: string; source: string; sourceKey?: DataSource }[] = []
   869→   869→  const skipped: { section: string; source: string; reason: string }[] = []
   870→   870→
   871→   871→  // Helper: only fill empty sections
   872→   872→  function fillSection(key: string, value: string, source: string, sourceKey?: DataSource) {
   873→   873→    const existing = mergedSections[key]
   874→   874→    const isEmpty = existing === null || existing === undefined ||
   875→   875→      (typeof existing === 'string' && existing.trim() === '')
   876→   876→
   877→   877→    if (isEmpty && value.trim()) {
   878→   878→      mergedSections[key] = value
   879→   879→      synced.push({ section: key, source, sourceKey })
   880→   880→      // Update section metadata with sync source
   881→   881→      if (sourceKey) {
   882→   882→        sectionMeta[key] = buildSectionMetaForSync(sectionMeta, key, value, sourceKey as 'marche' | 'juridique' | 'financier' | 'creasim' | 'parcours')
   883→   883→      }
   884→   884→    } else if (!isEmpty) {
   885→   885→      skipped.push({ section: key, source, reason: 'Déjà renseigné par l\'utilisateur' })
   886→   886→    }
   887→   887→  }
   888→   888→
   889→   889→  // 2. Fetch ALL simulator data in parallel
   890→   890→  const [
   891→   891→    marketAnalysis,
   892→   892→    financialForecast,
   893→   893→    creaSim,
   894→   894→    juridiqueAnalysis,
   895→   895→    businessModelCanvas,
