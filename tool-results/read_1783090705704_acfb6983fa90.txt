   280→function isSectionFilled(key: string, value: unknown): boolean {
   281→  if (value === null || value === undefined) return false
   282→  if (typeof value === 'string') return value.trim().length > 0
   283→  if (Array.isArray(value)) return value.length > 0
   284→  if (typeof value === 'object') {
   285→    const obj = value as Record<string, unknown>
   286→    return Object.values(obj).some(
   287→      (v) => typeof v === 'string' && v.trim().length > 0 || typeof v === 'number' && v !== 0,
   288→    )
   289→  }
   290→  return false
   291→}

   302→function formatRelativeTime(iso: string): string {
   303→  const now = Date.now()
   304→  const then = new Date(iso).getTime()
   305→  const diffMs = now - then
   306→  const seconds = Math.floor(diffMs / 1000)
   307→  const minutes = Math.floor(seconds / 60)
   308→  const hours = Math.floor(minutes / 60)
   309→  const days = Math.floor(hours / 24)
   310→
   311→  if (seconds < 60) return 'à l\'instant'
   312→  if (minutes < 60) return `il y a ${minutes} min`
   313→  if (hours < 24) return `il y a ${hours}h`
   314→
   315→  // Check if yesterday
   316→  const nowDate = new Date(now)
   317→  const thenDate = new Date(then)
   318→  if (
   319→    nowDate.getDate() - thenDate.getDate() === 1 &&
   320→    nowDate.getMonth() === thenDate.getMonth() &&
   321→    nowDate.getFullYear() === thenDate.getFullYear()
   322→  ) {
   323→    return 'hier'
   324→  }
   325→
   326→  return `il y a ${days}j`
   327→}
