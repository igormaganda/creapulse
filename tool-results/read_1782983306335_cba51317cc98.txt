    50→    50→async function getAuth(request: NextRequest) {
    51→    51→  const cookieToken = request.cookies.get('session')?.value
    52→    52→  const headerToken = getTokenFromHeader(request)
    53→    53→  const token = cookieToken || headerToken
    54→    54→  if (!token) throw Object.assign(new Error('No session token found'), { code: 'UNAUTHORIZED' })
    55→    55→  return verifyToken(token)
    56→    56→}

    60→    60→function parseJsonFromLLM(text: string): Record<string, string> | null {
    61→    61→  // Try direct JSON parse first
    62→    62→  try {
    63→    63→    const parsed = JSON.parse(text)
    64→    64→    if (typeof parsed === 'object' && parsed !== null) return parsed
    65→    65→  } catch {}
    66→    66→
    67→    67→  // Try matching JSON block
    68→    68→  const jsonMatch = text.match(/\{[\s\S]*\}/)
    69→    69→  if (jsonMatch) {
    70→    70→    try {
    71→    71→      const parsed = JSON.parse(jsonMatch[0])
    72→    72→      if (typeof parsed === 'object' && parsed !== null) return parsed
    73→    73→    } catch {}
    74→    74→  }
    75→    75→
    76→    76→  return null
    77→    77→}

   142→   142→// ─── PUT: Save business plan ─────────────────

   144→   144→export async function PUT(request: NextRequest) {
   145→   145→  try {
   146→   146→    const payload = await getAuth(request)
   147→   147→
   148→   148→    const body = await request.json()
   149→   149→    const parsed = saveBpSchema.safeParse(body)
   150→   150→
   151→   151→    if (!parsed.success) {
   152→   152→      return Errors.validation(
   153→   153→        parsed.error.issues.map((i) => ({
   154→   154→          field: i.path.join('.'),
   155→   155→          message: i.message,
   156→   156→        })),
   157→   157→      )
   158→   158→    }
   159→   159→
   160→   160→    const { sections, bpStatus } = parsed.data
   161→   161→
   162→   162→    // Calculate completion
   163→   163→    const filledSections = sections
   164→   164→      ? Object.values(sections).filter(
   165→   165→          (v) =>
   166→   166→            typeof v === 'string' && v.trim().length > 0 ||
   167→   167→            typeof v === 'object' && v !== null && !Array.isArray(v) && Object.keys(v).length > 0 ||
   168→   168→            Array.isArray(v) && v.length > 0,
   169→   169→        ).length
   170→   170→      : 0
   171→   171→    const totalSections = 24
   172→   172→    const bpScore = Math.min(100, Math.round((filledSections / totalSections) * 100))
   173→   173→
   174→   174→    // Determine status
   175→   175→    let status = (bpStatus ?? 'IN_PROGRESS') as string
   176→   176→    if (filledSections === 0) status = 'NOT_STARTED'
   177→   177→    else if (filledSections === totalSections) status = 'DRAFT'
   178→   178→    else status = 'IN_PROGRESS'
   179→   179→
   180→   180→    // Upsert CreatorJourney bp fields
   181→   181→    const journey = await db.creatorJourney.upsert({
   182→   182→      where: { userId: payload.userId },
   183→   183→      create: {
   184→   184→        userId: payload.userId,
   185→   185→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   186→   186→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   187→   187→        bpScore,
   188→   188→      },
   189→   189→      update: {
   190→   190→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   191→   191→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   192→   192→        bpScore,
   193→   193→        bpGeneratedAt: new Date(),
   194→   194→      },
   195→   195→    })
   196→   196→
   197→   197→    return success(
   198→   198→      {
   199→   199→        id: journey.id,
   200→   200→        bpStatus: status,
   201→   201→        bpScore,
   202→   202→        filledSections,
   203→   203→        totalSections,
   204→   204→      },
   205→   205→      'Business plan sauvegardé avec succès',
   206→   206→    )
   207→   207→  } catch (err) {
   208→   208→    if (err && typeof err === 'object' && 'code' in err) {
   209→   209→      const authErr = err as { code: string }
   210→   210→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   211→   211→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   212→   212→      }
   213→   213→    }
   214→   214→    return handleApiError(err)
   215→   215→  }
   216→   216→}

   220→   220→export async function POST(request: NextRequest) {
   221→   221→  try {
   222→   222→    const payload = await getAuth(request)
   223→   223→
   224→   224→    const body = await request.json()
   225→   225→    const parsed = bpActionSchema.safeParse(body)
   226→   226→
   227→   227→    if (!parsed.success) {
   228→   228→      return Errors.validation(
   229→   229→        parsed.error.issues.map((i) => ({
   230→   230→          field: i.path.join('.'),
   231→   231→          message: i.message,
   232→   232→        })),
   233→   233→      )
   234→   234→    }
   235→   235→
   236→   236→    switch (parsed.data.action) {
   237→   237→      case 'ai-suggest':
   238→   238→        return handleAiSuggest(parsed.data, payload.userId)
   239→   239→      case 'generate-from-parcours':
   240→   240→        return handleGenerateFromParcours(payload.userId)
   241→   241→      case 'sync-simulators':
   242→   242→        return handleSyncSimulators(payload.userId)
   243→   243→    }
   244→   244→  } catch (err) {
   245→   245→    if (err && typeof err === 'object' && 'code' in err) {
   246→   246→      const authErr = err as { code: string }
   247→   247→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   248→   248→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   249→   249→      }
   250→   250→    }
   251→   251→    return handleApiError(err)
   252→   252→  }
   253→   253→}

   288→   288→
   289→   289→  // Build system prompt
   290→   290→  const systemPrompt = `Tu es un expert en création d'entreprise et en rédaction de business plans. Tu aides des entrepreneurs francophones à rédiger un business plan professionnel et convaincant.
   291→   291→
   292→   292→RÈGLES IMPORTANTES :
   293→   293→- Réponds TOUJOURS en français
   294→   294→- Sois concis mais professionnel (200-400 mots par section)
   295→   295→- Utilise un ton formel et encourageant
   296→   296→- Structure le contenu avec des paragraphes clairs
   297→   297→- Adapte les conseils au contexte du projet fourni
   298→   298→- Si du contenu existe déjà, améliore-le et complète-le plutôt que de le remplacer
   299→   299→- Utilise des données concrètes et des exemples quand c'est pertinent

   301→   301→${context ? `CONTEXTE DU PROJET :\n${context}` : 'Aucun contexte de projet fourni. Génére un contenu générique adaptable.'}`
   302→   302→
   303→   303→  // Build user prompt
   304→   304→  let userPrompt = `Génère un contenu professionnel pour la section "${sectionTitle}" d'un business plan.`
   305→   305→
   306→   306→  if (existingContent && existingContent.trim()) {
   307→   307→    userPrompt += `\n\nContenu actuel de l'utilisateur :\n"${existingContent}"\n\nAméliore et complète ce contenu. Garde les informations importantes déjà présentes.`
   308→   308→  } else {
   309→   309→    userPrompt += `\n\nRédige un contenu complet et structuré pour cette section. L'utilisateur n'a encore rien écrit.`
   310→   310→  }
   311→   311→
   312→   312→  userPrompt += `\n\nSection ID: ${sectionId}`
   313→   313→
   314→   314→  // Call LLM via shared ZAI helper (never throws)

   316→   316→    { role: 'system', content: systemPrompt },
   317→   317→    { role: 'user', content: userPrompt },
   318→   318→  ], { temperature: 0.7, max_tokens: 1200 })
   319→   319→
   320→   320→  if (!result.success) {
   321→   321→    return aiUnavailableResponse(getZAIErrorMessage(result))

   326→   326→  return success(
   327→   327→    {
   328→   328→      sectionId,
   329→   329→      suggestion,
   330→   330→    },
   331→   331→    'Suggestion IA générée avec succès',
   332→   332→  )
   333→   333→}
