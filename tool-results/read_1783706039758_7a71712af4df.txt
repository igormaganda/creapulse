   362→/* ─── Module data reader (try/catch safe) ─── */
   363→
   364→function readModuleData(moduleCode: string): Record<string, string> {
   365→  const key = MODULE_STORAGE_KEYS[moduleCode]
   366→  if (!key) return {}
   367→
   368→  try {
   369→    if (typeof window === 'undefined') return {}
   370→    const raw = localStorage.getItem(key)
   371→    if (!raw) return {}
   372→
   373→    const data = JSON.parse(raw)
   374→    if (!data || typeof data !== 'object') return {}
   375→
   376→    // Flatten the data for the AI context (max 6 entries, truncate long values)
   377→    const flat: Record<string, string> = {}
   378→    let count = 0
   379→    for (const [k, v] of Object.entries(data)) {
   380→      if (count >= 6) break
   381→      if (v === null || v === undefined) continue
   382→      const strVal = typeof v === 'string' ? v : JSON.stringify(v)
   383→      flat[k] = strVal.length > 300 ? strVal.slice(0, 300) + '...' : strVal
   384→      count++
   385→    }
   386→    return flat
   387→  } catch {
   388→    return {}
   389→  }
   390→}

   392→/* ─── History persistence helpers ─── */

   394→function loadHistory(moduleId: string): ChatMessage[] {
   395→  try {
   396→    if (typeof window === 'undefined') return []
   397→    const key = `creapulse-ia-history-${moduleId}`
   398→    const raw = localStorage.getItem(key)
   399→    if (!raw) return []
   400→    const parsed = JSON.parse(raw) as Array<{ id: string; role: string; content: string; timestamp: string }>
   401→    return parsed.map((m) => ({
   402→      ...m,
   403→      timestamp: new Date(m.timestamp),
   404→      role: m.role as 'user' | 'assistant',
   405→    }))
   406→  } catch {
   407→    return []
   408→  }
   409→}

   411→function saveHistory(moduleId: string, messages: ChatMessage[]) {
   412→  try {
   413→    if (typeof window === 'undefined') return
   414→    const key = `creapulse-ia-history-${moduleId}`
   415→    // Keep max 50 messages
   416→    const toSave = messages.slice(-50).map((m) => ({
   417→      id: m.id,
   418→      role: m.role,
   419→      content: m.content,
   420→      timestamp: m.timestamp instanceof Date ? m.timestamp.toISOString() : m.timestamp,
   421→    }))
   422→    localStorage.setItem(key, JSON.stringify(toSave))
   423→  } catch {
   424→    // Silently fail
   425→  }
   426→}

   470→/* ─── Copy icon component ─── */
   471→
   472→function CopyButton({ content }: { content: string }) {
   473→  const [copied, setCopied] = useState(false)
   474→
   475→  const handleCopy = useCallback(() => {
   476→    navigator.clipboard.writeText(content).then(() => {
   477→      setCopied(true)
   478→      setTimeout(() => setCopied(false), 2000)
   479→    })
   480→  }, [content])
   481→
   482→  return (
   483→    <TooltipProvider delayDuration={300}>
   484→      <Tooltip>
   485→        <TooltipTrigger asChild>
   486→          <button
   487→            onClick={handleCopy}
   488→            className={cn(
   489→              'p-1 rounded-md transition-all duration-200 cursor-pointer',
   490→              'text-muted-foreground/0 group-hover/message:text-muted-foreground/70',
   491→              'hover:text-foreground hover:bg-muted/80',
   492→              copied && 'text-emerald-500',
   493→            )}
