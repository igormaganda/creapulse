   214→  const [data, setData] = useState<ELearningData>(() => {
   215→    try {
   216→      const saved = localStorage.getItem(STORAGE_KEY)
   217→      if (saved) return { ...DEFAULT_DATA, ...JSON.parse(saved) }
   218→    } catch { /* ignore */ }
   219→    return DEFAULT_DATA
   220→  })
   221→  const [searchQuery, setSearchQuery] = useState('')
   222→  const [catFilter, setCatFilter] = useState<string>('all')
   223→  const [diffFilter, setDiffFilter] = useState<string>('all')
   224→  const [activeFormation, setActiveFormation] = useState<Formation | null>(null)
   225→  const [selectedBadge, setSelectedBadge] = useState<BadgeItem | null>(null)
   226→
   227→  useEffect(() => { localStorage.setItem(STORAGE_KEY, JSON.stringify(data)) }, [data])
   228→
   229→  // ─── Helpers ───────────────────────────────
   230→  const getFormationProgress = useCallback((f: Formation) => {
   231→    const fd = data.formations[f.id]
   232→    if (!fd) return { percent: 0, completed: 0, total: f.lessons.length }
   233→    const completed = fd.lessonsCompleted.length
   234→    return { percent: Math.round((completed / f.lessons.length) * 100), completed, total: f.lessons.length }
   235→  }, [data.formations])
   236→
   237→  const isLessonCompleted = useCallback((formationId: string, lessonId: string) => {
   238→    return data.formations[formationId]?.lessonsCompleted?.includes(lessonId) || false
   239→  }, [data.formations])
   240→
   241→  const toggleLesson = useCallback((formationId: string, lessonId: string) => {
   242→    setData(prev => {
   243→      const fd = prev.formations[formationId] || { lessonsCompleted: [], started: false, completedAt: null }
   244→      const completed = fd.lessonsCompleted.includes(lessonId)
   245→        ? fd.lessonsCompleted.filter(id => id !== lessonId)
   246→        : [...fd.lessonsCompleted, lessonId]
   247→      const formation = FORMATIONS_CATALOG.find(f => f.id === formationId)
   248→      const allDone = formation && completed.length === formation.lessons.length
   249→      return {
   250→        ...prev,
   251→        formations: {
   252→          ...prev.formations,
   253→          [formationId]: {
   254→            ...fd,
   255→            lessonsCompleted: completed,
   256→            started: true,
   257→            completedAt: allDone ? new Date().toISOString() : null,
   258→          },
   259→        },
   260→        xp: prev.xp + (completed.includes(lessonId) ? 0 : 25),
   261→        lastStudyDate: new Date().toISOString().split('T')[0],
   262→      }
   263→    })
   264→  }, [])
   265→
   266→  // ─── Badges logic ─────────────────────────
   267→  const earnedBadges = useMemo(() => {
   268→    const completedFormations = FORMATIONS_CATALOG.filter(f => {
   269→      const fd = data.formations[f.id]
   270→      return fd && fd.completedAt
   271→    })
   272→    const strategyCompleted = completedFormations.filter(f => f.category === 'strategie').length
   273→    const financeCompleted = completedFormations.filter(f => f.category === 'finance').length
   274→    const marketingCompleted = completedFormations.filter(f => f.category === 'marketing').length
   275→
   276→    const updates: Record<string, { unlocked: boolean; unlockedAt: string | null }> = {}
   277→    const check = (id: string, condition: boolean) => {
   278→      const current = data.badges[id]
   279→      if (condition && !current?.unlocked) updates[id] = { unlocked: true, unlockedAt: new Date().toISOString() }
   280→      else if (!condition) updates[id] = { unlocked: false, unlockedAt: null }
   281→    }
   282→    check('explorateur', completedFormations.length >= 1)
   283→    check('stratege', strategyCompleted >= 3)
   284→    check('financier', financeCompleted >= 1)
   285→    check('marketeur', marketingCompleted >= 2)
   286→    check('champion', completedFormations.length >= 8)
   287→    check('expert', completedFormations.length >= 12)
   288→
   289→    const mergedBadges = { ...data.badges, ...updates }
   290→    const list = BADGES_CATALOG.map(b => ({ ...b, ...(mergedBadges[b.id] || { unlocked: false, unlockedAt: null }) }))
   291→    return { mergedBadges, list }
   292→  }, [data])
   293→
   294→  // Sync badge updates via render-time adjustment (avoids setState-in-effect)
   295→  const [prevBadges, setPrevBadges] = useState(data.badges)
   296→  if (earnedBadges.mergedBadges !== prevBadges &&
   297→      JSON.stringify(earnedBadges.mergedBadges) !== JSON.stringify(data.badges)) {
   298→    setPrevBadges(earnedBadges.mergedBadges)
   299→    setData(prev => ({ ...prev, badges: earnedBadges.mergedBadges }))
   300→  }
   301→
   302→  const unlockedCount = earnedBadges.list.filter(b => b.unlocked).length
   303→
   304→  // ─── Filtered formations ──────────────────
   305→  const filteredFormations = useMemo(() => {
   306→    let result = FORMATIONS_CATALOG
   307→    if (catFilter !== 'all') result = result.filter(f => f.category === catFilter)
   308→    if (diffFilter !== 'all') result = result.filter(f => f.difficulty === diffFilter)
   309→    if (searchQuery) {
   310→      const q = searchQuery.toLowerCase()
   311→      result = result.filter(f => f.title.toLowerCase().includes(q) || f.description.toLowerCase().includes(q))
   312→    }
   313→    return result
   314→  }, [searchQuery, catFilter, diffFilter])
   315→
   316→  const totalCompleted = FORMATIONS_CATALOG.filter(f => data.formations[f.id]?.completedAt).length
   317→
   318→  // ─── Main render ───────────────────────
   319→  return (
   320→    <motion.div initial={{ opacity: 0, y: 20 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.4 }} className="flex-1 overflow-y-auto">
   321→      {/* Header */}
   322→      <div className="flex flex-col sm:flex-row sm:items-center sm:justify-between gap-3 p-4 md:px-6 border-b bg-background">
   323→        <div className="flex items-center gap-3">
   324→          <div className="flex h-10 w-10 items-center justify-center rounded-xl bg-violet-500/10">
   325→            <BookOpen className="h-5 w-5 text-violet-600 dark:text-violet-400" />
   326→          </div>
   327→          <div>
   328→            <div className="flex items-center gap-2">
   329→              <h2 className="text-xl font-bold text-foreground">Formation & E-Learning</h2>
   330→              <Badge className="bg-violet-100 text-violet-700 dark:bg-violet-900/30 dark:text-violet-300 text-[10px] px-1.5">Nouveau</Badge>
   331→            </div>
   332→            <p className="text-xs text-muted-foreground">{totalCompleted}/{FORMATIONS_CATALOG.length} formations · {data.xp} XP · {unlockedCount}/{BADGES_CATALOG.length} badges</p>
   333→          </div>
   334→        </div>
   335→      </div>
   336→
   337→      <Tabs defaultValue="catalogue" className="flex-1">
   338→        <div className="border-b px-4 md:px-6">
   339→          <TabsList className="bg-transparent h-auto p-0 gap-6">
   340→            <TabsTrigger value="catalogue" className="rounded-none border-b-2 border-transparent data-[state=active]:border-violet-600 data-[state=active]:bg-transparent data-[state=active]:shadow-none pb-3 px-0">Catalogue</TabsTrigger>
   341→            <TabsTrigger value="parcours" className="rounded-none border-b-2 border-transparent data-[state=active]:border-violet-600 data-[state=active]:bg-transparent data-[state=active]:shadow-none pb-3 px-0">Mon Parcours</TabsTrigger>
   342→            <TabsTrigger value="badges" className="rounded-none border-b-2 border-transparent data-[state=active]:border-violet-600 data-[state=active]:bg-transparent data-[state=active]:shadow-none pb-3 px-0">Badges</TabsTrigger>
   343→          </TabsList>
   344→        </div>
   345→
   346→        {/* ── Tab: Catalogue ── */}
   347→        <TabsContent value="catalogue" className="p-4 md:p-6 space-y-4">
   348→          <div className="flex flex-col sm:flex-row gap-3">
   349→            <div className="relative flex-1">
   350→              <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
   351→              <Input value={searchQuery} onChange={e => setSearchQuery(e.target.value)} placeholder="Rechercher une formation..." className="pl-9" />
   352→            </div>
   353→            <Select value={catFilter} onValueChange={setCatFilter}>
   354→              <SelectTrigger className="w-full sm:w-36"><SelectValue placeholder="Catégorie" /></SelectTrigger>
   355→              <SelectContent>
   356→                <SelectItem value="all">Toutes</SelectItem>
   357→                {Object.entries(CATEGORY_LABELS).map(([k, v]) => <SelectItem key={k} value={k}>{v}</SelectItem>)}
   358→              </SelectContent>
   359→            </Select>
   360→            <Select value={diffFilter} onValueChange={setDiffFilter}>
   361→              <SelectTrigger className="w-full sm:w-36"><SelectValue placeholder="Niveau" /></SelectTrigger>
   362→              <SelectContent>
   363→                <SelectItem value="all">Tous</SelectItem>
   364→                {Object.entries(DIFFICULTY_CONFIG).map(([k, v]) => <SelectItem key={k} value={k}>{v.label}</SelectItem>)}
   365→              </SelectContent>
   366→            </Select>
   367→          </div>
   368→
   369→          <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
   370→            {filteredFormations.map(f => {
   371→              const prog = getFormationProgress(f)
   372→              const dc = DIFFICULTY_CONFIG[f.difficulty]
   373→              return (
   374→                <motion.div key={f.id} whileHover={{ y: -2 }} whileTap={{ scale: 0.98 }}>
   375→                  <Card className="h-full cursor-pointer hover:shadow-md transition-shadow" onClick={() => setActiveFormation(f)}>
   376→                    <CardHeader className="pb-3">
   377→                      <div className="flex items-start justify-between">
   378→                        <span className="text-2xl">{f.icon}</span>
   379→                        <Badge variant="outline" className={cn('text-[10px] shrink-0', dc.color)}>{dc.label}</Badge>
   380→                      </div>
   381→                      <CardTitle className="text-sm leading-tight mt-2">{f.title}</CardTitle>
   382→                      <CardDescription className="text-xs line-clamp-2">{f.description}</CardDescription>
   383→                    </CardHeader>
   384→                    <CardContent className="pt-0 space-y-3">
   385→                      <div className="flex items-center justify-between text-xs text-muted-foreground">
   386→                        <span className="flex items-center gap-1"><Clock className="h-3 w-3" />{f.duration}</span>
   387→                        <span className="flex items-center gap-1"><BookOpen className="h-3 w-3" />{f.lessons.length} leçons</span>
   388→                      </div>
   389→                      <div className="space-y-1">
   390→                        <div className="flex items-center justify-between text-xs">
   391→                          <span>{prog.completed}/{prog.total}</span>
   392→                          <span className={cn(prog.percent === 100 && 'text-emerald-600 font-medium')}>{prog.percent}%</span>
   393→                        </div>
   394→                        <Progress value={prog.percent} className="h-2" />
   395→                      </div>
   396→                      {prog.percent === 100 ? (
   397→                        <Button size="sm" variant="outline" className="w-full gap-1 text-emerald-600 border-emerald-300" disabled>
   398→                          <CheckCircle2 className="h-3.5 w-3.5" /> Terminée
   399→                        </Button>
   400→                      ) : prog.completed > 0 ? (
   401→                        <Button size="sm" className="w-full gap-1 bg-violet-600 hover:bg-violet-700 text-white">
   402→                          <Play className="h-3.5 w-3.5" /> Continuer
   403→                        </Button>
   404→                      ) : (
   405→                        <Button size="sm" variant="outline" className="w-full gap-1">
   406→                          <Play className="h-3.5 w-3.5" /> Commencer
   407→                        </Button>
   408→                      )}
   409→                    </CardContent>
   410→                  </Card>
   411→                </motion.div>
   412→              )
   413→            })}
   414→          </div>
   415→        </TabsContent>
   416→
   417→        {/* ── Tab: Mon Parcours ── */}
   418→        <TabsContent value="parcours" className="p-4 md:p-6 space-y-4">
   419→          {activeFormation ? (
   420→            <div>
   421→              <Button variant="ghost" size="sm" className="gap-1 mb-4" onClick={() => setActiveFormation(null)}>
   422→                <ArrowRight className="h-4 w-4 rotate-180" /> Retour au catalogue
   423→              </Button>
   424→              <Card>
   425→                <CardHeader>
   426→                  <div className="flex items-center gap-3">
   427→                    <span className="text-3xl">{activeFormation.icon}</span>
   428→                    <div>
   429→                      <CardTitle>{activeFormation.title}</CardTitle>
   430→                      <CardDescription>{activeFormation.description}</CardDescription>
   431→                    </div>
   432→                  </div>
   433→                  <div className="flex items-center gap-4 mt-2 text-xs text-muted-foreground">
   434→                    <Badge variant="outline" className={cn(DIFFICULTY_CONFIG[activeFormation.difficulty].color)}>{DIFFICULTY_CONFIG[activeFormation.difficulty].label}</Badge>
   435→                    <span className="flex items-center gap-1"><Clock className="h-3 w-3" />{activeFormation.duration}</span>
   436→                    <span>{getFormationProgress(activeFormation).percent}% complété</span>
   437→                  </div>
   438→                  <Progress value={getFormationProgress(activeFormation).percent} className="h-2 mt-2" />
   439→                </CardHeader>
   440→                <CardContent className="space-y-4">
   441→                  {activeFormation.lessons.map((lesson, idx) => {
   442→                    const done = isLessonCompleted(activeFormation.id, lesson.id)
   443→                    return (
   444→                      <motion.div key={lesson.id} initial={{ opacity: 0, y: 8 }} animate={{ opacity: 1, y: 0 }} transition={{ delay: idx * 0.05 }}>
   445→                        <Card className={cn('transition-colors', done && 'border-emerald-300 dark:border-emerald-800 bg-emerald-50/50 dark:bg-emerald-950/20')}>
   446→                          <CardContent className="p-4">
   447→                            <div className="flex items-start gap-3">
   448→                              <Checkbox checked={done} onCheckedChange={() => toggleLesson(activeFormation.id, lesson.id)} className="mt-0.5" />
   449→                              <div className="flex-1">
   450→                                <div className="flex items-center justify-between">
   451→                                  <p className={cn('text-sm font-semibold', done && 'line-through text-muted-foreground')}>{lesson.title}</p>
   452→                                  <span className="text-xs text-muted-foreground flex items-center gap-1"><Clock className="h-3 w-3" />{lesson.duration}</span>
   453→                                </div>
   454→                                <p className="text-xs text-muted-foreground mt-1 whitespace-pre-wrap leading-relaxed">{lesson.content}</p>
   455→                              </div>
   456→                            </div>
   457→                          </CardContent>
   458→                        </Card>
   459→                      </motion.div>
   460→                    )
   461→                  })}
   462→                </CardContent>
   463→              </Card>
   464→            </div>
   465→          ) : (
   466→            <>
   467→              <div className="grid grid-cols-3 gap-4 mb-6">
   468→                <Card><CardContent className="pt-4 pb-4 text-center"><p className="text-2xl font-bold text-violet-600">{totalCompleted}</p><p className="text-xs text-muted-foreground">Formations terminées</p></CardContent></Card>
   469→                <Card><CardContent className="pt-4 pb-4 text-center"><p className="text-2xl font-bold text-violet-600">{data.xp}</p><p className="text-xs text-muted-foreground">Points XP</p></CardContent></Card>
   470→                <Card><CardContent className="pt-4 pb-4 text-center"><p className="text-2xl font-bold text-violet-600">{unlockedCount}</p><p className="text-xs text-muted-foreground">Badges débloqués</p></CardContent></Card>
   471→              </div>
   472→              <p className="text-sm text-muted-foreground mb-4">Sélectionnez une formation dans le catalogue pour commencer votre apprentissage.</p>
   473→              {FORMATIONS_CATALOG.filter(f => data.formations[f.id]?.started).map(f => (
   474→                <Card key={f.id} className="mb-3 cursor-pointer hover:shadow-sm" onClick={() => setActiveFormation(f)}>
   475→                  <CardContent className="flex items-center gap-4 p-4">
   476→                    <span className="text-2xl">{f.icon}</span>
   477→                    <div className="flex-1 min-w-0">
   478→                      <p className="text-sm font-semibold">{f.title}</p>
   479→                      <Progress value={getFormationProgress(f).percent} className="h-1.5 mt-2" />
   480→                    </div>
   481→                    <div className="text-right shrink-0">
   482→                      <p className="text-sm font-bold text-violet-600">{getFormationProgress(f).percent}%</p>
   483→                      {data.formations[f.id]?.completedAt && <CheckCircle2 className="h-4 w-4 text-emerald-500 mx-auto mt-1" />}
   484→                    </div>
   485→                  </CardContent>
   486→                </Card>
   487→              ))}
   488→              {FORMATIONS_CATALOG.filter(f => data.formations[f.id]?.started).length === 0 && (
   489→                <Card className="p-8 text-center">
   490→                  <BookOpen className="h-10 w-10 mx-auto text-muted-foreground mb-3" />
   491→                  <p className="text-sm font-medium">Aucune formation commencée</p>
   492→                  <p className="text-xs text-muted-foreground mt-1">Explorez le catalogue pour commencer.</p>
   493→                </Card>
   494→              )}
   495→            </>
   496→          )}
   497→        </TabsContent>
   498→
   499→        {/* ── Tab: Badges ── */}
   500→        <TabsContent value="badges" className="p-4 md:p-6 space-y-6">
   501→          <div className="flex items-center gap-3 mb-2">
   502→            <div className="flex h-12 w-12 items-center justify-center rounded-xl bg-violet-500/10">
   503→              <Trophy className="h-6 w-6 text-violet-600 dark:text-violet-400" />
   504→            </div>
   505→            <div>
   506→              <p className="text-2xl font-bold">{unlockedCount}/{BADGES_CATALOG.length}</p>
   507→              <p className="text-xs text-muted-foreground">Badges débloqués</p>
   508→            </div>
   509→          </div>
   510→          <div className="grid grid-cols-2 sm:grid-cols-3 lg:grid-cols-4 gap-4">
   511→            {earnedBadges.list.map(badge => (
   512→              <motion.div key={badge.id} whileHover={{ scale: 1.03 }} whileTap={{ scale: 0.97 }}>
   513→                <Card className={cn('cursor-pointer hover:shadow-md transition-all h-full', badge.unlocked ? 'border-violet-300 dark:border-violet-800' : 'opacity-60')} onClick={() => setSelectedBadge(badge)}>
   514→                  <CardContent className="flex flex-col items-center text-center p-4">
   515→                    <div className={cn('flex h-16 w-16 items-center justify-center rounded-2xl text-3xl mb-3', badge.unlocked ? 'bg-violet-100 dark:bg-violet-900/30' : 'bg-muted')}>
   516→                      {badge.unlocked ? badge.emoji : <Lock className="h-6 w-6 text-muted-foreground" />}
   517→                    </div>
   518→                    <p className="text-sm font-semibold">{badge.name}</p>
   519→                    <p className="text-xs text-muted-foreground mt-1 line-clamp-2">{badge.description}</p>
   520→                    {badge.unlocked && badge.unlockedAt && (
   521→                      <p className="text-[10px] text-violet-500 mt-2">{new Date(badge.unlockedAt).toLocaleDateString('fr-FR')}</p>
   522→                    )}
   523→                  </CardContent>
   524→                </Card>
   525→              </motion.div>
   526→            ))}
   527→          </div>
   528→        </TabsContent>
   529→      </Tabs>
   530→
   531→      {/* Badge Detail Dialog */}
   532→      <Dialog open={!!selectedBadge} onOpenChange={() => setSelectedBadge(null)}>
   533→        <DialogContent className="max-w-sm">
   534→          <DialogHeader>
   535→            <div className="flex flex-col items-center">
   536→              <div className="flex h-20 w-20 items-center justify-center rounded-2xl text-5xl mb-3 bg-violet-100 dark:bg-violet-900/30">
   537→                {selectedBadge?.unlocked ? selectedBadge.emoji : <Lock className="h-8 w-8 text-muted-foreground" />}
   538→              </div>
   539→              <DialogTitle className="text-center">{selectedBadge?.name}</DialogTitle>
   540→            </div>
   541→          </DialogHeader>
   542→          <div className="text-center space-y-2">
   543→            <p className="text-sm text-muted-foreground">{selectedBadge?.description}</p>
   544→            <Separator />
   545→            <div>
   546→              <p className="text-xs font-medium text-muted-foreground">Comment le débloquer</p>
   547→              <p className="text-sm mt-1">{selectedBadge?.howToUnlock}</p>
   548→            </div>
   549→            {selectedBadge?.unlocked && selectedBadge?.unlockedAt && (
   550→              <Badge className="bg-violet-100 text-violet-700 dark:bg-violet-900/30 dark:text-violet-300">
   551→                <Award className="h-3 w-3 mr-1" />Débloqué le {new Date(selectedBadge.unlockedAt).toLocaleDateString('fr-FR')}
   552→              </Badge>
   553→            )}
   554→          </div>
   555→        </DialogContent>
   556→      </Dialog>
   557→    </motion.div>
   558→  )
   559→}
