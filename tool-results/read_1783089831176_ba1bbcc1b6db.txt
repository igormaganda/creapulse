   248→   248→function isSectionFilled(key: string, value: unknown): boolean {
   249→   249→  if (value === null || value === undefined) return false
   250→   250→  if (typeof value === 'string') return value.trim().length > 0
   251→   251→  if (Array.isArray(value)) return value.length > 0
   252→   252→  if (typeof value === 'object') {
   253→   253→    const obj = value as Record<string, unknown>
   254→   254→    return Object.values(obj).some(
   255→   255→      (v) => typeof v === 'string' && v.trim().length > 0 || typeof v === 'number' && v !== 0,
   256→   256→    )
   257→   257→  }
   258→   258→  return false
   259→   259→}

   288→   288→export function BusinessPlanModule() {

   291→   291→  const [activeSection, setActiveSection] = useState<string | null>(null)

   304→   304→  // ─── Load data on mount ──────────────────
   305→   305→  useEffect(() => {
   306→   306→    async function load() {
   307→   307→      // Try localStorage first
   308→   308→      const saved = localStorage.getItem('creapulse-bp')
   309→   309→      if (saved) {
   310→   310→        try {
   311→   311→          const parsed = JSON.parse(saved) as BpSections
   312→   312→          setSections(parsed)
   313→   313→        } catch { /* ignore */ }
   314→   314→      }
   315→   315→
   316→   316→      // Then try API
   317→   317→      try {
   318→   318→        const res = await fetch('/api/business-plan')
   319→   319→        if (res.ok) {
   320→   320→          const json = await res.json()
   321→   321→          if (json.success && json.data?.sections) {
   322→   322→            const apiSections = json.data.sections as BpSections
   323→   323→            // Merge with defaults for any missing keys
   324→   324→            const merged = { ...DEFAULT_SECTIONS, ...apiSections }
   325→   325→            setSections(merged)
   326→   326→            localStorage.setItem('creapulse-bp', JSON.stringify(merged))
   327→   327→          }
   328→   328→          if (json.data?.projectContext) {
   329→   329→            setProjectContext(json.data.projectContext)
   330→   330→          }
   331→   331→        }
   332→   332→      } catch { /* ignore */ }
   333→   333→
   334→   334→      setIsLoading(false)
   335→   335→    }
   336→   336→    load()
   337→   337→  }, [])
   338→   338→
   339→   339→  // ─── Auto-save to localStorage ──────────
   340→   340→  useEffect(() => {
   341→   341→    if (!isLoading) {
   342→   342→      localStorage.setItem('creapulse-bp', JSON.stringify(sections))
   343→   343→    }
   344→   344→  }, [sections, isLoading])
   345→   345→
   346→   346→  // ─── Completion tracking ────────────────
   347→   347→  const completionInfo = useMemo(() => {
   348→   348→    let filled = 0
   349→   349→    ALL_SECTIONS.forEach((s) => {
   350→   350→      if (isSectionFilled(s.id, sections[s.id])) filled++
   351→   351→    })
   352→   352→    return { filled, total: ALL_SECTIONS.length, percent: Math.round((filled / ALL_SECTIONS.length) * 100) }
   353→   353→  }, [sections])
   354→   354→
   355→   355→  // ─── Update section helper ──────────────
   356→   356→  const updateSection = useCallback((key: string, value: unknown) => {
   357→   357→    setSections((prev) => ({ ...prev, [key]: value }))
   358→   358→  }, [])
   359→   359→
   360→   360→  // ─── Save to API ────────────────────────
   361→   361→  const handleSave = useCallback(async () => {
   362→   362→    setIsSaving(true)
   363→   363→    try {
   364→   364→      const res = await fetch('/api/business-plan', {
   365→   365→        method: 'PUT',
   366→   366→        headers: { 'Content-Type': 'application/json' },
   367→   367→        body: JSON.stringify({ sections }),
   368→   368→      })
   369→   369→      const json = await res.json()
   370→   370→      if (json.success) {
   371→   371→        toast.success(`Business plan sauvegardé — ${completionInfo.filled}/${completionInfo.total} sections remplies (${completionInfo.percent}%)`)
   372→   372→      } else {
   373→   373→        toast.error(json.error?.message || 'Erreur lors de la sauvegarde')
   374→   374→      }
   375→   375→    } catch {
   376→   376→      toast.error('Erreur de connexion au serveur')
   377→   377→    } finally {
   378→   378→      setIsSaving(false)
   379→   379→    }
   380→   380→  }, [sections, completionInfo])
   381→   381→
   382→   382→  // ─── AI Suggestion ──────────────────────
   383→   383→  const handleAiSuggest = useCallback(async (sectionDef: SectionDef) => {
   384→   384→    setAiLoading(sectionDef.id)
   385→   385→    try {
   386→   386→      const contextStr = projectContext
   387→   387→        ? `${projectContext.projectTitle || ''} ${projectContext.projectSector || ''} ${projectContext.targetAudience || ''}`
   388→   388→        : ''
   389→   389→
   390→   390→      const res = await fetch('/api/business-plan', {
   391→   391→        method: 'POST',
   392→   392→        headers: { 'Content-Type': 'application/json' },
   393→   393→        body: JSON.stringify({
   394→   394→          action: 'ai-suggest',
   395→   395→          sectionId: sectionDef.id,
   396→   396→          sectionTitle: sectionDef.title,
   397→   397→          existingContent: typeof sections[sectionDef.id] === 'string' ? (sections[sectionDef.id] as string) : undefined,
   398→   398→          projectContext: contextStr || undefined,
   399→   399→        }),
   400→   400→      })
   401→   401→
