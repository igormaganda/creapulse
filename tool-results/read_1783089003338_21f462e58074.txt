    58→export interface SectionMetaEntry {
    59→  source: DataSource
    60→  lastModified: string
    61→  manuallyEditedAt: string | null
    62→  syncedAt: string | null
    63→  wordCount: number
    64→  version: number
    65→}
    66→
    67→export type BpSectionMeta = Record<string, SectionMetaEntry>

    69→// ─── Helpers ─────────────────────────────────
    70→
    71→function isFilled(value: unknown): boolean {
    72→  if (value === null || value === undefined) return false
    73→  if (typeof value === 'string') return value.trim().length > 0
    74→  if (Array.isArray(value)) return value.length > 0
    75→  if (typeof value === 'object') {
    76→    return Object.values(value).some((v) =>
    77→      typeof v === 'string' && v.trim().length > 0 ||
    78→      typeof v === 'number' && v !== 0 ||
    79→      Array.isArray(v) && v.length > 0
    80→    )
    81→  }
    82→  return false
    83→}
    84→
    85→function computeWordCount(value: unknown): number {
    86→  if (typeof value === 'string') return value.trim().split(/\s+/).filter(Boolean).length
    87→  return 0
    88→}
    89→
    90→const SYNC_SOURCES: DataSource[] = ['marche', 'financier', 'juridique', 'creasim', 'parcours']
    91→
    92→function buildSectionMetaForManual(
    93→  existingMeta: BpSectionMeta,
    94→  sectionId: string,
    95→  content: unknown,
    96→): SectionMetaEntry {
    97→  const prev = existingMeta[sectionId]
    98→  return {
    99→    source: (prev && SYNC_SOURCES.includes(prev.source)) ? prev.source : 'manual',
   100→    lastModified: new Date().toISOString(),
   101→    manuallyEditedAt: new Date().toISOString(),
   102→    syncedAt: prev?.syncedAt ?? null,
   103→    wordCount: computeWordCount(content),
   104→    version: (prev?.version ?? 0) + 1,
   105→  }
   106→}
   107→
   108→function buildSectionMetaForSync(
   109→  existingMeta: BpSectionMeta,
   110→  sectionId: string,
   111→  content: unknown,
   112→  source: 'marche' | 'financier' | 'juridique' | 'creasim' | 'parcours',
   113→): SectionMetaEntry {
   114→  const prev = existingMeta[sectionId]
   115→  return {
   116→    source,
   117→    lastModified: new Date().toISOString(),
   118→    manuallyEditedAt: prev?.manuallyEditedAt ?? null,
   119→    syncedAt: new Date().toISOString(),
   120→    wordCount: computeWordCount(content),
   121→    version: (prev?.version ?? 0) + 1,
   122→  }
   123→}

   125→/** Auto-create a snapshot if last one is older than 5 minutes (debounce) */
   126→async function maybeAutoSnapshot(
   127→  userId: string,
   128→  journeyId: string,
   129→  tenantId: string,
   130→  bpSections: Record<string, unknown>,
   131→  bpProjectContext: Record<string, unknown> | null,
   132→  trigger: 'auto' | 'ai-generate' = 'auto',
   133→) {
   134→  try {
   135→    const lastSnapshot = await db.bpSnapshot.findFirst({
   136→      where: { userId },
   137→      orderBy: { createdAt: 'desc' },
   138→      select: { createdAt: true, version: true },
   139→    })
   140→    const fiveMinAgo = new Date(Date.now() - 5 * 60_000)
   141→    if (lastSnapshot && lastSnapshot.createdAt > fiveMinAgo) return
   142→
   143→    const sectionCount = Object.values(bpSections).filter(isFilled).length
   144→    const wordCount = Object.values(bpSections).reduce(
   145→      (sum: number, v) => sum + computeWordCount(v), 0,
   146→    )
   147→    const version: number = (lastSnapshot?.version ?? 0) + 1
   148→    const now = new Date()
   149→    const label = `V${version} — ${now.toLocaleDateString('fr-FR', { day: 'numeric', month: 'short', year: 'numeric' })} ${now.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' })}`
   150→
   151→    await db.bpSnapshot.create({
   152→      data: {
   153→        userId,
   154→        tenantId,
   155→        creatorJourneyId: journeyId,
   156→        bpSections: bpSections as Prisma.InputJsonValue,
   157→        bpProjectContext: (bpProjectContext ?? {}) as Prisma.InputJsonValue,
   158→        version,
   159→        label,
   160→        trigger,
   161→        sectionCount,
   162→        wordCount,
   163→      },
   164→    })
   165→  } catch (err) {
   166→    // Non-critical: log but don't fail the parent operation
   167→    console.error('[AutoSnapshot] Failed to create auto-snapshot:', err)
   168→  }
   169→}

   173→async function getAuth(request: NextRequest) {
   174→  const cookieToken = request.cookies.get('session')?.value
   175→  const headerToken = getTokenFromHeader(request)
   176→  const token = cookieToken || headerToken
   177→  if (!token) throw Object.assign(new Error('No session token found'), { code: 'UNAUTHORIZED' })
   178→  return verifyToken(token)
   179→}

   181→// ─── Helper: Parse JSON from LLM response ───
   182→
   183→function parseJsonFromLLM(text: string): Record<string, string> | null {
   184→  // Try direct JSON parse first
   185→  try {
   186→    const parsed = JSON.parse(text)
   187→    if (typeof parsed === 'object' && parsed !== null) return parsed
   188→  } catch {}
   189→
   190→  // Try matching JSON block
   191→  const jsonMatch = text.match(/\{[\s\S]*\}/)
   192→  if (jsonMatch) {
   193→    try {
   194→      const parsed = JSON.parse(jsonMatch[0])
   195→      if (typeof parsed === 'object' && parsed !== null) return parsed
   196→    } catch {}
   197→  }
   198→
   199→  return null
   200→}

   267→export async function PUT(request: NextRequest) {
   268→  try {
   269→    const payload = await getAuth(request)
   270→    const body = await request.json()
   271→
   272→    // ── Single-section save: action=save-section ──
   273→    if (body.action === 'save-section') {
   274→      const { sectionId, content } = body
   275→      if (!sectionId || content === undefined) {
   276→        return Errors.validation([{ field: 'sectionId/content', message: 'sectionId et content sont requis' }])
   277→      }
   278→
   279→      const existingJourney = await db.creatorJourney.findUnique({
   280→        where: { userId: payload.userId },
   281→        select: {
   282→          id: true, bpSections: true, bpSectionMeta: true,
   283→          bpStatus: true, bpScore: true, userId: true,
   284→          user: { select: { tenantId: true } },
   285→        },
   286→      })
   287→
   288→      const existingSections = (existingJourney?.bpSections as Record<string, unknown>) ?? {}
   289→      const existingMeta = (existingJourney?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   290→
   291→      // Update the section
   292→      existingSections[sectionId] = content
   293→
   294→      // Update metadata for this section only
   295→      existingMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)
   296→
   297→      // Recalculate bpScore
   298→      const filledSections = Object.values(existingSections).filter(isFilled).length
   299→      const bpScore = Math.min(100, Math.round((filledSections / 24) * 100))
   300→
   301→      let status = existingJourney?.bpStatus ?? 'IN_PROGRESS'
   302→      if (filledSections === 0) status = 'NOT_STARTED'
   303→      else if (filledSections === 24) status = 'DRAFT'
   304→      else status = 'IN_PROGRESS'
   305→
   306→      const journey = await db.creatorJourney.upsert({
   307→        where: { userId: payload.userId },
   308→        create: {
   309→          userId: payload.userId,
   310→          bpSections: existingSections as Prisma.InputJsonValue,
   311→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   312→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   313→          bpScore,
   314→        },
   315→        update: {
   316→          bpSections: existingSections as Prisma.InputJsonValue,
   317→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   318→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   319→          bpScore,
   320→        },
   321→      })
   322→
   323→      return success(
   324→        {
   325→          id: journey.id,
   326→          sectionId,
   327→          meta: existingMeta[sectionId],
   328→          bpScore,
   329→          filledSections,
   330→          totalSections: 24,
   331→        },
   332→        'Section sauvegardée avec succès',
   333→      )
   334→    }
   335→
   336→    // ── Bulk save (existing logic) ──
   337→    const parsed = saveBpSchema.safeParse(body)
   338→
   339→    if (!parsed.success) {
   340→      return Errors.validation(
   341→        parsed.error.issues.map((i) => ({
   342→          field: i.path.join('.'),
   343→          message: i.message,
   344→        })),
   345→      )
   346→    }
   347→
   348→    const { sections, bpStatus } = parsed.data
   349→
   350→    // Read existing bpSectionMeta for metadata merge
   351→    const previousJourney = await db.creatorJourney.findUnique({
   352→      where: { userId: payload.userId },
   353→      select: {
   354→        bpSectionMeta: true, bpStatus: true, bpSections: true, userId: true,
   355→        user: { select: { tenantId: true, firstName: true, lastName: true } },
   356→      },
   357→    })
   358→    const existingMeta = (previousJourney?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   359→    const updatedMeta: BpSectionMeta = { ...existingMeta }
   360→    const previousStatus = previousJourney?.bpStatus ?? null
   361→    const userTenantId = previousJourney?.user?.tenantId
   362→
   363→    // Calculate completion & update metadata for all provided sections
   364→    const filledSections = sections
   365→      ? Object.values(sections).filter(isFilled).length
   366→      : 0
   367→    const totalSections = 24
   368→    const bpScore = Math.min(100, Math.round((filledSections / totalSections) * 100))
   369→
   370→    // Update bpSectionMeta for each non-empty section
   371→    if (sections) {
   372→      for (const [sectionId, content] of Object.entries(sections)) {
   373→        if (isFilled(content)) {
   374→          updatedMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)
   375→        }
   376→      }
   377→    }
   378→
   379→    // Determine status — preserve SUBMITTED when explicitly requested
   380→    let status: string = bpStatus ?? 'IN_PROGRESS'
   381→    if (status !== 'SUBMITTED') {
   382→      if (filledSections === 0) status = 'NOT_STARTED'
   383→      else if (filledSections === totalSections) status = 'DRAFT'
   384→      else status = 'IN_PROGRESS'
   385→    }
   386→
   387→    // Upsert CreatorJourney bp fields (now includes bpSectionMeta)
   388→    const journey = await db.creatorJourney.upsert({
   389→      where: { userId: payload.userId },
   390→      create: {
   391→        userId: payload.userId,
   392→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   393→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   394→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   395→        bpScore,
   396→      },
   397→      update: {
   398→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   399→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   400→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   401→        bpScore,
   402→        bpGeneratedAt: new Date(),
   403→      },
   404→    })
   405→
   406→    // ── Auto-snapshot: >3 sections changed, debounced to 5 min ──
   407→    if (sections && Object.keys(sections).length > 3 && userTenantId) {
   408→      const projectContext = previousJourney ? {
   409→        projectTitle: (previousJourney as any).projectTitle ?? null,
   410→        projectSector: (previousJourney as any).projectSector ?? null,
   411→      } : null
   412→      maybeAutoSnapshot(payload.userId, journey.id, userTenantId, sections ?? {}, projectContext, 'auto')
   413→    }
   414→
   415→    // ── Fire-and-forget notifications on status change ──
   416→    if (previousStatus && previousStatus !== status) {
   417→      const statusLabel = status === 'NOT_STARTED' ? 'Non démarré'
   418→        : status === 'IN_PROGRESS' ? 'En cours'
   419→        : status === 'DRAFT' ? 'Brouillon'
   420→        : status === 'SUBMITTED' ? 'Soumis'
   421→        : status
   422→
   423→      createNotification({
   424→        userId: payload.userId,
   425→        title: 'Business plan mis à jour',
   426→        content: `Votre business plan a été mis à jour — statut : ${statusLabel}`,
   427→        type: 'INFO',
   428→        link: '/bureau/business-plan',
   429→      }).catch(() => {})
   430→
   431→      // If SUBMITTED, notify all COUNSELOR users in the same tenant
   432→      if (status === 'SUBMITTED' && userTenantId) {
   433→        const beneficiaryName = [previousJourney?.user?.firstName, previousJourney?.user?.lastName].filter(Boolean).join(' ') || 'Un bénéficiaire'
   434→
   435→        // Fetch counselors + send notifications (fire-and-forget)
   436→        db.user.findMany({
   437→          where: { tenantId: userTenantId, role: 'COUNSELOR', isActive: true },
   438→          select: { id: true, email: true, firstName: true },
   439→        }).then((counselors) => {
   440→          const notificationPromises = counselors.map((c) =>
   441→            createNotification({
   442→              userId: c.id,
   443→              title: 'Business plan à reviewer',
   444→              content: `${beneficiaryName} a soumis son business plan pour review`,
   445→              type: 'ACTION_REQUIRED',
   446→              link: '/admin-centre/business-plan',
   447→            })
   448→          )
   449→          return Promise.all(notificationPromises)
   450→        }).catch(() => {})
   451→
   452→        // Send email to counselors (fire-and-forget)
   453→        import('@/lib/email').then(({ sendBpSubmittedEmail }) => {
   454→          db.user.findMany({
   455→            where: { tenantId: userTenantId, role: 'COUNSELOR', isActive: true },
   456→            select: { email: true, firstName: true },
   457→          }).then((counselors) => {
   458→            const beneficiaryFirstName = previousJourney?.user?.firstName || ''
   459→            counselors.forEach((c) => {
   460→              sendBpSubmittedEmail(beneficiaryFirstName, c.email, c.firstName || '').catch(() => {})
   461→            })
   462→          }).catch(() => {})
   463→        }).catch(() => {})
   464→
   465→        log.info('BP submitted', { userId: payload.userId, beneficiaryName })
   466→      }
   467→    }
   468→
   469→    return success(
   470→      {
   471→        id: journey.id,
   472→        bpStatus: status,
   473→        bpScore,
   474→        filledSections,
   475→        totalSections,
   476→      },
   477→      'Business plan sauvegardé avec succès',
   478→    )
   479→  } catch (err) {
   480→    if (err && typeof err === 'object' && 'code' in err) {
   481→      const authErr = err as { code: string }
   482→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   483→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   484→      }
   485→    }
   486→    return handleApiError(err)
   487→  }
   488→}

   490→// ─── POST: Multi-action handler ──────────────

   492→export async function POST(request: NextRequest) {
   493→  try {
   494→    const payload = await getAuth(request)
   495→
   496→    const body = await request.json()
   497→    const parsed = bpActionSchema.safeParse(body)
   498→
   499→    if (!parsed.success) {
   500→      return Errors.validation(
   501→        parsed.error.issues.map((i) => ({
   502→          field: i.path.join('.'),
   503→          message: i.message,
   504→        })),
   505→      )
   506→    }
   507→
   508→    switch (parsed.data.action) {
   509→      case 'ai-suggest':
   510→        return handleAiSuggest(parsed.data, payload.userId)
   511→      case 'generate-from-parcours':
   512→        return handleGenerateFromParcours(payload.userId)
   513→      case 'sync-simulators':
   514→        return handleSyncSimulators(payload.userId)
   515→    }
   516→  } catch (err) {
   517→    if (err && typeof err === 'object' && 'code' in err) {
   518→      const authErr = err as { code: string }
   519→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   520→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   521→      }
   522→    }
   523→    return handleApiError(err)
   524→  }
   525→}

   557→      context = parts.join('\n')

   559→  }
   560→
   561→  // Build system prompt
   562→  const systemPrompt = `Tu es un expert en création d'entreprise et en rédaction de business plans. Tu aides des entrepreneurs francophones à rédiger un business plan professionnel et convaincant.
   563→
   564→RÈGLES IMPORTANTES :
   565→- Réponds TOUJOURS en français

  1012→  if (creaSim) {
  1013→    const rentParts: string[] = []
  1014→    const fmtEur = (n: number) => n.toLocaleString('fr-FR')
  1015→    const fmtPct = (n: number) => `${n.toFixed(1)}%`
  1016→
  1017→    if (creaSim.monthlyRevenue) rentParts.push(`**CA mensuel** : ${fmtEur(creaSim.monthlyRevenue)} €`)
  1018→    if (creaSim.grossMarginRate) rentParts.push(`**Marge brute** : ${fmtPct(creaSim.grossMarginRate)}`)
  1019→    if (creaSim.netMarginRate) rentParts.push(`**Marge nette** : ${fmtPct(creaSim.netMarginRate)}`)
  1020→    if (creaSim.monthlyBreakeven) rentParts.push(`**Seuil de rentabilité mensuel** : ${fmtEur(creaSim.monthlyBreakeven)} €`)
  1021→    if (creaSim.breakevenMonths) rentParts.push(`**Point mort** : ${creaSim.breakevenMonths.toFixed(1)} mois`)
  1022→
  1023→    if (creaSim.profitability1Y != null) {
  1024→      rentParts.push(`**Rentabilité Année 1** : ${fmtEur(creaSim.profitability1Y)} € ${creaSim.profitability1Y >= 0 ? '✅' : '⚠️'}`)
  1025→    }
  1026→    if (creaSim.profitability2Y != null) {
  1027→      rentParts.push(`**Rentabilité Année 2** : ${fmtEur(creaSim.profitability2Y)} € ${creaSim.profitability2Y >= 0 ? '✅' : '⚠️'}`)
  1028→    }
  1029→    if (creaSim.profitability3Y != null) {
  1030→      rentParts.push(`**Rentabilité Année 3** : ${fmtEur(creaSim.profitability3Y)} € ${creaSim.profitability3Y >= 0 ? '✅' : '⚠️'}`)
  1031→    }
  1032→
  1033→    // Margin analysis
  1034→    if (creaSim.averageSellingPrice && creaSim.unitCost) {
  1035→      const marginPerUnit = creaSim.averageSellingPrice - creaSim.unitCost
  1036→      rentParts.push(`**Marge unitaire** : ${fmtEur(marginPerUnit)} € (PV ${fmtEur(creaSim.averageSellingPrice)} € - Coût ${fmtEur(creaSim.unitCost)} €)`)
  1037→    }
  1038→    if (creaSim.targetMarginRate) {
  1039→      rentParts.push(`**Objectif de marge** : ${fmtPct(creaSim.targetMarginRate)}`)
  1040→      if (creaSim.grossMarginRate) {
  1041→        const gap = creaSim.grossMarginRate - creaSim.targetMarginRate
  1042→        rentParts.push(`**Écart par rapport à l'objectif** : ${gap >= 0 ? '+' : ''}${fmtPct(gap)} ${gap >= 0 ? '✅ Atteint' : '⚠️ Non atteint'}`)
  1043→      }
  1044→    }
  1045→
  1046→    if (rentParts.length > 0) {
  1047→      fillSection('seuil-rentabilite', `## Analyse de rentabilité (CreaSim)\n\n${rentParts.join('\n\n')}`, 'CreaSim', 'creasim')
  1048→    } else {
  1049→      skipped.push({ section: 'seuil-rentabilite', source: 'CreaSim', reason: 'Données insuffisantes' })
  1050→    }
  1051→  } else {
  1052→    skipped.push({ section: 'seuil-rentabilite', source: 'CreaSim', reason: 'Simulation non complétée' })
  1053→  }
  1054→
  1055→  // 6. Map juridiqueAnalysis → statutJuridique
  1056→  if (juridiqueAnalysis) {
  1057→    const jurParts: string[] = []
  1058→    if (juridiqueAnalysis.recommendedStatus) jurParts.push(`**Statut recommandé** : ${juridiqueAnalysis.recommendedStatus}`)
  1059→    if (juridiqueAnalysis.fiscalRegime) jurParts.push(`**Régime fiscal** : ${juridiqueAnalysis.fiscalRegime}`)
  1060→    if (juridiqueAnalysis.legalStructure) jurParts.push(`**Structure juridique** : ${juridiqueAnalysis.legalStructure}`)
  1061→
  1062→    const charges = juridiqueAnalysis.socialCharges as unknown[]
  1063→    if (Array.isArray(charges) && charges.length > 0) {
  1064→      const chargeItems = charges.map((c: unknown) => {
  1065→        const charge = c as Record<string, string>
  1066→        return `- ${charge.name || charge.type || 'Charge'} : ${charge.amount || charge.rate || ''}`
  1067→      })
  1068→      jurParts.push(`**Charges sociales** :\n${chargeItems.join('\n')}`)
  1069→    }
  1070→
  1071→    if (jurParts.length > 0) {
  1072→      fillSection('statut-juridique', `## Statut juridique\n\n${jurParts.join('\n\n')}`, 'Analyse juridique', 'juridique')
  1073→    } else {
  1074→      skipped.push({ section: 'statut-juridique', source: 'Analyse juridique', reason: 'Données insuffisantes' })
  1075→    }
  1076→  } else {
  1077→    skipped.push({ section: 'statut-juridique', source: 'Analyse juridique', reason: 'Simulation non complétée' })
  1078→  }
  1079→
  1080→  // 7. Map businessModelCanvas → businessModelCanvas
  1081→  if (businessModelCanvas) {
  1082→    const bmcParts: string[] = []
  1083→    const bmcLabels: Record<string, string> = {
  1084→      partenairesCles: 'Partenaires clés',
  1085→      activitesCles: 'Activités clés',
  1086→      ressourcesCles: 'Ressources clés',
  1087→      propositionValeur: 'Proposition de valeur',
  1088→      relationsClients: 'Relations clients',
  1089→      canaux: 'Canaux de distribution',
  1090→      segmentsClients: 'Segments de clientèle',
  1091→      structureCouts: 'Structure des coûts',
  1092→      sourcesRevenus: 'Sources de revenus',
  1093→    }
  1094→
  1095→    let hasContent = false
  1096→    for (const [field, label] of Object.entries(bmcLabels)) {
  1097→      const value = businessModelCanvas[field as keyof typeof businessModelCanvas]
  1098→      if (value && typeof value === 'string' && value.trim()) {
  1099→        bmcParts.push(`### ${label}\n${value}`)
  1100→        hasContent = true
  1101→      }
  1102→    }
  1103→
  1104→    // BMC has no matching section in the BP — data is available in the BMC module
  1105→    skipped.push({ section: 'businessModelCanvas', source: 'Business Model Canvas', reason: 'Aucune section correspondante dans le BP' })
  1106→  } else {
  1107→    skipped.push({ section: 'businessModelCanvas', source: 'Business Model Canvas', reason: 'Simulation non complétée' })
  1108→  }
  1109→
  1110→  // 8. Recalculate bpScore and save (including bpSectionMeta)
  1111→  const filledCount = Object.values(mergedSections).filter(isFilled).length
  1112→  const totalSections = 24
  1113→  const bpScore = Math.min(100, Math.round((filledCount / totalSections) * 100))
  1114→
  1115→  await db.creatorJourney.upsert({
  1116→    where: { userId },
  1117→    create: {
  1118→      userId,
  1119→      bpSections: mergedSections as Prisma.InputJsonValue,
  1120→      bpSectionMeta: sectionMeta as unknown as Prisma.InputJsonValue,
  1121→      bpScore,
  1122→      bpGeneratedAt: new Date(),
  1123→    },
  1124→    update: {
  1125→      bpSections: mergedSections as Prisma.InputJsonValue,
  1126→      bpSectionMeta: sectionMeta as unknown as Prisma.InputJsonValue,
  1127→      bpScore,
  1128→      bpGeneratedAt: new Date(),
  1129→    },
  1130→  })
  1131→
  1132→  // Derive syncedSources from synced array
  1133→  const syncedSources: Record<string, boolean> = {}
  1134→  for (const item of synced) {
  1135→    const source = item.source as string
  1136→    if (source === 'Analyse de marché') syncedSources.marche = true
  1137→    if (source === 'Prévisions financières' || source === 'CreaSim') syncedSources.financier = true
  1138→    if (source === 'Analyse juridique') syncedSources.juridique = true
  1139→    if (source === 'CreaSim') syncedSources.creasim = true
  1140→  }
  1141→  const summary = `${synced.length} sections synchronisées depuis ${[...new Set(synced.map(s => s.source))].join(', ')}`
  1142→
  1143→  return success(
  1144→    {
  1145→      sections: mergedSections,
  1146→      synced,
  1147→      skipped,
  1148→      bpScore,
  1149→      totalSections,
  1150→      syncedSources,
  1151→      summary,
  1152→    },
  1153→    `Synchronisation terminée : ${synced.length} sections ajoutées, ${skipped.length} ignorées`,
  1154→  )
  1155→}
