   360→/* ─── Module data reader (try/catch safe) ─── */
   361→
   362→function readModuleData(moduleCode: string): Record<string, string> {
   363→  const key = MODULE_STORAGE_KEYS[moduleCode]
   364→  if (!key) return {}
   365→
   366→  try {
   367→    if (typeof window === 'undefined') return {}
   368→    const raw = localStorage.getItem(key)
   369→    if (!raw) return {}
   370→
   371→    const data = JSON.parse(raw)
   372→    if (!data || typeof data !== 'object') return {}
   373→
   374→    // Flatten the data for the AI context (max 6 entries, truncate long values)
   375→    const flat: Record<string, string> = {}
   376→    let count = 0
   377→    for (const [k, v] of Object.entries(data)) {
   378→      if (count >= 6) break
   379→      if (v === null || v === undefined) continue
   380→      const strVal = typeof v === 'string' ? v : JSON.stringify(v)
   381→      flat[k] = strVal.length > 300 ? strVal.slice(0, 300) + '...' : strVal
   382→      count++
   383→    }
   384→    return flat
   385→  } catch {
   386→    return {}
   387→  }
   388→}

   390→/* ─── History persistence helpers ─── */
   391→
   392→function loadHistory(moduleId: string): ChatMessage[] {
   393→  try {
   394→    if (typeof window === 'undefined') return []
   395→    const key = `creapulse-ia-history-${moduleId}`
   396→    const raw = localStorage.getItem(key)
   397→    if (!raw) return []
   398→    const parsed = JSON.parse(raw) as Array<{ id: string; role: string; content: string; timestamp: string }>
   399→    return parsed.map((m) => ({
   400→      ...m,
   401→      timestamp: new Date(m.timestamp),
   402→      role: m.role as 'user' | 'assistant',
   403→    }))
   404→  } catch {
   405→    return []
   406→  }
   407→}

   409→function saveHistory(moduleId: string, messages: ChatMessage[]) {
   410→  try {
   411→    if (typeof window === 'undefined') return
   412→    const key = `creapulse-ia-history-${moduleId}`
   413→    // Keep max 50 messages
   414→    const toSave = messages.slice(-50).map((m) => ({
   415→      id: m.id,
   416→      role: m.role,
   417→      content: m.content,
   418→      timestamp: m.timestamp instanceof Date ? m.timestamp.toISOString() : m.timestamp,
   419→    }))
   420→    localStorage.setItem(key, JSON.stringify(toSave))
   421→  } catch {
   422→    // Silently fail
   423→  }
   424→}

   468→/* ─── Copy icon component ─── */
   469→
   470→function CopyButton({ content }: { content: string }) {
   471→  const [copied, setCopied] = useState(false)
   472→
   473→  const handleCopy = useCallback(() => {
   474→    navigator.clipboard.writeText(content).then(() => {
   475→      setCopied(true)
   476→      setTimeout(() => setCopied(false), 2000)
   477→    })
   478→  }, [content])
   479→
   480→  return (
   481→    <TooltipProvider delayDuration={300}>
   482→      <Tooltip>
   483→        <TooltipTrigger asChild>
   484→          <button
   485→            onClick={handleCopy}
   486→            className={cn(
   487→              'p-1 rounded-md transition-all duration-200 cursor-pointer',
   488→              'text-muted-foreground/0 group-hover/message:text-muted-foreground/70',
   489→              'hover:text-foreground hover:bg-muted/80',
   490→              copied && 'text-emerald-500',
   491→            )}
   492→            aria-label="Copier le message"
   493→          >
