   281→function isSectionFilled(key: string, value: unknown): boolean {
   282→  if (value === null || value === undefined) return false
   283→  if (typeof value === 'string') return value.trim().length > 0
   284→  if (Array.isArray(value)) return value.length > 0
   285→  if (typeof value === 'object') {
   286→    const obj = value as Record<string, unknown>
   287→    return Object.values(obj).some(
   288→      (v) => typeof v === 'string' && v.trim().length > 0 || typeof v === 'number' && v !== 0,
   289→    )
   290→  }
   291→  return false
   292→}

   303→function formatRelativeTime(iso: string): string {
   304→  const now = Date.now()
   305→  const then = new Date(iso).getTime()
   306→  const diffMs = now - then
   307→  const seconds = Math.floor(diffMs / 1000)
   308→  const minutes = Math.floor(seconds / 60)
   309→  const hours = Math.floor(minutes / 60)
   310→  const days = Math.floor(hours / 24)
   311→
   312→  if (seconds < 60) return 'à l\'instant'
   313→  if (minutes < 60) return `il y a ${minutes} min`
   314→  if (hours < 24) return `il y a ${hours}h`
   315→  if (days < 7) return `il y a ${days}j`
   316→
   317→  return new Date(iso).toLocaleDateString('fr-FR', {
   318→    day: 'numeric',
   319→    month: 'short',
   320→    year: 'numeric',
   321→  })
   322→}
