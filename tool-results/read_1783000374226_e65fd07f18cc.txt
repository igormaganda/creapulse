   159→function loadFromStorage(): ProjectFormData | null {
   160→  if (typeof window === 'undefined') return null
   161→  try {
   162→    const raw = localStorage.getItem(STORAGE_KEY)
   163→    return raw ? JSON.parse(raw) : null
   164→  } catch {
   165→    return null
   166→  }
   167→}
   168→
   169→function saveToStorage(data: ProjectFormData) {
   170→  if (typeof window === 'undefined') return
   171→  try {
   172→    localStorage.setItem(STORAGE_KEY, JSON.stringify(data))
   173→  } catch {
   174→    // localStorage may be full
   175→  }
   176→}
   177→
   178→function clearStorage() {
   179→  if (typeof window === 'undefined') return
   180→  try {
   181→    localStorage.removeItem(STORAGE_KEY)
   182→  } catch {
   183→    // ignore
   184→  }
   185→}

   187→function calculateMaturityScore(data: ProjectFormData): number {
   188→  const fields: (string | string[])[] = [
   189→    data.projectTitle,
   190→    data.projectSector,
   191→    data.projectDescription,
   192→    data.projectStage,
   193→    data.primaryTarget,
   194→    data.secondaryTarget,
   195→    data.problemSolved,
   196→    data.competitiveAdvantage,
   197→    data.marketSize,
   198→    ...data.revenueSources,
   199→    data.revenueYear1,
   200→    data.revenueYear2,
   201→    data.revenueYear3,
   202→    data.initialInvestment,
   203→    data.financingNeed,
   204→    data.teamType,
   205→    data.associateCount,
   206→    data.motivation,
   207→    ...data.keyCompetencies,
   208→  ]
   209→
   210→  const filled = fields.filter(
   211→    (f) => (typeof f === 'string' ? f.trim().length > 0 : f.length > 0),
   212→  ).length
   213→  return Math.round((filled / fields.length) * 100)
   214→}

   258→function validateStep(step: number, data: ProjectFormData): StepErrors {
   259→  const errors: StepErrors = {}
   260→
   261→  switch (step) {
   262→    case 1:
   263→      if (!data.projectTitle.trim()) errors.projectTitle = 'Le nom du projet est requis'
   264→      if (!data.projectSector) errors.projectSector = 'Le secteur d\'activité est requis'
   265→      if (data.projectDescription.trim().length < 50)
   266→        errors.projectDescription = `Minimum 50 caractères (${data.projectDescription.trim().length}/50)`
   267→      if (!data.projectStage) errors.projectStage = 'Le stade actuel est requis'
   268→      break
   269→    case 2:
   270→      if (!data.primaryTarget.trim()) errors.primaryTarget = 'Le client cible principal est requis'
   271→      if (!data.problemSolved.trim()) errors.problemSolved = 'Décrivez le problème résolu par votre projet'
   272→      break
   273→    case 3:
   274→      // All optional
   275→      break
   276→    case 4:
   277→      if (!data.teamType) errors.teamType = 'Indiquez si la création est individuelle ou en équipe'
   278→      if (!data.motivation) errors.motivation = 'La motivation principale est requise'
   279→      break
   280→  }
   281→
   282→  return errors
   283→}

   287→function StepIndicator({ currentStep, completedSteps, onStepClick }: {
   288→  currentStep: number
   289→  completedSteps: Set<number>
   290→  onStepClick: (step: number) => void
   291→}) {
   292→  return (
   293→    <div className="mb-6 md:mb-8">
   294→      {/* Desktop step labels */}
   295→      <div className="hidden md:flex items-center justify-between relative">
   296→        {/* Connecting line */}
   297→        <div className="absolute top-5 left-0 right-0 h-0.5 bg-border z-0" />
   298→        <div
   299→          className="absolute top-5 left-0 h-0.5 bg-primary z-0 transition-all duration-500"
   300→          style={{ width: `${((currentStep - 1) / (STEPS.length - 1)) * 100}%` }}
   301→        />
   302→
   303→        {STEPS.map((step) => {
   304→          const Icon = step.icon
   305→          const isCompleted = completedSteps.has(step.id)
   306→          const isCurrent = step.id === currentStep
   307→          const isPast = step.id < currentStep
   308→
   309→          return (
   310→            <button
   311→              key={step.id}
   312→              onClick={() => {
   313→                if (isPast || isCompleted) onStepClick(step.id)
   314→              }}
   315→              className={cn(
   316→                'relative z-10 flex flex-col items-center gap-1.5 group cursor-pointer',
   317→                (!isPast && !isCompleted) && 'cursor-not-allowed'
   318→              )}
   319→              disabled={!isPast && !isCompleted}
   320→            >
   321→              <div
   322→                className={cn(
   323→                  'flex h-10 w-10 items-center justify-center rounded-full border-2 transition-all duration-300',
   324→                  isCompleted && 'border-primary bg-primary text-primary-foreground',
   325→                  isCurrent && !isCompleted && 'border-primary bg-background text-primary shadow-md shadow-primary/20',
   326→                  !isCurrent && !isCompleted && 'border-muted-foreground/30 bg-background text-muted-foreground/50',
   327→                )}
   328→              >
   329→                {isCompleted ? (
   330→                  <CheckCircle2 className="h-5 w-5" />

   332→                  <Icon className="h-4 w-4" />
   333→                )}
   334→              </div>
   335→              <span
   336→                className={cn(
   337→                  'text-xs font-medium transition-colors',
   338→                  isCurrent && 'text-primary',
   339→                  isCompleted && 'text-primary/80',
   340→                  !isCurrent && !isCompleted && 'text-muted-foreground/50',
   341→                )}
   342→              >
   343→                {step.label}
   344→              </span>
   345→            </button>
   346→          )
   347→        })}
   348→      </div>
   349→
   350→      {/* Mobile step dots */}
   351→      <div className="flex md:hidden items-center justify-center gap-2">
   352→        {STEPS.map((step) => {
   353→          const isCompleted = completedSteps.has(step.id)
   354→          const isCurrent = step.id === currentStep
   355→          return (
   356→            <button
   357→              key={step.id}
   358→              onClick={() => onStepClick(step.id)}
   359→              className={cn(
   360→                'flex h-8 w-8 items-center justify-center rounded-full text-xs font-bold transition-all',
   361→                isCompleted && 'bg-primary text-primary-foreground',
   362→                isCurrent && !isCompleted && 'border-2 border-primary text-primary',
   363→                !isCurrent && !isCompleted && 'border border-muted-foreground/30 text-muted-foreground/40',
   364→              )}
   365→            >
   366→              {step.id}
   367→            </button>
   368→          )
   369→        })}
   370→      </div>
   371→    </div>
   372→  )

   505→function StepMarche({ data, errors, onChange }: {
   506→  data: ProjectFormData
   507→  errors: StepErrors
   508→  onChange: <K extends keyof ProjectFormData>(field: K, value: ProjectFormData[K]) => void
   509→}) {
   510→  const token = useAuthStore((s) => s.token)
   511→
   512→  const [suggestions, setSuggestions] = useState<{
   513→    clients: string[]
   514→    problems: string[]
   515→    advantages: string[]
   516→  }>({ clients: [], problems: [], advantages: [] })
   517→
   518→  // Fetch AI suggestions when sector changes
   519→  useEffect(() => {
   520→    let cancelled = false
   521→    if (data.projectSector) {
   522→      fetch('/api/ai/suggestions', {
   523→        method: 'POST',
   524→        headers: {
   525→          'Content-Type': 'application/json',
   526→          ...(token ? { Authorization: `Bearer ${token}` } : {}),
   527→        },
   528→        credentials: 'include',
   529→        body: JSON.stringify({ sector: data.projectSector, activity: data.projectTitle, useAI: true }),
   530→      })
   531→        .then((r) => r.json())
   532→        .then((json) => {
   533→          if (!cancelled && json.success && json.data) setSuggestions(json.data)
   534→        })
   535→        .catch(() => {})
   536→    }
   537→    return () => { cancelled = true }
   538→  }, [data.projectSector, data.projectTitle, token])
   539→
   540→  // Handler: replace value for single-line, append for multi-line
   541→  const applySuggestion = (
   542→    field: 'primaryTarget' | 'problemSolved' | 'competitiveAdvantage',
   543→    value: string,
   544→  ) => {
   545→    if (field === 'primaryTarget') {
   546→      onChange(field, value)
   547→    } else {
   548→      const current = data[field]
   549→      onChange(field, current.trim() ? `${current.trim()}\n${value}` : value)
   550→    }
   551→  }
   552→
   553→  return (
   554→    <div className="space-y-6">
   555→      <div>
   556→        <h3 className="text-lg font-semibold text-foreground flex items-center gap-2">
   557→          <Target className="h-5 w-5 text-primary" />
   558→          Marché &amp; Clientèle
   559→        </h3>
   560→        <p className="text-sm text-muted-foreground mt-1">
   561→          Définissez votre marché cible et votre positionnement concurrentiel.
   562→        </p>
   563→      </div>
   564→
   565→      <Separator />
   566→
   567→      <div className="grid grid-cols-1 gap-5 md:grid-cols-2">
   568→        {/* Client cible principal */}
   569→        <div className="space-y-2">
   570→          <Label htmlFor="primaryTarget" className="text-sm font-medium">
   571→            Client cible principal <span className="text-red-500">*</span>
   572→          </Label>
   573→          <Input
   574→            id="primaryTarget"
   575→            placeholder="Ex: Entreprises PME du secteur tech"
   576→            value={data.primaryTarget}
   577→            onChange={(e) => onChange('primaryTarget', e.target.value)}
   578→            className={cn(errors.primaryTarget && 'border-red-500 focus-visible:ring-red-500')}
   579→          />
   580→          {errors.primaryTarget && (
   581→            <p className="text-xs text-red-500">{errors.primaryTarget}</p>
   582→          )}
   583→          {/* AI Suggestion chips — clients */}
   584→          {suggestions.clients.length > 0 && (
   585→            <div className="mt-1">
   586→              <p className="text-[11px] text-muted-foreground flex items-center gap-1 mb-1">
   587→                <Sparkles className="h-3 w-3 text-primary" />
   588→                Suggestions IA
   589→              </p>
   590→              <div className="flex flex-wrap gap-1.5">
   591→                {suggestions.clients.map((s, i) => (
   592→                  <button
   593→                    key={i}
   594→                    type="button"
   595→                    className="inline-flex items-center rounded-full border border-primary/30 bg-primary/5 px-2.5 py-0.5 text-[11px] text-primary hover:bg-primary/10 transition-colors"

  1112→              </p>

  1114→            </div>
  1115→          </div>
  1116→        </CardContent>
  1117→      </Card>
  1118→
  1119→      {/* Summary sections */}
  1120→      <div className="space-y-4">
  1121→        {summarySections.map((section) => {
  1122→          const Icon = section.icon
  1123→          return (
  1124→            <Card key={section.step} className="overflow-hidden">
  1125→              <CardHeader className="pb-3">
  1126→                <div className="flex items-center justify-between">
  1127→                  <CardTitle className="text-base flex items-center gap-2">
  1128→                    <Icon className="h-4 w-4 text-primary" />
  1129→                    {section.title}
  1130→                  </CardTitle>
  1131→                  <Button
  1132→                    variant="ghost"
  1133→                    size="sm"
  1134→                    onClick={() => onGoToStep(section.step)}
  1135→                    className="text-primary hover:text-primary/80 hover:bg-primary/5"
  1136→                  >
  1137→                    <Pencil className="h-3.5 w-3.5 mr-1" />
  1138→                    Modifier
  1139→                  </Button>
  1140→                </div>
  1141→              </CardHeader>
  1142→              <CardContent className="pt-0">
  1143→                <div className="grid grid-cols-1 gap-3 sm:grid-cols-2">
  1144→                  {section.fields.map((field) => (
  1145→                    <div key={field.label} className={cn('space-y-0.5', field.className)}>
  1146→                      <p className="text-xs font-medium text-muted-foreground">{field.label}</p>
  1147→                      <p className={cn(
  1148→                        'text-sm',
  1149→                        field.value === 'Non renseigné' || field.value === 'Non renseignée' || field.value === 'Non défini' || field.value === 'Non définie' || field.value === 'Aucune' || field.value === 'Aucun'
  1150→                          ? 'text-muted-foreground italic'
  1151→                          : 'text-foreground',
  1152→                      )}>
  1153→                        {field.value}
  1154→                      </p>
  1155→                    </div>
  1156→                  ))}
  1157→                </div>
  1158→              </CardContent>
  1159→            </Card>
  1160→          )
  1161→        })}
  1162→      </div>
  1163→    </div>
  1164→  )
  1165→}
