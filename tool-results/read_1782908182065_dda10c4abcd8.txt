   247→function isSectionFilled(key: string, value: unknown): boolean {
   248→  if (value === null || value === undefined) return false
   249→  if (typeof value === 'string') return value.trim().length > 0
   250→  if (Array.isArray(value)) return value.length > 0
   251→  if (typeof value === 'object') {
   252→    const obj = value as Record<string, unknown>
   253→    return Object.values(obj).some(
   254→      (v) => typeof v === 'string' && v.trim().length > 0 || typeof v === 'number' && v !== 0,
   255→    )
   256→  }
   257→  return false
   258→}

   265→    maximumFractionDigits: 0,
   266→  }).format(value)
   267→}

   287→export function BusinessPlanModule() {
   288→  const [sections, setSections] = useState<BpSections>(DEFAULT_SECTIONS)
   289→  const [activeTab, setActiveTab] = useState('presentation')
   290→  const [activeSection, setActiveSection] = useState<string | null>(null)
   291→  const [isLoading, setIsLoading] = useState(true)
   292→  const [isSaving, setIsSaving] = useState(false)
   293→  const [isGeneratingParcours, setIsGeneratingParcours] = useState(false)
   294→  const [isSyncing, setIsSyncing] = useState(false)
   295→  const [aiLoading, setAiLoading] = useState<string | null>(null)
   296→  const [previewOpen, setPreviewOpen] = useState(false)
   297→  const [exportPdfOpen, setExportPdfOpen] = useState(false)
   298→  const [sidebarCollapsed, setSidebarCollapsed] = useState(false)
   299→  const [projectContext, setProjectContext] = useState<Record<string, string | null> | null>(null)
   300→  const [pipelineStatus, setPipelineStatus] = useState<PipelineStatus>(DEFAULT_PIPELINE)
   301→
   302→
   303→  // ─── Load data on mount ──────────────────
   304→  useEffect(() => {
   305→    async function load() {
   306→      // Try localStorage first
   307→      const saved = localStorage.getItem('creapulse-bp')
   308→      if (saved) {
   309→        try {
   310→          const parsed = JSON.parse(saved) as BpSections
   311→          setSections(parsed)
   312→        } catch { /* ignore */ }
   313→      }
   314→
   315→      // Then try API
   316→      try {
   317→        const res = await fetch('/api/business-plan')
   318→        if (res.ok) {
   319→          const json = await res.json()
   320→          if (json.success && json.data?.sections) {
   321→            const apiSections = json.data.sections as BpSections
   322→            // Merge with defaults for any missing keys
   323→            const merged = { ...DEFAULT_SECTIONS, ...apiSections }
   324→            setSections(merged)
   325→            localStorage.setItem('creapulse-bp', JSON.stringify(merged))
   326→          }
   327→          if (json.data?.projectContext) {
   328→            setProjectContext(json.data.projectContext)
   329→          }
   330→        }
   331→      } catch { /* ignore */ }
   332→
   333→      setIsLoading(false)
   334→    }
   335→    load()
   336→  }, [])
   337→
   338→  // ─── Auto-save to localStorage ──────────
   339→  useEffect(() => {
   340→    if (!isLoading) {
   341→      localStorage.setItem('creapulse-bp', JSON.stringify(sections))
   342→    }
   343→  }, [sections, isLoading])
   344→
   345→  // ─── Completion tracking ────────────────
   346→  const completionInfo = useMemo(() => {
   347→    let filled = 0
   348→    ALL_SECTIONS.forEach((s) => {
   349→      if (isSectionFilled(s.id, sections[s.id])) filled++
   350→    })
   351→    return { filled, total: ALL_SECTIONS.length, percent: Math.round((filled / ALL_SECTIONS.length) * 100) }
   352→  }, [sections])
   353→
   354→  // ─── Update section helper ──────────────
   355→  const updateSection = useCallback((key: string, value: unknown) => {
   356→    setSections((prev) => ({ ...prev, [key]: value }))
   357→  }, [])
   358→
   359→  // ─── Save to API ────────────────────────
   360→  const handleSave = useCallback(async () => {
   361→    setIsSaving(true)
   362→    try {
   363→      const res = await fetch('/api/business-plan', {
   364→        method: 'PUT',
   365→        headers: { 'Content-Type': 'application/json' },
   366→        body: JSON.stringify({ sections }),
   367→      })
   368→      const json = await res.json()
   369→      if (json.success) {
   370→        toast.success(`Business plan sauvegardé — ${completionInfo.filled}/${completionInfo.total} sections remplies (${completionInfo.percent}%)`)
   371→      } else {
   372→        toast.error(json.error?.message || 'Erreur lors de la sauvegarde')
   373→      }
   374→    } catch {
   375→      toast.error('Erreur de connexion au serveur')
   376→    } finally {
   377→      setIsSaving(false)
   378→    }
   379→  }, [sections, completionInfo])
   380→
   381→  // ─── AI Suggestion ──────────────────────
   382→  const handleAiSuggest = useCallback(async (sectionDef: SectionDef) => {
   383→    setAiLoading(sectionDef.id)
   384→    try {
   385→      const contextStr = projectContext
   386→        ? `${projectContext.projectTitle || ''} ${projectContext.projectSector || ''} ${projectContext.targetAudience || ''}`
   387→        : ''
   388→
   389→      const res = await fetch('/api/business-plan', {
   390→        method: 'POST',
   391→        headers: { 'Content-Type': 'application/json' },
   392→        body: JSON.stringify({
   393→          action: 'ai-suggest',
   394→          sectionId: sectionDef.id,
   395→          sectionTitle: sectionDef.title,
   396→          existingContent: typeof sections[sectionDef.id] === 'string' ? (sections[sectionDef.id] as string) : undefined,
   397→          projectContext: contextStr || undefined,
   398→        }),
   399→      })
   400→
   401→      const json = await res.json()
   402→      if (json.success && json.data?.suggestion) {
   403→        if (sectionDef.type === 'swot') {
   404→          updateSection(sectionDef.id, {
   405→            ...(sections[sectionDef.id] as SwotData || { strengths: '', weaknesses: '', opportunities: '', threats: '' }),
   406→            strengths: json.data.suggestion,
   407→          })
   408→        } else {
   409→          updateSection(sectionDef.id, json.data.suggestion)
   410→        }
   411→        toast.success('Suggestion IA appliquée ! Modifiez le contenu si nécessaire.')
   412→      } else {
   413→        toast.error(json.error?.message || 'Erreur lors de la génération IA')
   414→      }
   415→    } catch {
   416→      toast.error('Erreur de connexion au serveur')
   417→    } finally {
   418→      setAiLoading(null)
   419→    }
   420→  }, [sections, projectContext, updateSection])
   421→
   422→  // ─── Generate from Parcours ─────────────
   423→  const handleGenerateFromParcours = useCallback(async () => {
   424→    setIsGeneratingParcours(true)
   425→    try {
   426→      const res = await fetch('/api/business-plan', {
   427→        method: 'POST',
   428→        headers: { 'Content-Type': 'application/json' },
   429→        body: JSON.stringify({ action: 'generate-from-parcours' }),
   430→      })
   431→      const json = await res.json()
   432→      if (json.success && json.data?.sections) {
   433→        setSections(prev => {
   434→          const merged = { ...DEFAULT_SECTIONS, ...prev, ...json.data.sections }
   435→          localStorage.setItem('creapulse-bp', JSON.stringify(merged))
   436→          return merged
   437→        })
   438→        setPipelineStatus(prev => ({ ...prev, parcours: true }))
   439→        toast.success('Business Plan généré depuis votre parcours entrepreneurial')
   440→      } else {
   441→        toast.error(json.error?.message || 'Erreur lors de la génération depuis le parcours')
   442→      }
   443→    } catch {
   444→      toast.error('Erreur de connexion au serveur')
   445→    } finally {
   446→      setIsGeneratingParcours(false)
   447→    }
   448→  }, [])
   449→
   450→  // ─── Sync Simulators ─────────────────────
   451→  const handleSyncSimulators = useCallback(async () => {
   452→    setIsSyncing(true)
   453→    try {
   454→      const res = await fetch('/api/business-plan', {
   455→        method: 'POST',
   456→        headers: { 'Content-Type': 'application/json' },
   457→        body: JSON.stringify({ action: 'sync-simulators' }),
   458→      })
   459→      const json = await res.json()
   460→      if (json.success && json.data) {
   461→        if (json.data.sections) {
   462→          setSections(prev => {
   463→            const merged = { ...DEFAULT_SECTIONS, ...prev, ...json.data.sections }
   464→            localStorage.setItem('creapulse-bp', JSON.stringify(merged))
   465→            return merged
   466→          })
   467→        }
   468→        if (json.data.syncedSources) {
   469→          setPipelineStatus(prev => ({
   470→            ...prev,
   471→            ...(json.data.syncedSources.marche && { marche: true }),
   472→            ...(json.data.syncedSources.financier && { financier: true }),
   473→            ...(json.data.syncedSources.juridique && { juridique: true }),
   474→            ...(json.data.syncedSources.creasim && { creasim: true }),
   475→          }))
   476→        }
   477→        const summary = json.data.summary || 'Données synchronisées'
   478→        toast.success(`Simulateurs synchronisés — ${summary}`)
   479→      } else {
   480→        toast.error(json.error?.message || 'Erreur lors de la synchronisation')
