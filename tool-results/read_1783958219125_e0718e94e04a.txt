    58→export interface SectionMetaEntry {
    59→  source: DataSource
    60→  lastModified: string
    61→  manuallyEditedAt: string | null
    62→  syncedAt: string | null
    63→  wordCount: number
    64→  version: number
    65→}
    66→
    67→export type BpSectionMeta = Record<string, SectionMetaEntry>

    69→// ─── Helpers ─────────────────────────────────
    70→
    71→function isFilled(value: unknown): boolean {
    72→  if (value === null || value === undefined) return false
    73→  if (typeof value === 'string') return value.trim().length > 0
    74→  if (Array.isArray(value)) return value.length > 0
    75→  if (typeof value === 'object') {
    76→    return Object.values(value).some((v) =>
    77→      typeof v === 'string' && v.trim().length > 0 ||
    78→      typeof v === 'number' && v !== 0 ||
    79→      Array.isArray(v) && v.length > 0
    80→    )
    81→  }
    82→  return false
    83→}
    84→
    85→function computeWordCount(value: unknown): number {
    86→  if (typeof value === 'string') return value.trim().split(/\s+/).filter(Boolean).length
    87→  return 0
    88→}

    90→const SYNC_SOURCES: DataSource[] = ['marche', 'financier', 'juridique', 'creasim', 'parcours']

    92→function buildSectionMetaForManual(
    93→  existingMeta: BpSectionMeta,
    94→  sectionId: string,
    95→  content: unknown,
    96→): SectionMetaEntry {
    97→  const prev = existingMeta[sectionId]
    98→  return {
    99→    source: 'manual' as DataSource,
   100→    lastModified: new Date().toISOString(),
   101→    manuallyEditedAt: new Date().toISOString(),
   102→    syncedAt: prev?.syncedAt ?? null,
   103→    wordCount: computeWordCount(content),
   104→    version: (prev?.version ?? 0) + 1,
   105→  }
   106→}
   107→
   108→function buildSectionMetaForSync(
   109→  existingMeta: BpSectionMeta,
   110→  sectionId: string,
   111→  content: unknown,
   112→  source: 'marche' | 'financier' | 'juridique' | 'creasim' | 'parcours',
   113→): SectionMetaEntry {
   114→  const prev = existingMeta[sectionId]
   115→  return {
   116→    source,
   117→    lastModified: new Date().toISOString(),
   118→    manuallyEditedAt: prev?.manuallyEditedAt ?? null,
   119→    syncedAt: new Date().toISOString(),
   120→    wordCount: computeWordCount(content),
   121→    version: (prev?.version ?? 0) + 1,
   122→  }
   123→}

   125→/** Auto-create a snapshot if last one is older than 5 minutes (debounce) */
   126→async function maybeAutoSnapshot(
   127→  userId: string,
   128→  journeyId: string,
   129→  tenantId: string,
   130→  bpSections: Record<string, unknown>,
   131→  bpProjectContext: Record<string, unknown> | null,
   132→  trigger: 'auto' | 'ai-generate' = 'auto',
   133→) {
   134→  try {
   135→    const lastSnapshot = await db.bpSnapshot.findFirst({
   136→      where: { userId },
   137→      orderBy: { createdAt: 'desc' },
   138→      select: { createdAt: true, version: true },
   139→    })
   140→    const fiveMinAgo = new Date(Date.now() - 5 * 60_000)
   141→    if (lastSnapshot && lastSnapshot.createdAt > fiveMinAgo) return
   142→
   143→    const sectionCount = Object.values(bpSections).filter(isFilled).length
   144→    const wordCount = Object.values(bpSections).reduce(
   145→      (sum: number, v) => sum + computeWordCount(v), 0,
   146→    )
   147→    const version: number = (lastSnapshot?.version ?? 0) + 1
   148→    const now = new Date()
   149→    const label = `V${version} — ${now.toLocaleDateString('fr-FR', { day: 'numeric', month: 'short', year: 'numeric' })} ${now.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' })}`
   150→
   151→    await db.bpSnapshot.create({
   152→      data: {
   153→        userId,
   154→        tenantId,
   155→        creatorJourneyId: journeyId,
   156→        bpSections: bpSections as Prisma.InputJsonValue,
   157→        bpProjectContext: (bpProjectContext ?? {}) as Prisma.InputJsonValue,
   158→        version,
   159→        label,
   160→        trigger,
   161→        sectionCount,
   162→        wordCount,
   163→      },
   164→    })
   165→  } catch (err) {
   166→    // Non-critical: log but don't fail the parent operation
   167→    console.error('[AutoSnapshot] Failed to create auto-snapshot:', err)
   168→  }
   169→}

   173→async function getAuth(request: NextRequest) {
   174→  const cookieToken = request.cookies.get('session')?.value
   175→  const headerToken = getTokenFromHeader(request)
   176→  const token = cookieToken || headerToken
   177→  if (!token) throw Object.assign(new Error('No session token found'), { code: 'UNAUTHORIZED' })
   178→  return verifyToken(token)
   179→}

   181→// ─── Helper: Parse JSON from LLM response ───
   182→
   183→function parseJsonFromLLM(text: string): Record<string, string> | null {
   184→  // Try direct JSON parse first
   185→  try {
   186→    const parsed = JSON.parse(text)
   187→    if (typeof parsed === 'object' && parsed !== null) return parsed
   188→  } catch {}
   189→
   190→  // Try matching JSON block
   191→  const jsonMatch = text.match(/\{[\s\S]*\}/)
   192→  if (jsonMatch) {
   193→    try {
   194→      const parsed = JSON.parse(jsonMatch[0])
   195→      if (typeof parsed === 'object' && parsed !== null) return parsed
   196→    } catch {}
   197→  }
   198→
   199→  return null
   200→}

   276→async function handleGenerateFromParcours(userId: string) {

   291→      where: { userId },
   292→      select: { skills: true, educationLevel: true, employmentStatus: true },

   295→        projectTitle: true,
   296→        projectDescription: true,
   297→        projectSector: true,
   298→        projectStage: true,
   299→        targetAudience: true,
   300→        valueProposition: true,
   301→        creationMotivation: true,

   303→        bpSections: true,
   304→        bpSectionMeta: true,
   305→      },
   306→    }),

   308→      where: { userId },
   309→      select: { profileType: true, score: true, isDominant: true },
   310→      orderBy: { score: 'desc' },

   313→      where: { userId },

   318→      where: { userId },
   319→      select: { moduleCode: true, score: true, maxScore: true, feedback: true },

   324→    return Errors.notFound('Utilisateur')

   326→

   382→  if (moduleResults.length > 0) {
   383→    contextParts.push(`\n## MODULES COMPLÉTÉS`)
   384→    for (const m of moduleResults) {
   385→      contextParts.push(`- ${m.moduleCode} : ${m.score}/${m.maxScore}`)
   386→    }
   387→  }

   393→
   394→MISSION : À partir des données du parcours entrepreneur d'un porteur de projet (profil, tests de personnalité, compétences, vision, description de projet), tu dois générer les premières sections de son Business Plan.

   397→- Réponds TOUJOURS en français

   401→- Structure chaque section avec des paragraphes clairs

   405→FORMAT DE RÉPONSE :

   407→{
   408→  "resumeOperationnel": "...",
   409→  "presentationPorteur": "...",
   410→  "descriptionProjet": "...",
   411→  "conceptProposition": "...",
   412→  "clienteleCible": "...",
   413→  "positionnement": "...",
   414→  "equipeProjet": "...",
   415→  "objectifs": "..."
   416→}
   417→
   418→- resumeOperationnel : Résumé opérationnel du projet (vue d'ensemble)
   419→- presentationPorteur : Présentation du porteur de projet (profil, compétences, motivations)
   420→- descriptionProjet : Description détaillée du projet
   421→- conceptProposition : Concept et proposition de valeur
   422→- clienteleCible : Description de la clientèle cible
   423→- positionnement : Positionnement sur le marché
   424→- equipeProjet : Description de l'équipe et des compétences (basé sur RIASEC+Kiviat)
   425→- objectifs : Objectifs à court, moyen et long terme (basé sur la vision)`

   429→  // 4. Call LLM via shared ZAI helper (never throws)
   430→  const result = await callZAI([
   431→    { role: 'system', content: systemPrompt },
   432→    { role: 'user', content: userPrompt },
   433→  ], { temperature: 0.7, max_tokens: 3000 })
   434→
   435→  if (!result.success) {
   436→    return aiUnavailableResponse(getZAIErrorMessage(result))
   437→  }
   438→
   439→  // 5. Parse JSON response
   440→  const generatedSections = parseJsonFromLLM(result.content)
   441→
   442→  if (!generatedSections) {
   443→    return aiErrorResponse("La réponse IA n'a pas pu être interprétée. Veuillez réessayer.")
   444→  }
   445→
   446→  // 6. Merge into existing bpSections (don't overwrite filled sections)
   447→  const existingSections = (journey?.bpSections as Record<string, unknown>) ?? {}
   448→  const existingMeta = (journey?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   449→  const mergedSections = { ...existingSections }

   451→
   452→  for (const [key, value] of Object.entries(generatedSections)) {

   455→      (typeof existing === 'string' && existing.trim() === '') ||
   456→      (typeof existing === 'object' && !Array.isArray(existing) && Object.keys(existing as object).length === 0)
   457→

   460→      // Mark as generated from parcours

   471→    where: { userId },
   472→    create: {
   473→      userId,
   474→      bpSections: mergedSections as Prisma.InputJsonValue,
   475→      bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   476→      bpStatus: 'DRAFT',
   477→      bpScore,
   478→      bpGeneratedAt: new Date(),
   479→    },
   480→    update: {
   481→      bpSections: mergedSections as Prisma.InputJsonValue,
   482→      bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   483→      bpStatus: 'DRAFT',
   484→      bpScore,
   485→      bpGeneratedAt: new Date(),
   486→    },
   487→  })
   488→
   489→  // Auto-snapshot after AI generation
   490→  const userForTenant = await db.user.findUnique({ where: { id: userId }, select: { tenantId: true } })
   491→  if (userForTenant?.tenantId) {
   492→    maybeAutoSnapshot(userId, savedJourney.id, userForTenant.tenantId, mergedSections, {
   493→      projectTitle: journey?.projectTitle,
   494→      projectSector: journey?.projectSector,
   495→    }, 'ai-generate')
   496→  }
   497→
   498→  // 9. Count what was generated vs skipped
   499→  const generatedKeys = Object.keys(generatedSections).filter(
   500→    (k) => typeof generatedSections[k] === 'string' && generatedSections[k].trim(),
   501→  )
   502→  const skippedKeys = generatedKeys.filter(
   503→    (k) => mergedSections[k] !== generatedSections[k],
   504→  )
   505→
   506→  return success(
   507→    {
   508→      sections: mergedSections,
   509→      generated: generatedKeys,
   510→      skipped: skippedKeys,
   511→      bpStatus: 'DRAFT',
   512→      bpScore,
   513→    },
   514→    `Business plan généré à partir du parcours (${generatedKeys.length} sections)`,
   515→  )
   516→}

   520→async function handleSyncSimulators(userId: string) {
   521→  // 1. Fetch existing BP sections and metadata
   522→  const journey = await db.creatorJourney.findFirst({ where: { userId: userId, enrollmentId: null }, select: { bpSections: true, bpSectionMeta: true },
   523→  })
   524→
   525→  const existingSections = (journey?.bpSections as Record<string, unknown>) ?? {}
   526→  const mergedSections = { ...existingSections }
   527→  const sectionMeta = (journey?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   528→
   529→  // Track what was synced vs skipped (with source key for meta)
   530→  const synced: { section: string; source: string; sourceKey?: DataSource }[] = []
   531→  const skipped: { section: string; source: string; reason: string }[] = []
   532→
   533→  // Helper: only fill empty sections
   534→  function fillSection(key: string, value: string, source: string, sourceKey?: DataSource) {
   535→    const existing = mergedSections[key]
   536→    const isEmpty = existing === null || existing === undefined ||
   537→      (typeof existing === 'string' && existing.trim() === '')
   538→
   539→    if (isEmpty && value.trim()) {
   540→      mergedSections[key] = value
   541→      synced.push({ section: key, source, sourceKey })
   542→      // Update section metadata with sync source
   543→      if (sourceKey) {
   544→        sectionMeta[key] = buildSectionMetaForSync(sectionMeta, key, value, sourceKey as 'marche' | 'juridique' | 'financier' | 'creasim' | 'parcours')
   545→      } } else if (!isEmpty) {
   546→      skipped.push({ section: key, source, reason: 'Déjà renseigné par l\'utilisateur' })
   547→    }
   548→  }
   549→
   550→  // 2. Fetch ALL simulator data in parallel
   551→  const [
   552→    marketAnalysis,
   553→    financialForecast,
   554→    creaSim,
   555→    juridiqueAnalysis,
   556→    businessModelCanvas,
   557→  ] = await Promise.all([
   558→    db.marketAnalysis.findFirst({ where: { userId: userId, enrollmentId: null } }),
   559→    db.financialForecast.findFirst({ where: { userId: userId, enrollmentId: null } }),
   560→    db.creaSimSimulation.findUnique({ where: { userId } }),
   561→    db.juridiqueAnalysis.findFirst({ where: { userId: userId, enrollmentId: null } }),
   562→    db.businessModelCanvas.findFirst({ where: { userId: userId, enrollmentId: null } }),
   563→  ])
   564→
   565→  // 3. Map marketAnalysis → etudeMarche, concurrence, swot
   566→  if (marketAnalysis) {
   567→    const parts: string[] = []
   568→    if (marketAnalysis.sector) parts.push(`**Secteur** : ${marketAnalysis.sector}`)
   569→    if (marketAnalysis.marketSize) parts.push(`**Taille du marché** : ${marketAnalysis.marketSize}`)
   570→    if (marketAnalysis.targetAudience) parts.push(`**Audience cible** : ${marketAnalysis.targetAudience}`)
   571→    if (marketAnalysis.opportunities) parts.push(`**Opportunités** : ${marketAnalysis.opportunities}`)
   572→    if (marketAnalysis.threats) parts.push(`**Menaces** : ${marketAnalysis.threats}`)
   573→    if (marketAnalysis.aiSynthesis) parts.push(`**Synthèse IA** : ${marketAnalysis.aiSynthesis}`)
   574→
   575→    const trends = marketAnalysis.trends as unknown[]
   576→    if (Array.isArray(trends) && trends.length > 0) {
   577→      parts.push(`**Tendances** : ${trends.join(', ')}`)
   578→    }
   579→
   580→    if (parts.length > 0) {
   581→      fillSection('etude-marche', `## Étude de marché\n\n${parts.join('\n\n')}`, 'Analyse de marché', 'marche')
   582→    } else {
   583→      skipped.push({ section: 'etude-marche', source: 'Analyse de marché', reason: 'Données insuffisantes' })
   584→    }
   585→
   586→    // Competition
   587→    const competitors = marketAnalysis.competitors as unknown[]
   588→    if (Array.isArray(competitors) && competitors.length > 0) {
   589→      const compParts = competitors.map((c: unknown, i: number) => {
   590→        const comp = c as Record<string, string>
   591→        return `### Concurrent ${i + 1}\n${Object.entries(comp).map(([k, v]) => `- **${k}** : ${v}`).join('\n')}`
   592→      })
   593→      fillSection('concurrence', `## Analyse concurrentielle\n\n${compParts.join('\n\n')}`, 'Analyse de marché', 'marche')
   594→    } else {
   595→      skipped.push({ section: 'concurrence', source: 'Analyse de marché', reason: 'Aucun concurrent identifié' })
   596→    }
   597→
   598→    // SWOT
   599→    const swotParts: string[] = []
   600→    if (marketAnalysis.opportunities) swotParts.push(`### Opportunités\n${marketAnalysis.opportunities}`)
   601→    if (marketAnalysis.threats) swotParts.push(`### Menaces\n${marketAnalysis.threats}`)
   602→    if (swotParts.length > 0) {
   603→      fillSection('swot', `## Analyse SWOT\n\n${swotParts.join('\n\n')}`, 'Analyse de marché', 'marche')
   604→    } else {
   605→      skipped.push({ section: 'swot', source: 'Analyse de marché', reason: 'Données insuffisantes' })
   606→    }
   607→  } else {
   608→    skipped.push({ section: 'etude-marche', source: 'Analyse de marché', reason: 'Simulation non complétée' })
   609→    skipped.push({ section: 'concurrence', source: 'Analyse de marché', reason: 'Simulation non complétée' })
   610→    skipped.push({ section: 'swot', source: 'Analyse de marché', reason: 'Simulation non complétée' })
   611→  }
   612→
   613→  // 4. Map financialForecast → planFinancier, compteResultat, investissements
   614→  if (financialForecast) {
   615→    // Financial plan summary
   616→    const finParts: string[] = []
   617→    if (financialForecast.sector) finParts.push(`**Secteur** : ${financialForecast.sector}`)
   618→    if (financialForecast.year1Revenue) finParts.push(`**CA Année 1** : ${financialForecast.year1Revenue.toLocaleString('fr-FR')} €`)
   619→    if (financialForecast.year2Revenue) finParts.push(`**CA Année 2** : ${financialForecast.year2Revenue.toLocaleString('fr-FR')} €`)
   620→    if (financialForecast.year3Revenue) finParts.push(`**CA Année 3** : ${financialForecast.year3Revenue.toLocaleString('fr-FR')} €`)
   621→    if (financialForecast.year1Expenses) finParts.push(`**Charges Année 1** : ${financialForecast.year1Expenses.toLocaleString('fr-FR')} €`)
   622→    if (financialForecast.year2Expenses) finParts.push(`**Charges Année 2** : ${financialForecast.year2Expenses.toLocaleString('fr-FR')} €`)
   623→    if (financialForecast.year3Expenses) finParts.push(`**Charges Année 3** : ${financialForecast.year3Expenses.toLocaleString('fr-FR')} €`)
   624→    if (financialForecast.breakevenMonth) finParts.push(`**Seuil de rentabilité** : Mois ${financialForecast.breakevenMonth}`)
   625→    if (financialForecast.initialInvestment) finParts.push(`**Investissement initial** : ${financialForecast.initialInvestment.toLocaleString('fr-FR')} €`)
   626→
   627→    if (finParts.length > 0) {
   628→      fillSection('financement', `## Plan financier (synthèse)\n\n${finParts.join('\n\n')}`, 'Prévisions financières', 'financier')
   629→    } else {
   630→      skipped.push({ section: 'financement', source: 'Prévisions financières', reason: 'Données insuffisantes' })
   631→    }
   632→
   633→    // P&L summary
   634→    const plParts: string[] = []
   635→    const fmtEur = (n: number) => n.toLocaleString('fr-FR')
   636→    const years = [
   637→      { year: 1, rev: financialForecast.year1Revenue, exp: financialForecast.year1Expenses },
   638→      { year: 2, rev: financialForecast.year2Revenue, exp: financialForecast.year2Expenses },
   639→      { year: 3, rev: financialForecast.year3Revenue, exp: financialForecast.year3Expenses },
   640→    ]
   641→    for (const y of years) {
   642→      if (y.rev != null || y.exp != null) {
   643→        const rev = y.rev ?? 0
   644→        const exp = y.exp ?? 0
   645→        const result = rev - exp
   646→        plParts.push(`### Année ${y.year}\n- Chiffre d'affaires : ${fmtEur(rev)} €\n- Charges : ${fmtEur(exp)} €\n- **Résultat net** : ${fmtEur(result)} € ${result >= 0 ? '✅' : '⚠️'}`)
   647→      }
   648→    }
   649→    if (plParts.length > 0) {
   650→      fillSection('compte-resultat', `## Compte de résultat prévisionnel (3 ans)\n\n${plParts.join('\n\n')}`, 'Prévisions financières', 'financier')
   651→    } else {
   652→      skipped.push({ section: 'compte-resultat', source: 'Prévisions financières', reason: 'Données insuffisantes' })
   653→    }
   654→
   655→    // Investments
   656→    if (financialForecast.initialInvestment) {
   657→      fillSection(
   658→        'investissements',
   659→        `## Investissements\n\n**Investissement initial** : ${financialForecast.initialInvestment.toLocaleString('fr-FR')} €\n\n*Cette donnée provient du module Prévisions Financières. Détaillez les postes d'investissement (matériel, logiciel, caution, etc.) pour compléter cette section.*`,
   660→        'Prévisions financières',
   661→        'financier',
   662→      )
   663→    } else {
   664→      skipped.push({ section: 'investissements', source: 'Prévisions financières', reason: 'Aucun investissement renseigné' })
   665→    }
   666→  } else {
   667→    skipped.push({ section: 'financement', source: 'Prévisions financières', reason: 'Simulation non complétée' })
   668→    skipped.push({ section: 'compte-resultat', source: 'Prévisions financières', reason: 'Simulation non complétée' })
   669→    skipped.push({ section: 'investissements', source: 'Prévisions financières', reason: 'Simulation non complétée' })
   670→  }
   671→
   672→  // 5. Map creaSim → rentabilite
   673→  if (creaSim) {
   674→    const rentParts: string[] = []
   675→    const fmtEur = (n: number) => n.toLocaleString('fr-FR')
   676→    const fmtPct = (n: number) => `${n.toFixed(1)}%`
   677→
   678→    if (creaSim.monthlyRevenue) rentParts.push(`**CA mensuel** : ${fmtEur(creaSim.monthlyRevenue)} €`)
   679→    if (creaSim.grossMarginRate) rentParts.push(`**Marge brute** : ${fmtPct(creaSim.grossMarginRate)}`)
   680→    if (creaSim.netMarginRate) rentParts.push(`**Marge nette** : ${fmtPct(creaSim.netMarginRate)}`)
   681→    if (creaSim.monthlyBreakeven) rentParts.push(`**Seuil de rentabilité mensuel** : ${fmtEur(creaSim.monthlyBreakeven)} €`)
   682→    if (creaSim.breakevenMonths) rentParts.push(`**Point mort** : ${creaSim.breakevenMonths.toFixed(1)} mois`)
   683→
   684→    if (creaSim.profitability1Y != null) {
   685→      rentParts.push(`**Rentabilité Année 1** : ${fmtEur(creaSim.profitability1Y)} € ${creaSim.profitability1Y >= 0 ? '✅' : '⚠️'}`)
   686→    }
   687→    if (creaSim.profitability2Y != null) {
   688→      rentParts.push(`**Rentabilité Année 2** : ${fmtEur(creaSim.profitability2Y)} € ${creaSim.profitability2Y >= 0 ? '✅' : '⚠️'}`)
   689→    }
   690→    if (creaSim.profitability3Y != null) {
   691→      rentParts.push(`**Rentabilité Année 3** : ${fmtEur(creaSim.profitability3Y)} € ${creaSim.profitability3Y >= 0 ? '✅' : '⚠️'}`)
   692→    }
   693→
   694→    // Margin analysis
   695→    if (creaSim.averageSellingPrice && creaSim.unitCost) {
   696→      const marginPerUnit = creaSim.averageSellingPrice - creaSim.unitCost
   697→      rentParts.push(`**Marge unitaire** : ${fmtEur(marginPerUnit)} € (PV ${fmtEur(creaSim.averageSellingPrice)} € - Coût ${fmtEur(creaSim.unitCost)} €)`)
   698→    }
   699→    if (creaSim.targetMarginRate) {
   700→      rentParts.push(`**Objectif de marge** : ${fmtPct(creaSim.targetMarginRate)}`)
   701→      if (creaSim.grossMarginRate) {
   702→        const gap = creaSim.grossMarginRate - creaSim.targetMarginRate
   703→        rentParts.push(`**Écart par rapport à l'objectif** : ${gap >= 0 ? '+' : ''}${fmtPct(gap)} ${gap >= 0 ? '✅ Atteint' : '⚠️ Non atteint'}`)
   704→      }
   705→    }
   706→
   707→    if (rentParts.length > 0) {
   708→      fillSection('seuil-rentabilite', `## Analyse de rentabilité (CreaSim)\n\n${rentParts.join('\n\n')}`, 'CreaSim', 'creasim')
   709→    } else {
   710→      skipped.push({ section: 'seuil-rentabilite', source: 'CreaSim', reason: 'Données insuffisantes' })
   711→    }
   712→  } else {
   713→    skipped.push({ section: 'seuil-rentabilite', source: 'CreaSim', reason: 'Simulation non complétée' })
   714→  }
   715→
   716→  // 6. Map juridiqueAnalysis → statutJuridique
   717→  if (juridiqueAnalysis) {
   718→    const jurParts: string[] = []
   719→    if (juridiqueAnalysis.recommendedStatus) jurParts.push(`**Statut recommandé** : ${juridiqueAnalysis.recommendedStatus}`)
   720→    if (juridiqueAnalysis.fiscalRegime) jurParts.push(`**Régime fiscal** : ${juridiqueAnalysis.fiscalRegime}`)
   721→    if (juridiqueAnalysis.legalStructure) jurParts.push(`**Structure juridique** : ${juridiqueAnalysis.legalStructure}`)
   722→
   723→    const charges = juridiqueAnalysis.socialCharges as unknown[]
   724→    if (Array.isArray(charges) && charges.length > 0) {
   725→      const chargeItems = charges.map((c: unknown) => {
   726→        const charge = c as Record<string, string>
   727→        return `- ${charge.name || charge.type || 'Charge'} : ${charge.amount || charge.rate || ''}`
   728→      })
   729→      jurParts.push(`**Charges sociales** :\n${chargeItems.join('\n')}`)
   730→    }
   731→
   732→    if (jurParts.length > 0) {
   733→      fillSection('statut-juridique', `## Statut juridique\n\n${jurParts.join('\n\n')}`, 'Analyse juridique', 'juridique')
   734→    } else {
   735→      skipped.push({ section: 'statut-juridique', source: 'Analyse juridique', reason: 'Données insuffisantes' })
   736→    }
   737→  } else {
   738→    skipped.push({ section: 'statut-juridique', source: 'Analyse juridique', reason: 'Simulation non complétée' })
   739→  }
   740→
   741→  // 7. Map businessModelCanvas → businessModelCanvas
   742→  if (businessModelCanvas) {
   743→    const bmcParts: string[] = []
   744→    const bmcLabels: Record<string, string> = {
   745→      partenairesCles: 'Partenaires clés',
   746→      activitesCles: 'Activités clés',
   747→      ressourcesCles: 'Ressources clés',
   748→      propositionValeur: 'Proposition de valeur',
   749→      relationsClients: 'Relations clients',
   750→      canaux: 'Canaux de distribution',
   751→      segmentsClients: 'Segments de clientèle',
   752→      structureCouts: 'Structure des coûts',
   753→      sourcesRevenus: 'Sources de revenus',
   754→    }
   755→
   756→    let hasContent = false
   757→    for (const [field, label] of Object.entries(bmcLabels)) {
   758→      const value = businessModelCanvas[field as keyof typeof businessModelCanvas]
   759→      if (value && typeof value === 'string' && value.trim()) {
   760→        bmcParts.push(`### ${label}\n${value}`)
   761→        hasContent = true
   762→      }
   763→    }
   764→
   765→    // BMC has no matching section in the BP — data is available in the BMC module
   766→    skipped.push({ section: 'businessModelCanvas', source: 'Business Model Canvas', reason: 'Aucune section correspondante dans le BP' })
   767→  } else {
   768→    skipped.push({ section: 'businessModelCanvas', source: 'Business Model Canvas', reason: 'Simulation non complétée' })
   769→  }
   770→
   771→  // 8. Recalculate bpScore and save (including bpSectionMeta)
   772→  const filledCount = Object.values(mergedSections).filter(isFilled).length
   773→  const totalSections = 24
   774→  const bpScore = Math.min(100, Math.round((filledCount / totalSections) * 100))
   775→
   776→  await db.creatorJourney.upsert({
   777→    where: { userId },
   778→    create: {
   779→      userId,
   780→      bpSections: mergedSections as Prisma.InputJsonValue,
   781→      bpSectionMeta: sectionMeta as unknown as Prisma.InputJsonValue,
   782→      bpScore,
   783→      bpGeneratedAt: new Date(),
   784→    },
   785→    update: {
   786→      bpSections: mergedSections as Prisma.InputJsonValue,
   787→      bpSectionMeta: sectionMeta as unknown as Prisma.InputJsonValue,
   788→      bpScore,
   789→      bpGeneratedAt: new Date(),
   790→    },
   791→  })
   792→
   793→  // Derive syncedSources from synced array
   794→  const syncedSources: Record<string, boolean> = {}
   795→  for (const item of synced) {
   796→    const source = item.source as string
   797→    if (source === 'Analyse de marché') syncedSources.marche = true
   798→    if (source === 'Prévisions financières' || source === 'CreaSim') syncedSources.financier = true
   799→    if (source === 'Analyse juridique') syncedSources.juridique = true
   800→    if (source === 'CreaSim') syncedSources.creasim = true
   801→  }
   802→  const summary = `${synced.length} sections synchronisées depuis ${[...new Set(synced.map(s => s.source))].join(', ')}`
   803→
   804→  return success(
   805→    {
   806→      sections: mergedSections,
   807→      synced,
   808→      skipped,
   809→      bpScore,
   810→      totalSections,
   811→      syncedSources,
   812→      summary,
   813→    },
   814→    `Synchronisation terminée : ${synced.length} sections ajoutées, ${skipped.length} ignorées`,
   815→  )
   816→}
