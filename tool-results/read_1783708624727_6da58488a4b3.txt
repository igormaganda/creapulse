    69→// ─── Helpers ─────────────────────────────────

    71→function isFilled(value: unknown): boolean {
    72→  if (value === null || value === undefined) return false
    73→  if (typeof value === 'string') return value.trim().length > 0
    74→  if (Array.isArray(value)) return value.length > 0
    75→  if (typeof value === 'object') {
    76→    return Object.values(value).some((v) =>
    77→      typeof v === 'string' && v.trim().length > 0 ||
    78→      typeof v === 'number' && v !== 0 ||
    79→      Array.isArray(v) && v.length > 0
    80→    )
    81→  }
    82→  return false
    83→}
    84→
    85→function computeWordCount(value: unknown): number {
    86→  if (typeof value === 'string') return value.trim().split(/\s+/).filter(Boolean).length
    87→  return 0
    88→}

    92→function buildSectionMetaForManual(
    93→  existingMeta: BpSectionMeta,
    94→  sectionId: string,
    95→  content: unknown,

    97→  const prev = existingMeta[sectionId]

    99→    source: 'manual' as DataSource,
   100→    lastModified: new Date().toISOString(),
   101→    manuallyEditedAt: new Date().toISOString(),
   102→    syncedAt: prev?.syncedAt ?? null,
   103→    wordCount: computeWordCount(content),
   104→    version: (prev?.version ?? 0) + 1,
   105→  }
   106→}

   125→/** Auto-create a snapshot if last one is older than 5 minutes (debounce) */
   126→async function maybeAutoSnapshot(
   127→  userId: string,
   128→  journeyId: string,
   129→  tenantId: string,
   130→  bpSections: Record<string, unknown>,
   131→  bpProjectContext: Record<string, unknown> | null,
   132→  trigger: 'auto' | 'ai-generate' = 'auto',
   133→) {
   134→  try {
   135→    const lastSnapshot = await db.bpSnapshot.findFirst({
   136→      where: { userId },
   137→      orderBy: { createdAt: 'desc' },
   138→      select: { createdAt: true, version: true },
   139→    })
   140→    const fiveMinAgo = new Date(Date.now() - 5 * 60_000)
   141→    if (lastSnapshot && lastSnapshot.createdAt > fiveMinAgo) return
   142→
   143→    const sectionCount = Object.values(bpSections).filter(isFilled).length
   144→    const wordCount = Object.values(bpSections).reduce(
   145→      (sum: number, v) => sum + computeWordCount(v), 0,
   146→    )
   147→    const version: number = (lastSnapshot?.version ?? 0) + 1
   148→    const now = new Date()
   149→    const label = `V${version} — ${now.toLocaleDateString('fr-FR', { day: 'numeric', month: 'short', year: 'numeric' })} ${now.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' })}`
   150→
   151→    await db.bpSnapshot.create({
   152→      data: {
   153→        userId,
   154→        tenantId,
   155→        creatorJourneyId: journeyId,
   156→        bpSections: bpSections as Prisma.InputJsonValue,
   157→        bpProjectContext: (bpProjectContext ?? {}) as Prisma.InputJsonValue,
   158→        version,
   159→        label,
   160→        trigger,
   161→        sectionCount,
   162→        wordCount,
   163→      },
   164→    })
   165→  } catch (err) {
   166→    // Non-critical: log but don't fail the parent operation
   167→    console.error('[AutoSnapshot] Failed to create auto-snapshot:', err)
   168→  }
   169→}

   173→async function getAuth(request: NextRequest) {
   174→  const cookieToken = request.cookies.get('session')?.value
   175→  const headerToken = getTokenFromHeader(request)
   176→  const token = cookieToken || headerToken
   177→  if (!token) throw Object.assign(new Error('No session token found'), { code: 'UNAUTHORIZED' })
   178→  return verifyToken(token)
   179→}

   183→function parseJsonFromLLM(text: string): Record<string, string> | null {
   184→  // Try direct JSON parse first
   185→  try {
   186→    const parsed = JSON.parse(text)
   187→    if (typeof parsed === 'object' && parsed !== null) return parsed
   188→  } catch {}
   189→
   190→  // Try matching JSON block
   191→  const jsonMatch = text.match(/\{[\s\S]*\}/)
   192→  if (jsonMatch) {
   193→    try {
   194→      const parsed = JSON.parse(jsonMatch[0])
   195→      if (typeof parsed === 'object' && parsed !== null) return parsed
   196→    } catch {}
   197→  }
   198→
   199→  return null
   200→}

   267→export async function PUT(request: NextRequest) {
   268→  try {
   269→    const payload = await getAuth(request)
   270→    const body = await request.json()
   271→
   272→    // ── Single-section save: action=save-section ──
   273→    if (body.action === 'save-section') {
   274→      const { sectionId, content } = body
   275→      if (!sectionId || content === undefined) {
   276→        return Errors.validation([{ field: 'sectionId/content', message: 'sectionId et content sont requis' }])
   277→      }
   278→
   279→      const existingJourney = await db.creatorJourney.findUnique({
   280→        where: { userId: payload.userId },
   281→        select: {
   282→          id: true, bpSections: true, bpSectionMeta: true,
   283→          bpStatus: true, bpScore: true, userId: true,
   284→          user: { select: { tenantId: true } },
   285→        },
   286→      })
   287→
   288→      const existingSections = (existingJourney?.bpSections as Record<string, unknown>) ?? {}
   289→      const existingMeta = (existingJourney?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   290→
   291→      // Update the section
   292→      existingSections[sectionId] = content
   293→
   294→      // Update metadata for this section only
   295→      existingMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)
   296→
   297→      // Recalculate bpScore
   298→      const filledSections = Object.values(existingSections).filter(isFilled).length
   299→      const bpScore = Math.min(100, Math.round((filledSections / 24) * 100))
   300→
   301→      let status = existingJourney?.bpStatus ?? 'IN_PROGRESS'
   302→      if (filledSections === 0) status = 'NOT_STARTED'
   303→      else if (filledSections === 24) status = 'DRAFT'
   304→      else status = 'IN_PROGRESS'
   305→
   306→      const journey = await db.creatorJourney.upsert({
   307→        where: { userId: payload.userId },
   308→        create: {
   309→          userId: payload.userId,
   310→          bpSections: existingSections as Prisma.InputJsonValue,
   311→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   312→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   313→          bpScore,
   314→        },
   315→        update: {
   316→          bpSections: existingSections as Prisma.InputJsonValue,
   317→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   318→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   319→          bpScore,
   320→        },
   321→      })
   322→
   323→      // ── Auto-snapshot: debounced to 5 min ──
   324→      const sectionTenantId = existingJourney?.user?.tenantId
   325→      if (sectionTenantId) {
   326→        maybeAutoSnapshot(payload.userId, journey.id, sectionTenantId, existingSections, null, 'auto')
   327→      }
   328→
   329→      return success(
   330→        {
   331→          id: journey.id,
   332→          sectionId,
   333→          meta: existingMeta[sectionId],
   334→          bpScore,
   335→          filledSections,
   336→          totalSections: 24,
   337→        },
   338→        'Section sauvegardée avec succès',
   339→      )
   340→    }
   341→
   342→    // ── Bulk save (existing logic) ──
   343→    const parsed = saveBpSchema.safeParse(body)
   344→
   345→    if (!parsed.success) {
   346→      return Errors.validation(
   347→        parsed.error.issues.map((i) => ({
   348→          field: i.path.join('.'),
   349→          message: i.message,
   350→        })),
   351→      )
   352→    }
   353→
   354→    const { sections, bpStatus } = parsed.data
   355→
   356→    // Read existing bpSectionMeta for metadata merge
   357→    const previousJourney = await db.creatorJourney.findUnique({
   358→      where: { userId: payload.userId },
   359→      select: {
   360→        bpSectionMeta: true, bpStatus: true, bpSections: true, userId: true,
   361→        user: { select: { tenantId: true, firstName: true, lastName: true } },
   362→      },
   363→    })
   364→    const existingMeta = (previousJourney?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   365→    const updatedMeta: BpSectionMeta = { ...existingMeta }
   366→    const previousStatus = previousJourney?.bpStatus ?? null
   367→    const userTenantId = previousJourney?.user?.tenantId
   368→
   369→    // Calculate completion & update metadata for all provided sections
   370→    const filledSections = sections
   371→      ? Object.values(sections).filter(isFilled).length
   372→      : 0
   373→    const totalSections = 24
   374→    const bpScore = Math.min(100, Math.round((filledSections / totalSections) * 100))
   375→
   376→    // Update bpSectionMeta for each non-empty section
   377→    if (sections) {
   378→      for (const [sectionId, content] of Object.entries(sections)) {
   379→        if (isFilled(content)) {
   380→          updatedMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)
   381→        }
   382→      }
   383→    }
   384→
   385→    // Determine status — preserve SUBMITTED when explicitly requested
   386→    let status: string = bpStatus ?? 'IN_PROGRESS'
   387→    if (status !== 'SUBMITTED') {
   388→      if (filledSections === 0) status = 'NOT_STARTED'
   389→      else if (filledSections === totalSections) status = 'DRAFT'
   390→      else status = 'IN_PROGRESS'
   391→    }
   392→
   393→    // Upsert CreatorJourney bp fields (now includes bpSectionMeta)
   394→    const journey = await db.creatorJourney.upsert({
   395→      where: { userId: payload.userId },
   396→      create: {
   397→        userId: payload.userId,
   398→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   399→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   400→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   401→        bpScore,
   402→      },
   403→      update: {
   404→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   405→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   406→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   407→        bpScore,
   408→        bpGeneratedAt: new Date(),
   409→      },
   410→    })
   411→
   412→    // ── Auto-snapshot: debounced to 5 min ──
   413→    if (sections && Object.keys(sections).length > 0 && userTenantId) {
   414→      const projectContext = previousJourney ? {
   415→        projectTitle: (previousJourney as any).projectTitle ?? null,
   416→        projectSector: (previousJourney as any).projectSector ?? null,
   417→      } : null
   418→      maybeAutoSnapshot(payload.userId, journey.id, userTenantId, sections ?? {}, projectContext, 'auto')
   419→    }
   420→
   421→    // ── Fire-and-forget notifications on status change ──
   422→    if (previousStatus && previousStatus !== status) {
   423→      const statusLabel = status === 'NOT_STARTED' ? 'Non démarré'
   424→        : status === 'IN_PROGRESS' ? 'En cours'
   425→        : status === 'DRAFT' ? 'Brouillon'
   426→        : status === 'SUBMITTED' ? 'Soumis'
   427→        : status
   428→
   429→      createNotification({
   430→        userId: payload.userId,
   431→        title: 'Business plan mis à jour',
   432→        content: `Votre business plan a été mis à jour — statut : ${statusLabel}`,
   433→        type: 'INFO',
   434→        link: '/bureau/business-plan',
   435→      }).catch(() => {})
   436→
   437→      // If SUBMITTED, notify all COUNSELOR users in the same tenant
   438→      if (status === 'SUBMITTED' && userTenantId) {
   439→        const beneficiaryName = [previousJourney?.user?.firstName, previousJourney?.user?.lastName].filter(Boolean).join(' ') || 'Un bénéficiaire'
   440→
   441→        // Fetch counselors + send notifications (fire-and-forget)
   442→        db.user.findMany({
   443→          where: { tenantId: userTenantId, role: 'COUNSELOR', isActive: true },
   444→          select: { id: true, email: true, firstName: true },
   445→        }).then((counselors) => {
   446→          const notificationPromises = counselors.map((c) =>
   447→            createNotification({
   448→              userId: c.id,
   449→              title: 'Business plan à reviewer',
   450→              content: `${beneficiaryName} a soumis son business plan pour review`,
   451→              type: 'ACTION_REQUIRED',
   452→              link: '/admin-centre/business-plan',
   453→            })
   454→          )
   455→          return Promise.all(notificationPromises)
   456→        }).catch(() => {})
   457→
   458→        // Send email to counselors (fire-and-forget)
   459→        import('@/lib/email').then(({ sendBpSubmittedEmail }) => {
   460→          db.user.findMany({
   461→            where: { tenantId: userTenantId, role: 'COUNSELOR', isActive: true },
   462→            select: { email: true, firstName: true },
   463→          }).then((counselors) => {
   464→            const beneficiaryFirstName = previousJourney?.user?.firstName || ''
   465→            counselors.forEach((c) => {
   466→              sendBpSubmittedEmail(beneficiaryFirstName, c.email, c.firstName || '').catch(() => {})
   467→            })
   468→          }).catch(() => {})
   469→        }).catch(() => {})
   470→
   471→        log.info('BP submitted', { userId: payload.userId, beneficiaryName })
   472→      }
   473→    }
   474→
   475→    return success(
   476→      {
   477→        id: journey.id,
   478→        bpStatus: status,
   479→        bpScore,
   480→        filledSections,
   481→        totalSections,
   482→      },
   483→      'Business plan sauvegardé avec succès',
   484→    )
   485→  } catch (err) {
   486→    if (err && typeof err === 'object' && 'code' in err) {
   487→      const authErr = err as { code: string }
   488→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   489→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   490→      }
   491→    }
   492→    return handleApiError(err)
   493→  }
   494→}

   498→export async function POST(request: NextRequest) {
   499→  try {
   500→    const payload = await getAuth(request)
   501→
   502→    const body = await request.json()
   503→    const parsed = bpActionSchema.safeParse(body)
   504→
   505→    if (!parsed.success) {
   506→      return Errors.validation(
   507→        parsed.error.issues.map((i) => ({
   508→          field: i.path.join('.'),
   509→          message: i.message,
   510→        })),
   511→      )
   512→    }
   513→
   514→    switch (parsed.data.action) {
   515→      case 'ai-suggest':
   516→        return handleAiSuggest(parsed.data, payload.userId)
   517→      case 'generate-from-parcours':
   518→        return handleGenerateFromParcours(payload.userId)
   519→      case 'sync-simulators':
   520→        return handleSyncSimulators(payload.userId)
   521→    }
   522→  } catch (err) {
   523→    if (err && typeof err === 'object' && 'code' in err) {
   524→      const authErr = err as { code: string }
   525→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   526→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   527→      }
   528→    }
   529→    return handleApiError(err)
   530→  }

   862→async function handleSyncSimulators(userId: string) {
   863→  // 1. Fetch existing BP sections and metadata
   864→  const journey = await db.creatorJourney.findUnique({
   865→    where: { userId },
   866→    select: { bpSections: true, bpSectionMeta: true },
   867→  })
   868→
   869→  const existingSections = (journey?.bpSections as Record<string, unknown>) ?? {}
   870→  const mergedSections = { ...existingSections }
   871→  const sectionMeta = (journey?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   872→
   873→  // Track what was synced vs skipped (with source key for meta)
   874→  const synced: { section: string; source: string; sourceKey?: DataSource }[] = []
   875→  const skipped: { section: string; source: string; reason: string }[] = []
   876→
   877→  // Helper: only fill empty sections
   878→  function fillSection(key: string, value: string, source: string, sourceKey?: DataSource) {
   879→    const existing = mergedSections[key]
   880→    const isEmpty = existing === null || existing === undefined ||
   881→      (typeof existing === 'string' && existing.trim() === '')
   882→
   883→    if (isEmpty && value.trim()) {
   884→      mergedSections[key] = value
   885→      synced.push({ section: key, source, sourceKey })
   886→      // Update section metadata with sync source
   887→      if (sourceKey) {
   888→        sectionMeta[key] = buildSectionMetaForSync(sectionMeta, key, value, sourceKey as 'marche' | 'juridique' | 'financier' | 'creasim' | 'parcours')
   889→      }
   890→    } else if (!isEmpty) {
   891→      skipped.push({ section: key, source, reason: 'Déjà renseigné par l\'utilisateur' })
   892→    }
   893→  }
   894→
   895→  // 2. Fetch ALL simulator data in parallel
   896→  const [
   897→    marketAnalysis,
   898→    financialForecast,
   899→    creaSim,
   900→    juridiqueAnalysis,
   901→    businessModelCanvas,
   902→  ] = await Promise.all([
   903→    db.marketAnalysis.findUnique({ where: { userId } }),
   904→    db.financialForecast.findUnique({ where: { userId } }),
   905→    db.creaSimSimulation.findUnique({ where: { userId } }),
   906→    db.juridiqueAnalysis.findUnique({ where: { userId } }),
   907→    db.businessModelCanvas.findUnique({ where: { userId } }),

   913→    if (marketAnalysis.sector) parts.push(`**Secteur** : ${marketAnalysis.sector}`)
   914→    if (marketAnalysis.marketSize) parts.push(`**Taille du marché** : ${marketAnalysis.marketSize}`)
   915→    if (marketAnalysis.targetAudience) parts.push(`**Audience cible** : ${marketAnalysis.targetAudience}`)
   916→    if (marketAnalysis.opportunities) parts.push(`**Opportunités** : ${marketAnalysis.opportunities}`)
   917→    if (marketAnalysis.threats) parts.push(`**Menaces** : ${marketAnalysis.threats}`)
   918→    if (marketAnalysis.aiSynthesis) parts.push(`**Synthèse IA** : ${marketAnalysis.aiSynthesis}`)
   919→
   920→    const trends = marketAnalysis.trends as unknown[]
   921→    if (Array.isArray(trends) && trends.length > 0) {
   922→      parts.push(`**Tendances** : ${trends.join(', ')}`)

   924→
   925→    if (parts.length > 0) {

   946→    if (marketAnalysis.threats) swotParts.push(`### Menaces\n${marketAnalysis.threats}`)
   947→    if (swotParts.length > 0) {

   949→    } else {
   950→      skipped.push({ section: 'swot', source: 'Analyse de marché', reason: 'Données insuffisantes' })

   952→  } else {
   953→    skipped.push({ section: 'etude-marche', source: 'Analyse de marché', reason: 'Simulation non complétée' })
   954→    skipped.push({ section: 'concurrence', source: 'Analyse de marché', reason: 'Simulation non complétée' })
   955→    skipped.push({ section: 'swot', source: 'Analyse de marché', reason: 'Simulation non complétée' })

   969→    if (financialForecast.breakevenMonth) finParts.push(`**Seuil de rentabilité** : Mois ${financialForecast.breakevenMonth}`)
   970→    if (financialForecast.initialInvestment) finParts.push(`**Investissement initial** : ${financialForecast.initialInvestment.toLocaleString('fr-FR')} €`)
   971→
   972→    if (finParts.length > 0) {
   973→      fillSection('financement', `## Plan financier (synthèse)\n\n${finParts.join('\n\n')}`, 'Prévisions financières', 'financier')

   983→      { year: 2, rev: financialForecast.year2Revenue, exp: financialForecast.year2Expenses },

  1006→        'financier',
  1007→      )
  1008→    } else {
  1009→      skipped.push({ section: 'investissements', source: 'Prévisions financières', reason: 'Aucun investissement renseigné' })
  1010→    }
  1011→  } else {
  1012→    skipped.push({ section: 'financement', source: 'Prévisions financières', reason: 'Simulation non complétée' })
  1013→    skipped.push({ section: 'compte-resultat', source: 'Prévisions financières', reason: 'Simulation non complétée' })
  1014→    skipped.push({ section: 'investissements', source: 'Prévisions financières', reason: 'Simulation non complétée' })
  1015→  }
  1016→
  1017→  // 5. Map creaSim → rentabilite
  1018→  if (creaSim) {
  1019→    const rentParts: string[] = []
  1020→    const fmtEur = (n: number) => n.toLocaleString('fr-FR')
  1021→    const fmtPct = (n: number) => `${n.toFixed(1)}%`
  1022→
  1023→    if (creaSim.monthlyRevenue) rentParts.push(`**CA mensuel** : ${fmtEur(creaSim.monthlyRevenue)} €`)
  1024→    if (creaSim.grossMarginRate) rentParts.push(`**Marge brute** : ${fmtPct(creaSim.grossMarginRate)}`)
  1025→    if (creaSim.netMarginRate) rentParts.push(`**Marge nette** : ${fmtPct(creaSim.netMarginRate)}`)
  1026→    if (creaSim.monthlyBreakeven) rentParts.push(`**Seuil de rentabilité mensuel** : ${fmtEur(creaSim.monthlyBreakeven)} €`)
  1027→    if (creaSim.breakevenMonths) rentParts.push(`**Point mort** : ${creaSim.breakevenMonths.toFixed(1)} mois`)
  1028→
  1029→    if (creaSim.profitability1Y != null) {
  1030→      rentParts.push(`**Rentabilité Année 1** : ${fmtEur(creaSim.profitability1Y)} € ${creaSim.profitability1Y >= 0 ? '✅' : '⚠️'}`)
  1031→    }
  1032→    if (creaSim.profitability2Y != null) {
  1033→      rentParts.push(`**Rentabilité Année 2** : ${fmtEur(creaSim.profitability2Y)} € ${creaSim.profitability2Y >= 0 ? '✅' : '⚠️'}`)
  1034→    }
  1035→    if (creaSim.profitability3Y != null) {
  1036→      rentParts.push(`**Rentabilité Année 3** : ${fmtEur(creaSim.profitability3Y)} € ${creaSim.profitability3Y >= 0 ? '✅' : '⚠️'}`)
  1037→    }
  1038→
  1039→    // Margin analysis
  1040→    if (creaSim.averageSellingPrice && creaSim.unitCost) {
  1041→      const marginPerUnit = creaSim.averageSellingPrice - creaSim.unitCost
  1042→      rentParts.push(`**Marge unitaire** : ${fmtEur(marginPerUnit)} € (PV ${fmtEur(creaSim.averageSellingPrice)} € - Coût ${fmtEur(creaSim.unitCost)} €)`)
  1043→    }
  1044→    if (creaSim.targetMarginRate) {
  1045→      rentParts.push(`**Objectif de marge** : ${fmtPct(creaSim.targetMarginRate)}`)
  1046→      if (creaSim.grossMarginRate) {
  1047→        const gap = creaSim.grossMarginRate - creaSim.targetMarginRate
  1048→        rentParts.push(`**Écart par rapport à l'objectif** : ${gap >= 0 ? '+' : ''}${fmtPct(gap)} ${gap >= 0 ? '✅ Atteint' : '⚠️ Non atteint'}`)
  1049→      }
  1050→    }
  1051→
  1052→    if (rentParts.length > 0) {
  1053→      fillSection('seuil-rentabilite', `## Analyse de rentabilité (CreaSim)\n\n${rentParts.join('\n\n')}`, 'CreaSim', 'creasim')
  1054→    } else {
  1055→      skipped.push({ section: 'seuil-rentabilite', source: 'CreaSim', reason: 'Données insuffisantes' })
  1056→    }
  1057→  } else {
  1058→    skipped.push({ section: 'seuil-rentabilite', source: 'CreaSim', reason: 'Simulation non complétée' })
  1059→  }
  1060→
  1061→  // 6. Map juridiqueAnalysis → statutJuridique
  1062→  if (juridiqueAnalysis) {
  1063→    const jurParts: string[] = []
  1064→    if (juridiqueAnalysis.recommendedStatus) jurParts.push(`**Statut recommandé** : ${juridiqueAnalysis.recommendedStatus}`)
  1065→    if (juridiqueAnalysis.fiscalRegime) jurParts.push(`**Régime fiscal** : ${juridiqueAnalysis.fiscalRegime}`)
  1066→    if (juridiqueAnalysis.legalStructure) jurParts.push(`**Structure juridique** : ${juridiqueAnalysis.legalStructure}`)
  1067→
  1068→    const charges = juridiqueAnalysis.socialCharges as unknown[]
  1069→    if (Array.isArray(charges) && charges.length > 0) {
  1070→      const chargeItems = charges.map((c: unknown) => {
  1071→        const charge = c as Record<string, string>
  1072→        return `- ${charge.name || charge.type || 'Charge'} : ${charge.amount || charge.rate || ''}`
  1073→      })
  1074→      jurParts.push(`**Charges sociales** :\n${chargeItems.join('\n')}`)
  1075→    }
  1076→
  1077→    if (jurParts.length > 0) {
  1078→      fillSection('statut-juridique', `## Statut juridique\n\n${jurParts.join('\n\n')}`, 'Analyse juridique', 'juridique')
  1079→    } else {
  1080→      skipped.push({ section: 'statut-juridique', source: 'Analyse juridique', reason: 'Données insuffisantes' })
  1081→    }
  1082→  } else {
  1083→    skipped.push({ section: 'statut-juridique', source: 'Analyse juridique', reason: 'Simulation non complétée' })
  1084→  }
  1085→
  1086→  // 7. Map businessModelCanvas → businessModelCanvas
  1087→  if (businessModelCanvas) {
  1088→    const bmcParts: string[] = []
  1089→    const bmcLabels: Record<string, string> = {
  1090→      partenairesCles: 'Partenaires clés',
  1091→      activitesCles: 'Activités clés',
  1092→      ressourcesCles: 'Ressources clés',
  1093→      propositionValeur: 'Proposition de valeur',
  1094→      relationsClients: 'Relations clients',
  1095→      canaux: 'Canaux de distribution',
  1096→      segmentsClients: 'Segments de clientèle',
  1097→      structureCouts: 'Structure des coûts',
  1098→      sourcesRevenus: 'Sources de revenus',
  1099→    }
  1100→
  1101→    let hasContent = false
  1102→    for (const [field, label] of Object.entries(bmcLabels)) {
  1103→      const value = businessModelCanvas[field as keyof typeof businessModelCanvas]
  1104→      if (value && typeof value === 'string' && value.trim()) {
  1105→        bmcParts.push(`### ${label}\n${value}`)
  1106→        hasContent = true
  1107→      }
  1108→    }
  1109→
  1110→    // BMC has no matching section in the BP — data is available in the BMC module
  1111→    skipped.push({ section: 'businessModelCanvas', source: 'Business Model Canvas', reason: 'Aucune section correspondante dans le BP' })
  1112→  } else {
  1113→    skipped.push({ section: 'businessModelCanvas', source: 'Business Model Canvas', reason: 'Simulation non complétée' })
  1114→  }
  1115→
  1116→  // 8. Recalculate bpScore and save (including bpSectionMeta)
  1117→  const filledCount = Object.values(mergedSections).filter(isFilled).length
  1118→  const totalSections = 24
  1119→  const bpScore = Math.min(100, Math.round((filledCount / totalSections) * 100))
  1120→
  1121→  await db.creatorJourney.upsert({
  1122→    where: { userId },
  1123→    create: {
  1124→      userId,
  1125→      bpSections: mergedSections as Prisma.InputJsonValue,
  1126→      bpSectionMeta: sectionMeta as unknown as Prisma.InputJsonValue,
  1127→      bpScore,
  1128→      bpGeneratedAt: new Date(),
  1129→    },
  1130→    update: {
  1131→      bpSections: mergedSections as Prisma.InputJsonValue,
  1132→      bpSectionMeta: sectionMeta as unknown as Prisma.InputJsonValue,
  1133→      bpScore,
  1134→      bpGeneratedAt: new Date(),
  1135→    },
  1136→  })
  1137→
  1138→  // Derive syncedSources from synced array
  1139→  const syncedSources: Record<string, boolean> = {}
  1140→  for (const item of synced) {
  1141→    const source = item.source as string
  1142→    if (source === 'Analyse de marché') syncedSources.marche = true
  1143→    if (source === 'Prévisions financières' || source === 'CreaSim') syncedSources.financier = true
  1144→    if (source === 'Analyse juridique') syncedSources.juridique = true
  1145→    if (source === 'CreaSim') syncedSources.creasim = true
  1146→  }
  1147→  const summary = `${synced.length} sections synchronisées depuis ${[...new Set(synced.map(s => s.source))].join(', ')}`
  1148→
  1149→  return success(
  1150→    {
  1151→      sections: mergedSections,
  1152→      synced,
  1153→      skipped,
  1154→      bpScore,
  1155→      totalSections,
  1156→      syncedSources,
  1157→      summary,
  1158→    },
  1159→    `Synchronisation terminée : ${synced.length} sections ajoutées, ${skipped.length} ignorées`,
  1160→  )
  1161→}
