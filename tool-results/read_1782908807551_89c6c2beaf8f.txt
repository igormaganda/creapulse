   274тЖТ   274тЖТfunction calculateResults(inputs: SimulationInputs): SimulationResults {
   275тЖТ   275тЖТ  const { monthlyRevenue, fixedCharges, variableChargesRate, initialInvestment } = inputs
   276тЖТ   276тЖТ  const fixedChargesTotal = fixedCharges.reduce((sum, c) => sum + c.amount, 0)
   277тЖТ   277тЖТ  const variableChargesAmount = monthlyRevenue * (variableChargesRate / 100)
   278тЖТ   278тЖТ  const totalCharges = fixedChargesTotal + variableChargesAmount
   279тЖТ   279тЖТ
   280тЖТ   280тЖТ  const grossMarginAmount = monthlyRevenue - variableChargesAmount
   281тЖТ   281тЖТ  const grossMarginRate = monthlyRevenue > 0 ? (grossMarginAmount / monthlyRevenue) * 100 : 0
   282тЖТ   282тЖТ
   283тЖТ   283тЖТ  const netMarginAmount = grossMarginAmount - fixedChargesTotal
   284тЖТ   284тЖТ  const netMarginRate = monthlyRevenue > 0 ? (netMarginAmount / monthlyRevenue) * 100 : 0
   285тЖТ   285тЖТ
   286тЖТ   286тЖТ  const coeff = 1 - variableChargesRate / 100
   287тЖТ   287тЖТ  const monthlyBreakeven = coeff > 0 ? fixedChargesTotal / coeff : Infinity
   288тЖТ   288тЖТ
   289тЖТ   289тЖТ  const netMarginPerMonth = netMarginAmount
   290тЖТ   290тЖТ  let breakevenMonths = Infinity
   291тЖТ   291тЖТ  if (netMarginPerMonth > 0 && initialInvestment > 0) {
   292тЖТ   292тЖТ    breakevenMonths = Math.ceil(initialInvestment / netMarginPerMonth)
   293тЖТ   293тЖТ  } else if (initialInvestment === 0 && netMarginPerMonth > 0) {
   294тЖТ   294тЖТ    breakevenMonths = 0
   295тЖТ   295тЖТ  } else if (netMarginPerMonth <= 0) {
   296тЖТ   296тЖТ    breakevenMonths = Infinity
   297тЖТ   297тЖТ  }
   298тЖТ   298тЖТ
   299тЖТ   299тЖТ  const growthRate = 1.005
   300тЖТ   300тЖТ
   301тЖТ   301тЖТ  let y1Rev = 0
   302тЖТ   302тЖТ  let y1Exp = 0
   303тЖТ   303тЖТ  for (let m = 0; m < 12; m++) {
   304тЖТ   304тЖТ    const ca = monthlyRevenue * Math.pow(growthRate, m)
   305тЖТ   305тЖТ    y1Rev += ca
   306тЖТ   306тЖТ    y1Exp += fixedChargesTotal + ca * (variableChargesRate / 100)
   307тЖТ   307тЖТ  }
   308тЖТ   308тЖТ
   309тЖТ   309тЖТ  const baseA2 = monthlyRevenue * Math.pow(growthRate, 12) * 1.02
   310тЖТ   310тЖТ  let y2Rev = 0
   311тЖТ   311тЖТ  let y2Exp = 0
   312тЖТ   312тЖТ  for (let m = 0; m < 12; m++) {
   313тЖТ   313тЖТ    const ca = baseA2 * Math.pow(1.002, m)
   314тЖТ   314тЖТ    y2Rev += ca
   315тЖТ   315тЖТ    y2Exp += fixedChargesTotal + ca * (variableChargesRate / 100)
   316тЖТ   316тЖТ  }
   317тЖТ   317тЖТ
   318тЖТ   318тЖТ  const baseA3 = baseA2 * Math.pow(1.002, 12) * 1.01
   319тЖТ   319тЖТ  let y3Rev = 0
   320тЖТ   320тЖТ  let y3Exp = 0
   321тЖТ   321тЖТ  for (let m = 0; m < 12; m++) {
   322тЖТ   322тЖТ    const ca = baseA3 * Math.pow(1.001, m)
   323тЖТ   323тЖТ    y3Rev += ca
   324тЖТ   324тЖТ    y3Exp += fixedChargesTotal + ca * (variableChargesRate / 100)
   325тЖТ   325тЖТ  }
   326тЖТ   326тЖТ
   327тЖТ   327тЖТ  const profitability1Y = (y1Rev - y1Exp) - initialInvestment
   328тЖТ   328тЖТ  const profitability2Y = profitability1Y + (y2Rev - y2Exp)
   329тЖТ   329тЖТ  const profitability3Y = profitability2Y + (y3Rev - y3Exp)
   330тЖТ   330тЖТ
   331тЖТ   331тЖТ  return {
   332тЖТ   332тЖТ    fixedChargesTotal,
   333тЖТ   333тЖТ    variableChargesAmount,

   355тЖТ   355тЖТfunction generateChartData(inputs: SimulationInputs): ChartDataPoint[] {
   356тЖТ   356тЖТ  const { monthlyRevenue, fixedCharges, variableChargesRate } = inputs
   357тЖТ   357тЖТ  const fixedChargesTotal = fixedCharges.reduce((sum, c) => sum + c.amount, 0)
   358тЖТ   358тЖТ  const growthRate = 1.005
   359тЖТ   359тЖТ  const months = ['Jan', 'F├йv', 'Mar', 'Avr', 'Mai', 'Jun', 'Jul', 'Ao├╗', 'Sep', 'Oct', 'Nov', 'D├йc']
   360тЖТ   360тЖТ
   361тЖТ   361тЖТ  return months.map((month, i) => {
   362тЖТ   362тЖТ    const ca = monthlyRevenue * Math.pow(growthRate, i)
   363тЖТ   363тЖТ    const charges = fixedChargesTotal + ca * (variableChargesRate / 100)
   364тЖТ   364тЖТ    return {
   365тЖТ   365тЖТ      month,
   366тЖТ   366тЖТ      ca: Math.round(ca),
   367тЖТ   367тЖТ      charges: Math.round(charges),
   368тЖТ   368тЖТ      resultat: Math.round(ca - charges),
   369тЖТ   369тЖТ    }
