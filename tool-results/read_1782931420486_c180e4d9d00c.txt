   146→// ─── Main Component ─────────────────────────
   147→
   148→export function ClotureRebondModule() {
   149→  const [isLoading, setIsLoading] = useState(true)
   150→  const [isSaving, setIsSaving] = useState(false)
   151→  const [aiLoading, setAiLoading] = useState(false)
   152→
   153→  const [bilan, setBilan] = useState('')
   154→  const [competences, setCompetences] = useState<CompetenceItem[]>([])
   155→  const [newCompetence, setNewCompetence] = useState('')
   156→  const [checklist, setChecklist] = useState<ClotureChecklist>({
   157→    declarationCessation: false,
   158→    urssaf: false,
   159→    impots: false,
   160→    rcs: false,
   161→    banque: false,
   162→    clients: false,
   163→    employes: false,
   164→    assurance: false,
   165→    bail: false,
   166→    comptable: false,
   167→  })
   168→  const [rebondPrefere, setRebondPrefere] = useState<string | null>(null)
   169→  const [aiOrientation, setAiOrientation] = useState('')
   170→
   171→  // ─── Load data on mount ──────────────────
   172→  useEffect(() => {
   173→    async function load() {
   174→      const saved = localStorage.getItem('creapulse-cloture-rebond')
   175→      if (saved) {
   176→        try {
   177→          const parsed = JSON.parse(saved)
   178→          if (parsed.bilan) setBilan(parsed.bilan)
   179→          if (parsed.competences?.length) setCompetences(parsed.competences)
   180→          if (parsed.checklist) setChecklist(prev => ({ ...prev, ...parsed.checklist }))
   181→          if (parsed.rebondPrefere) setRebondPrefere(parsed.rebondPrefere)
   182→          if (parsed.aiOrientation) setAiOrientation(parsed.aiOrientation)
   183→        } catch { /* ignore */ }
   184→      }
   185→
   186→      try {
   187→        const res = await authFetch('/api/paa/ateliers', {
   188→          method: 'POST',
   189→          body: JSON.stringify({ action: 'get', atelierCode: 'cloture-rebond' }),
   190→        })
   191→        if (res.ok) {
   192→          const json = await res.json()
   193→          if (json.success && json.data) {
   194→            const d = json.data
   195→            if (d.bilan) setBilan(d.bilan)
   196→            if (d.competences?.length) setCompetences(d.competences)
   197→            if (d.checklist) setChecklist(prev => ({ ...prev, ...d.checklist }))
   198→            if (d.rebondPrefere) setRebondPrefere(d.rebondPrefere)
   199→            if (d.aiOrientation) setAiOrientation(d.aiOrientation)
   200→          }
   201→        }
   202→      } catch { /* ignore */ }
   203→
   204→      setIsLoading(false)
   205→    }
   206→    load()
   207→  }, [])
   208→
   209→  // ─── Auto-save to localStorage ──────────
   210→  useEffect(() => {
   211→    if (!isLoading) {
   212→      localStorage.setItem('creapulse-cloture-rebond', JSON.stringify({
   213→        bilan, competences, checklist, rebondPrefere, aiOrientation,
   214→      }))
   215→    }
   216→  }, [isLoading, bilan, competences, checklist, rebondPrefere, aiOrientation])
   217→
   218→  // ─── Competence helpers ─────────────────
   219→  const addCompetence = useCallback(() => {
   220→    if (newCompetence.trim()) {
   221→      setCompetences(prev => [...prev, createCompetence(newCompetence)])
   222→      setNewCompetence('')
   223→    }
   224→  }, [newCompetence])
   225→
   226→  const removeCompetence = useCallback((id: string) => {
   227→    setCompetences(prev => prev.filter(c => c.id !== id))
   228→  }, [])
   229→
   230→  // ─── Checklist helper ────────────────────
   231→  const toggleChecklist = useCallback((key: keyof ClotureChecklist) => {
   232→    setChecklist(prev => ({ ...prev, [key]: !prev[key] }))
   233→  }, [])
   234→
   235→  // ─── Completion ─────────────────────────
   236→  const completion = useMemo(() => {
   237→    let filled = 0
   238→    const total = 4
   239→    if (bilan.trim()) filled++
   240→    if (competences.length > 0) filled++
   241→    const checked = Object.values(checklist).filter(Boolean).length
   242→    if (checked >= 5) filled++
   243→    if (rebondPrefere) filled++
   244→    return { filled, total, percent: Math.round((filled / total) * 100) }
   245→  }, [bilan, competences, checklist, rebondPrefere])
   246→
   247→  const checklistProgress = useMemo(() => {
   248→    const checked = Object.values(checklist).filter(Boolean).length
   249→    return { checked, total: CHECKLIST_ITEMS.length, percent: Math.round((checked / CHECKLIST_ITEMS.length) * 100) }
   250→  }, [checklist])
   251→
   252→  // ─── Save to API ────────────────────────
   253→  const handleSave = useCallback(async () => {
   254→    setIsSaving(true)
   255→    try {
   256→      const res = await authFetch('/api/paa/ateliers', {
   257→        method: 'PUT',
   258→        body: JSON.stringify({
   259→          atelierCode: 'cloture-rebond',
   260→          bilan,
   261→          competences,
   262→          checklist,
   263→          rebondPrefere,
   264→          aiOrientation,
   265→        }),
   266→      })
   267→      const json = await res.json()
   268→      if (!json.success) {
   269→        toast.error(json.error?.message || 'Erreur lors de la sauvegarde')
   270→        return
   271→      }
   272→      toast.success('Atelier Clôture & Rebond sauvegardé')
   273→    } catch {
   274→      toast.error('Erreur de connexion au serveur')
   275→    } finally {
   276→      setIsSaving(false)
   277→    }
   278→  }, [bilan, competences, checklist, rebondPrefere, aiOrientation])
   279→
   280→  // ─── AI Orientation ──────────────────────
   281→  const handleAiOrientation = useCallback(async () => {
   282→    setAiLoading(true)
   283→    try {
   284→      const res = await authFetch('/api/paa/ateliers', {
   285→        method: 'POST',
   286→        body: JSON.stringify({
   287→          action: 'ai-analyze',
   288→          atelierCode: 'cloture-rebond',
   289→          bilan,
   290→          competences: competences.map(c => c.name),
   291→          rebondPrefere,
   292→        }),
   293→      })
   294→      const json = await res.json()
   295→      if (json.success && json.data?.suggestion) {
   296→        setAiOrientation(typeof json.data.suggestion === 'string' ? json.data.suggestion : JSON.stringify(json.data.suggestion))
   297→        toast.success('Orientation IA générée !')
   298→      } else {
   299→        toast.error(json.error?.message || 'Erreur lors de la génération IA')
   300→      }
   301→    } catch {
   302→      toast.error('Erreur de connexion au serveur')
   303→    } finally {
   304→      setAiLoading(false)
   305→    }
   306→  }, [bilan, competences, rebondPrefere])
   307→
   308→  // ─── Loading skeleton ───────────────────
   309→  if (isLoading) {
   310→    return (
   311→      <div className="p-4 md:p-6 lg:p-8 space-y-6">
   312→        <div className="h-8 w-64 bg-muted animate-pulse rounded" />
   313→        <div className="h-4 w-96 bg-muted animate-pulse rounded" />
   314→        <div className="h-48 bg-muted animate-pulse rounded-xl" />
   315→      </div>
   316→    )
   317→  }
   318→
   319→  // ─── Main Render ─────────────────────────
   320→  return (
   321→    <motion.div
   322→      initial={{ opacity: 0, y: 20 }}
   323→      animate={{ opacity: 1, y: 0 }}
   324→      transition={{ duration: 0.4 }}
   325→      className="flex-1 overflow-y-auto"
   326→    >
   327→      {/* Header */}
   328→      <div className="flex flex-col sm:flex-row sm:items-center sm:justify-between gap-3 p-4 md:px-6 border-b bg-background">
   329→        <div className="flex items-center gap-3">
   330→          <div className="flex h-10 w-10 items-center justify-center rounded-xl bg-purple-500/10">
   331→            <RotateCcw className="h-5 w-5 text-purple-600 dark:text-purple-400" />
   332→          </div>
   333→          <div>
   334→            <div className="flex items-center gap-2">
   335→              <h2 className="text-xl font-bold text-foreground">Clôture &amp; Rebond Professionnel</h2>
   336→              <Badge className="bg-purple-100 text-purple-700 dark:bg-purple-900/30 dark:text-purple-300 text-[10px] px-1.5">Atelier 9</Badge>
   337→            </div>
   338→            <p className="text-xs text-muted-foreground">
   339→              {completion.filled}/{completion.total} sections — {completion.percent}%
   340→            </p>
   341→          </div>
   342→        </div>
   343→        <div className="flex items-center gap-2 flex-wrap">
   344→          <Button
   345→            variant="outline"
   346→            size="sm"
   347→            className="gap-1.5 border-amber-400/40 text-amber-500 hover:bg-amber-400/10"
   348→            onClick={handleAiOrientation}
   349→            disabled={aiLoading}
   350→          >
   351→            {aiLoading ? <Loader2 className="h-3.5 w-3.5 animate-spin" /> : <Sparkles className="h-3.5 w-3.5" />}
   352→            Orientation IA
   353→          </Button>
   354→          <Button
   355→            size="sm"
   356→            className="gap-1.5 bg-purple-600 hover:bg-purple-700 text-white"
   357→            onClick={handleSave}
   358→            disabled={isSaving}
   359→          >
   360→            {isSaving ? <Loader2 className="h-3.5 w-3.5 animate-spin" /> : <Save className="h-3.5 w-3.5" />}
   361→            Enregistrer
   362→          </Button>
   363→        </div>
   364→      </div>
   365→
   366→      {/* Content */}
   367→      <div className="p-4 md:p-6 space-y-6">
   368→        {/* ── Bilan de l'expérience ── */}
   369→        <Card className="border-purple-200 dark:border-purple-800">
   370→          <CardHeader className="pb-3">
   371→            <div className="flex items-center gap-2">
   372→              <CardTitle className="text-base flex items-center gap-2">
   373→                <FileCheck className="h-4 w-4 text-purple-600 dark:text-purple-400" />
   374→                Bilan de l&apos;expérience
   375→              </CardTitle>
   376→              <InfoPopover text="Réfléchissez à ce que vous avez appris, ce qui a fonctionné et ce qui n'a pas fonctionné. Ce bilan est précieux pour votre parcours futur." />
   377→            </div>
   378→            <CardDescription>Atelier 9 — Synthèse de votre parcours entrepreneurial</CardDescription>
   379→          </CardHeader>
   380→          <CardContent className="space-y-4">
   381→            <div className="space-y-3">
   382→              <div>
   383→                <p className="text-sm font-medium text-muted-foreground mb-1.5">Ce qui a fonctionné</p>
   384→                <Textarea
   385→                  value={bilan}
   386→                  onChange={(e) => setBilan(e.target.value)}
   387→                  placeholder={"- J'ai développé un réseau solide de 20+ contacts\n- J'ai acquis des compétences en gestion de projet\n- J'ai compris les besoins de mon marché cible\n- J'ai appris à gérer un budget prévisionnel"}
   388→                  className="min-h-[120px] text-sm resize-none"
   389→                />
   390→              </div>
   391→            </div>
   392→          </CardContent>
   393→        </Card>
   394→
   395→        {/* ── Compétences transférables ── */}
   396→        <Card className="border-purple-200 dark:border-purple-800">
   397→          <CardHeader className="pb-3">
   398→            <div className="flex items-center gap-2">
   399→              <CardTitle className="text-base flex items-center gap-2">
   400→                <GraduationCap className="h-4 w-4 text-purple-600 dark:text-purple-400" />
   401→                Compétences transférables
   402→              </CardTitle>
   403→              <InfoPopover text="Listez les compétences que vous avez acquises et qui sont valorisables dans d'autres contextes professionnels." />
   404→            </div>
   405→          </CardHeader>
   406→          <CardContent>
   407→            <div className="flex gap-2 mb-3">
   408→              <input
   409→                type="text"
   410→                value={newCompetence}
   411→                onChange={(e) => setNewCompetence(e.target.value)}
   412→                onKeyDown={(e) => e.key === 'Enter' && addCompetence()}
   413→                placeholder="Ex: Gestion de projet, Vente B2B, Community management..."
   414→                className="flex-1 text-sm rounded-lg border border-border bg-background/60 px-3 py-2 focus:outline-none focus:ring-2 focus:ring-purple-500/30"
   415→              />
   416→              <Button size="sm" variant="outline" onClick={addCompetence} disabled={!newCompetence.trim()} className="gap-1 border-purple-300 dark:border-purple-700 text-purple-600 dark:text-purple-400">
   417→                <Plus className="h-3.5 w-3.5" />
   418→                Ajouter
   419→              </Button>
   420→            </div>
   421→            {competences.length > 0 && (
   422→              <div className="flex flex-wrap gap-2">
   423→                {competences.map((comp) => (
   424→                  <Badge
   425→                    key={comp.id}
   426→                    variant="outline"
   427→                    className="gap-1.5 px-3 py-1.5 text-sm border-purple-300 dark:border-purple-700 text-purple-700 dark:text-purple-300"
   428→                  >
   429→                    <GraduationCap className="h-3 w-3" />
   430→                    {comp.name}
   431→                    <button
   432→                      type="button"
   433→                      onClick={() => removeCompetence(comp.id)}
   434→                      className="ml-1 hover:text-red-500 transition-colors"
   435→                    >
   436→                      ×
   437→                    </button>
   438→                  </Badge>
   439→                ))}
   440→              </div>
   441→            )}
   442→            {competences.length === 0 && (
   443→              <p className="text-xs text-muted-foreground">Aucune compétence ajoutée. Utilisez le champ ci-dessus.</p>
   444→            )}
   445→          </CardContent>
   446→        </Card>
   447→
   448→        <Separator />
   449→
   450→        {/* ── Formalités de clôture ── */}
   451→        <Card className="border-purple-200 dark:border-purple-800">
   452→          <CardHeader className="pb-3">
   453→            <div className="flex items-center justify-between">
   454→              <div className="flex items-center gap-2">
   455→                <CardTitle className="text-base flex items-center gap-2">
   456→                  <FileCheck className="h-4 w-4 text-purple-600 dark:text-purple-400" />
   457→                  Formalités de clôture
   458→                </CardTitle>
   459→                <InfoPopover text="Cochez chaque étape au fur et à mesure de son accomplissement. Ce checklist vous guidera dans les démarches administratives de fermeture." />
   460→              </div>
   461→              <Badge variant="outline" className="text-xs">
   462→                {checklistProgress.checked}/{checklistProgress.total} — {checklistProgress.percent}%
   463→              </Badge>
   464→            </div>
   465→          </CardHeader>
   466→          <CardContent>
   467→            <div className="space-y-2">
   468→              {CHECKLIST_ITEMS.map((item) => (
   469→                <button
   470→                  key={item.key}
   471→                  type="button"
   472→                  onClick={() => toggleChecklist(item.key)}
   473→                  className={cn(
   474→                    'flex items-start gap-3 w-full rounded-lg border p-3 text-left transition-all',
   475→                    checklist[item.key]
   476→                      ? 'border-purple-300 dark:border-purple-700 bg-purple-50/50 dark:bg-purple-950/20'
   477→                      : 'border-border bg-muted/20 hover:bg-muted/40'
   478→                  )}
   479→                >
   480→                  <div className={cn(
   481→                    'flex h-5 w-5 shrink-0 mt-0.5 items-center justify-center rounded-md border transition-colors',
   482→                    checklist[item.key]
   483→                      ? 'border-purple-500 bg-purple-500'
   484→                      : 'border-muted-foreground/30'
   485→                  )}>
   486→                    {checklist[item.key] && <Check className="h-3 w-3 text-white" />}
   487→                  </div>
   488→                  <div className="flex-1">
   489→                    <p className={cn(
   490→                      'text-sm font-medium',
