    71→    71→    71→function isFilled(value: unknown): boolean {
    72→    72→    72→  if (value === null || value === undefined) return false
    73→    73→    73→  if (typeof value === 'string') return value.trim().length > 0
    74→    74→    74→  if (Array.isArray(value)) return value.length > 0
    75→    75→    75→  if (typeof value === 'object') {
    76→    76→    76→    return Object.values(value).some((v) =>
    77→    77→    77→      typeof v === 'string' && v.trim().length > 0 ||
    78→    78→    78→      typeof v === 'number' && v !== 0 ||
    79→    79→    79→      Array.isArray(v) && v.length > 0
    80→    80→    80→    )
    81→    81→    81→  }
    82→    82→    82→  return false
    83→    83→    83→}
    84→    84→    84→
    85→    85→    85→function computeWordCount(value: unknown): number {
    86→    86→    86→  if (typeof value === 'string') return value.trim().split(/\s+/).filter(Boolean).length
    87→    87→    87→  return 0
    88→    88→    88→}

   125→   125→   125→/** Auto-create a snapshot if last one is older than 5 minutes (debounce) */
   126→   126→   126→async function maybeAutoSnapshot(
   127→   127→   127→  userId: string,
   128→   128→   128→  journeyId: string,
   129→   129→   129→  tenantId: string,
   130→   130→   130→  bpSections: Record<string, unknown>,
   131→   131→   131→  bpProjectContext: Record<string, unknown> | null,
   132→   132→   132→  trigger: 'auto' | 'ai-generate' = 'auto',
   133→   133→   133→) {
   134→   134→   134→  try {
   135→   135→   135→    const lastSnapshot = await db.bpSnapshot.findFirst({
   136→   136→   136→      where: { userId },
   137→   137→   137→      orderBy: { createdAt: 'desc' },
   138→   138→   138→      select: { createdAt: true, version: true },
   139→   139→   139→    })
   140→   140→   140→    const fiveMinAgo = new Date(Date.now() - 5 * 60_000)
   141→   141→   141→    if (lastSnapshot && lastSnapshot.createdAt > fiveMinAgo) return
   142→   142→   142→
   143→   143→   143→    const sectionCount = Object.values(bpSections).filter(isFilled).length
   144→   144→   144→    const wordCount = Object.values(bpSections).reduce(
   145→   145→   145→      (sum: number, v) => sum + computeWordCount(v), 0,
   146→   146→   146→    )
   147→   147→   147→    const version: number = (lastSnapshot?.version ?? 0) + 1
   148→   148→   148→    const now = new Date()
   149→   149→   149→    const label = `V${version} — ${now.toLocaleDateString('fr-FR', { day: 'numeric', month: 'short', year: 'numeric' })} ${now.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' })}`
   150→   150→   150→
   151→   151→   151→    await db.bpSnapshot.create({
   152→   152→   152→      data: {
   153→   153→   153→        userId,
   154→   154→   154→        tenantId,
   155→   155→   155→        creatorJourneyId: journeyId,
   156→   156→   156→        bpSections: bpSections as Prisma.InputJsonValue,
   157→   157→   157→        bpProjectContext: (bpProjectContext ?? {}) as Prisma.InputJsonValue,
   158→   158→   158→        version,
   159→   159→   159→        label,
   160→   160→   160→        trigger,
   161→   161→   161→        sectionCount,
   162→   162→   162→        wordCount,
   163→   163→   163→      },
   164→   164→   164→    })
   165→   165→   165→  } catch (err) {
   166→   166→   166→    // Non-critical: log but don't fail the parent operation
   167→   167→   167→    console.error('[AutoSnapshot] Failed to create auto-snapshot:', err)
   168→   168→   168→  }
   169→   169→   169→}

   173→   173→   173→async function getAuth(request: NextRequest) {
   174→   174→   174→  const cookieToken = request.cookies.get('session')?.value
   175→   175→   175→  const headerToken = getTokenFromHeader(request)
   176→   176→   176→  const token = cookieToken || headerToken
   177→   177→   177→  if (!token) throw Object.assign(new Error('No session token found'), { code: 'UNAUTHORIZED' })
   178→   178→   178→  return verifyToken(token)
   179→   179→   179→}

   183→   183→   183→function parseJsonFromLLM(text: string): Record<string, string> | null {
   184→   184→   184→  // Try direct JSON parse first
   185→   185→   185→  try {
   186→   186→   186→    const parsed = JSON.parse(text)
   187→   187→   187→    if (typeof parsed === 'object' && parsed !== null) return parsed
   188→   188→   188→  } catch {}
   189→   189→   189→
   190→   190→   190→  // Try matching JSON block
   191→   191→   191→  const jsonMatch = text.match(/\{[\s\S]*\}/)
   192→   192→   192→  if (jsonMatch) {
   193→   193→   193→    try {
   194→   194→   194→      const parsed = JSON.parse(jsonMatch[0])
   195→   195→   195→      if (typeof parsed === 'object' && parsed !== null) return parsed
   196→   196→   196→    } catch {}
   197→   197→   197→  }
   198→   198→   198→
   199→   199→   199→  return null
   200→   200→   200→}

   267→   267→   267→export async function PUT(request: NextRequest) {
   268→   268→   268→  try {
   269→   269→   269→    const payload = await getAuth(request)
   270→   270→   270→    const body = await request.json()

   300→   300→   300→

   302→   302→   302→      if (filledSections === 0) status = 'NOT_STARTED'
   303→   303→   303→      else if (filledSections === 24) status = 'DRAFT'
   304→   304→   304→      else status = 'IN_PROGRESS'

   313→   313→   313→          bpScore,
   314→   314→   314→        },
   315→   315→   315→        update: {
   316→   316→   316→          bpSections: existingSections as Prisma.InputJsonValue,
   317→   317→   317→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   318→   318→   318→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   319→   319→   319→          bpScore,
   320→   320→   320→        },
   321→   321→   321→      })
   322→   322→   322→
   323→   323→   323→      return success(
   324→   324→   324→        {
   325→   325→   325→          id: journey.id,
   326→   326→   326→          sectionId,
   327→   327→   327→          meta: existingMeta[sectionId],
   328→   328→   328→          bpScore,
   329→   329→   329→          filledSections,

   333→   333→   333→      )

   338→   338→   338→
   339→   339→   339→    if (!parsed.success) {
   340→   340→   340→      return Errors.validation(
   341→   341→   341→        parsed.error.issues.map((i) => ({
   342→   342→   342→          field: i.path.join('.'),
   343→   343→   343→          message: i.message,
   344→   344→   344→        })),
   345→   345→   345→      )
   346→   346→   346→    }
   347→   347→   347→
   348→   348→   348→    const { sections, bpStatus } = parsed.data
   349→   349→   349→
   350→   350→   350→    // Read existing bpSectionMeta for metadata merge
   351→   351→   351→    const previousJourney = await db.creatorJourney.findUnique({
   352→   352→   352→      where: { userId: payload.userId },
   353→   353→   353→      select: {
   354→   354→   354→        bpSectionMeta: true, bpStatus: true, bpSections: true, userId: true,
   355→   355→   355→        user: { select: { tenantId: true, firstName: true, lastName: true } },

   365→   365→   365→      ? Object.values(sections).filter(isFilled).length
   366→   366→   366→      : 0
   367→   367→   367→    const totalSections = 24

   369→   369→   369→

   373→   373→   373→        if (isFilled(content)) {
   374→   374→   374→          updatedMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)

   381→   381→   381→    if (status !== 'SUBMITTED') {
   382→   382→   382→      if (filledSections === 0) status = 'NOT_STARTED'
   383→   383→   383→      else if (filledSections === totalSections) status = 'DRAFT'
   384→   384→   384→      else status = 'IN_PROGRESS'
   385→   385→   385→    }
   386→   386→   386→
   387→   387→   387→    // Upsert CreatorJourney bp fields (now includes bpSectionMeta)

   390→   390→   390→      create: {
   391→   391→   391→        userId: payload.userId,
   392→   392→   392→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   393→   393→   393→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   394→   394→   394→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   395→   395→   395→        bpScore,
   396→   396→   396→      },
   397→   397→   397→      update: {
   398→   398→   398→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   399→   399→   399→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   400→   400→   400→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   401→   401→   401→        bpScore,
   402→   402→   402→        bpGeneratedAt: new Date(),
   403→   403→   403→      },
