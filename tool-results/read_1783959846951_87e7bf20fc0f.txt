    71→    71→function isFilled(value: unknown): boolean {
    72→    72→  if (value === null || value === undefined) return false
    73→    73→  if (typeof value === 'string') return value.trim().length > 0
    74→    74→  if (Array.isArray(value)) return value.length > 0
    75→    75→  if (typeof value === 'object') {
    76→    76→    return Object.values(value).some((v) =>
    77→    77→      typeof v === 'string' && v.trim().length > 0 ||
    78→    78→      typeof v === 'number' && v !== 0 ||
    79→    79→      Array.isArray(v) && v.length > 0
    80→    80→    )
    81→    81→  }
    82→    82→  return false
    83→    83→}
    84→    84→
    85→    85→function computeWordCount(value: unknown): number {
    86→    86→  if (typeof value === 'string') return value.trim().split(/\s+/).filter(Boolean).length
    87→    87→  return 0
    88→    88→}

   125→   125→/** Auto-create a snapshot if last one is older than 5 minutes (debounce) */
   126→   126→async function maybeAutoSnapshot(
   127→   127→  userId: string,
   128→   128→  journeyId: string,
   129→   129→  tenantId: string,
   130→   130→  bpSections: Record<string, unknown>,
   131→   131→  bpProjectContext: Record<string, unknown> | null,
   132→   132→  trigger: 'auto' | 'ai-generate' = 'auto',
   133→   133→) {
   134→   134→  try {
   135→   135→    const lastSnapshot = await db.bpSnapshot.findFirst({
   136→   136→      where: { userId },
   137→   137→      orderBy: { createdAt: 'desc' },
   138→   138→      select: { createdAt: true, version: true },
   139→   139→    })
   140→   140→    const fiveMinAgo = new Date(Date.now() - 5 * 60_000)
   141→   141→    if (lastSnapshot && lastSnapshot.createdAt > fiveMinAgo) return
   142→   142→
   143→   143→    const sectionCount = Object.values(bpSections).filter(isFilled).length
   144→   144→    const wordCount = Object.values(bpSections).reduce(
   145→   145→      (sum: number, v) => sum + computeWordCount(v), 0,
   146→   146→    )
   147→   147→    const version: number = (lastSnapshot?.version ?? 0) + 1
   148→   148→    const now = new Date()
   149→   149→    const label = `V${version} — ${now.toLocaleDateString('fr-FR', { day: 'numeric', month: 'short', year: 'numeric' })} ${now.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' })}`
   150→   150→
   151→   151→    await db.bpSnapshot.create({
   152→   152→      data: {
   153→   153→        userId,
   154→   154→        tenantId,
   155→   155→        creatorJourneyId: journeyId,
   156→   156→        bpSections: bpSections as Prisma.InputJsonValue,
   157→   157→        bpProjectContext: (bpProjectContext ?? {}) as Prisma.InputJsonValue,
   158→   158→        version,
   159→   159→        label,
   160→   160→        trigger,
   161→   161→        sectionCount,
   162→   162→        wordCount,
   163→   163→      },
   164→   164→    })
   165→   165→  } catch (err) {
   166→   166→    // Non-critical: log but don't fail the parent operation
   167→   167→    console.error('[AutoSnapshot] Failed to create auto-snapshot:', err)
   168→   168→  }
   169→   169→}

   172→   172→
   173→   173→async function getAuth(request: NextRequest) {
   174→   174→  const cookieToken = request.cookies.get('session')?.value
   175→   175→  const headerToken = getTokenFromHeader(request)
   176→   176→  const token = cookieToken || headerToken
   177→   177→  if (!token) throw Object.assign(new Error('No session token found'), { code: 'UNAUTHORIZED' })
   178→   178→  return verifyToken(token)
   179→   179→}

   182→   182→
   183→   183→function parseJsonFromLLM(text: string): Record<string, string> | null {
   184→   184→  // Try direct JSON parse first
   185→   185→  try {
   186→   186→    const parsed = JSON.parse(text)
   187→   187→    if (typeof parsed === 'object' && parsed !== null) return parsed
   188→   188→  } catch {}
   189→   189→
   190→   190→  // Try matching JSON block
   191→   191→  const jsonMatch = text.match(/\{[\s\S]*\}/)
   192→   192→  if (jsonMatch) {
   193→   193→    try {
   194→   194→      const parsed = JSON.parse(jsonMatch[0])
   195→   195→      if (typeof parsed === 'object' && parsed !== null) return parsed
   196→   196→    } catch {}
   197→   197→  }
   198→   198→
   199→   199→  return null
   200→   200→}

   238→   238→      {
   239→   239→        sections: journey.bpSections ?? {},
   240→   240→        bpStatus: journey.bpStatus,
   241→   241→        bpScore: journey.bpScore,
   242→   242→        bpGeneratedAt: journey.bpGeneratedAt,
   243→   243→        bpValidatedAt: journey.bpValidatedAt,
   244→   244→        projectContext: {
   245→   245→          projectTitle: journey.projectTitle,
   246→   246→          projectSector: journey.projectSector,
   247→   247→          projectStage: journey.projectStage,
   248→   248→          targetAudience: journey.targetAudience,
   249→   249→          valueProposition: journey.valueProposition,
   250→   250→        },
   251→   251→      },
   252→   252→      'Business plan chargé',
   253→   253→    )
   254→   254→  } catch (err) {
   255→   255→    if (err && typeof err === 'object' && 'code' in err) {
   256→   256→      const authErr = err as { code: string }

   258→   258→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   259→   259→      }
   260→   260→    }
   261→   261→    return handleApiError(err)
   262→   262→  }

   267→   267→export async function PUT(request: NextRequest) {
   268→   268→  try {
   269→   269→    const payload = await getAuth(request)
   270→   270→    const body = await request.json()
   271→   271→
   272→   272→    // ── Single-section save: action=save-section ──
   273→   273→    if (body.action === 'save-section') {
   274→   274→      const { sectionId, content } = body
   275→   275→      if (!sectionId || content === undefined) {
   276→   276→        return Errors.validation([{ field: 'sectionId/content', message: 'sectionId et content sont requis' }])
   277→   277→      }
   278→   278→
   279→   279→      const existingJourney = await db.creatorJourney.findUnique({
   280→   280→        where: { userId: payload.userId },
   281→   281→        select: {
   282→   282→          id: true, bpSections: true, bpSectionMeta: true,
   283→   283→          bpStatus: true, bpScore: true, userId: true,
   284→   284→          user: { select: { tenantId: true } },
   285→   285→        },
   286→   286→      })
   287→   287→
   288→   288→      const existingSections = (existingJourney?.bpSections as Record<string, unknown>) ?? {}
   289→   289→      const existingMeta = (existingJourney?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   290→   290→
   291→   291→      // Update the section
   292→   292→      existingSections[sectionId] = content
   293→   293→
   294→   294→      // Update metadata for this section only
   295→   295→      existingMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)
   296→   296→
   297→   297→      // Recalculate bpScore
   298→   298→      const filledSections = Object.values(existingSections).filter(isFilled).length
   299→   299→      const bpScore = Math.min(100, Math.round((filledSections / 24) * 100))
   300→   300→
   301→   301→      let status = existingJourney?.bpStatus ?? 'IN_PROGRESS'
   302→   302→      if (filledSections === 0) status = 'NOT_STARTED'
   303→   303→      else if (filledSections === 24) status = 'DRAFT'
   304→   304→      else status = 'IN_PROGRESS'
   305→   305→
   306→   306→      const journey = await db.creatorJourney.upsert({
   307→   307→        where: { userId: payload.userId },
   308→   308→        create: {
   309→   309→          userId: payload.userId,
   310→   310→          bpSections: existingSections as Prisma.InputJsonValue,
   311→   311→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   312→   312→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   313→   313→          bpScore,
   314→   314→        },
   315→   315→        update: {
   316→   316→          bpSections: existingSections as Prisma.InputJsonValue,
   317→   317→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   318→   318→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   319→   319→          bpScore,
   320→   320→        },
   321→   321→      })
   322→   322→
   323→   323→      // ── Auto-snapshot: debounced to 5 min ──
   324→   324→      const sectionTenantId = existingJourney?.user?.tenantId
   325→   325→      if (sectionTenantId) {
   326→   326→        maybeAutoSnapshot(payload.userId, journey.id, sectionTenantId, existingSections, null, 'auto')
   327→   327→      }
   328→   328→
   329→   329→      return success(
   330→   330→        {
   331→   331→          id: journey.id,
   332→   332→          sectionId,
   333→   333→          meta: existingMeta[sectionId],
   334→   334→          bpScore,
   335→   335→          filledSections,
   336→   336→          totalSections: 24,
   337→   337→        },
   338→   338→        'Section sauvegardée avec succès',
   339→   339→      )
   340→   340→    }
   341→   341→
   342→   342→    // ── Bulk save (existing logic) ──
   343→   343→    const parsed = saveBpSchema.safeParse(body)
   344→   344→
   345→   345→    if (!parsed.success) {
   346→   346→      return Errors.validation(
   347→   347→        parsed.error.issues.map((i) => ({
   348→   348→          field: i.path.join('.'),
   349→   349→          message: i.message,
   350→   350→        })),
   351→   351→      )
   352→   352→    }
   353→   353→
   354→   354→    const { sections, bpStatus } = parsed.data
   355→   355→
   356→   356→    // Read existing bpSectionMeta for metadata merge
   357→   357→    const previousJourney = await db.creatorJourney.findUnique({
   358→   358→      where: { userId: payload.userId },
   359→   359→      select: {
   360→   360→        bpSectionMeta: true, bpStatus: true, bpSections: true, userId: true,
   361→   361→        user: { select: { tenantId: true, firstName: true, lastName: true } },
   362→   362→      },
   363→   363→    })
   364→   364→    const existingMeta = (previousJourney?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   365→   365→    const updatedMeta: BpSectionMeta = { ...existingMeta }
   366→   366→    const previousStatus = previousJourney?.bpStatus ?? null
   367→   367→    const userTenantId = previousJourney?.user?.tenantId
   368→   368→
   369→   369→    // Calculate completion & update metadata for all provided sections
   370→   370→    const filledSections = sections
   371→   371→      ? Object.values(sections).filter(isFilled).length
   372→   372→      : 0
   373→   373→    const totalSections = 24
   374→   374→    const bpScore = Math.min(100, Math.round((filledSections / totalSections) * 100))
   375→   375→
   376→   376→    // Update bpSectionMeta for each non-empty section
   377→   377→    if (sections) {
   378→   378→      for (const [sectionId, content] of Object.entries(sections)) {
   379→   379→        if (isFilled(content)) {
   380→   380→          updatedMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)
   381→   381→        }
   382→   382→      }
   383→   383→    }
   384→   384→
   385→   385→    // Determine status — preserve SUBMITTED when explicitly requested
   386→   386→    let status: string = bpStatus ?? 'IN_PROGRESS'
   387→   387→    if (status !== 'SUBMITTED') {
   388→   388→      if (filledSections === 0) status = 'NOT_STARTED'
   389→   389→      else if (filledSections === totalSections) status = 'DRAFT'
   390→   390→      else status = 'IN_PROGRESS'
   391→   391→    }
   392→   392→
   393→   393→    // Upsert CreatorJourney bp fields (now includes bpSectionMeta)
   394→   394→    const journey = await db.creatorJourney.upsert({
   395→   395→      where: { userId: payload.userId },
   396→   396→      create: {
   397→   397→        userId: payload.userId,
   398→   398→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   399→   399→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   400→   400→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   401→   401→        bpScore,
   402→   402→      },
   403→   403→      update: {
   404→   404→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   405→   405→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   406→   406→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   407→   407→        bpScore,
   408→   408→        bpGeneratedAt: new Date(),
   409→   409→      },
   410→   410→    })
   411→   411→
   412→   412→    // ── Auto-snapshot: debounced to 5 min ──
   413→   413→    if (sections && Object.keys(sections).length > 0 && userTenantId) {
   414→   414→      const projectContext = previousJourney ? {
   415→   415→        projectTitle: (previousJourney as any).projectTitle ?? null,
   416→   416→        projectSector: (previousJourney as any).projectSector ?? null,
   417→   417→      } : null
   418→   418→      maybeAutoSnapshot(payload.userId, journey.id, userTenantId, sections ?? {}, projectContext, 'auto')
   419→   419→    }
   420→   420→
   421→   421→    // ── Fire-and-forget notifications on status change ──
   422→   422→    if (previousStatus && previousStatus !== status) {
   423→   423→      const statusLabel = status === 'NOT_STARTED' ? 'Non démarré'
   424→   424→        : status === 'IN_PROGRESS' ? 'En cours'
   425→   425→        : status === 'DRAFT' ? 'Brouillon'
   426→   426→        : status === 'SUBMITTED' ? 'Soumis'
   427→   427→        : status
   428→   428→
   429→   429→      createNotification({
   430→   430→        userId: payload.userId,
   431→   431→        title: 'Business plan mis à jour',
   432→   432→        content: `Votre business plan a été mis à jour — statut : ${statusLabel}`,
   433→   433→        type: 'INFO',
   434→   434→        link: '/bureau/business-plan',
   435→   435→      }).catch(() => {})
   436→   436→
   437→   437→      // If SUBMITTED, notify all COUNSELOR users in the same tenant
   438→   438→      if (status === 'SUBMITTED' && userTenantId) {
   439→   439→        const beneficiaryName = [previousJourney?.user?.firstName, previousJourney?.user?.lastName].filter(Boolean).join(' ') || 'Un bénéficiaire'
   440→   440→
   441→   441→        // Fetch counselors + send notifications (fire-and-forget)
   442→   442→        db.user.findMany({
   443→   443→          where: { tenantId: userTenantId, role: 'COUNSELOR', isActive: true },
   444→   444→          select: { id: true, email: true, firstName: true },
   445→   445→        }).then((counselors) => {
   446→   446→          const notificationPromises = counselors.map((c) =>
   447→   447→            createNotification({
   448→   448→              userId: c.id,
   449→   449→              title: 'Business plan à reviewer',
   450→   450→              content: `${beneficiaryName} a soumis son business plan pour review`,
   451→   451→              type: 'ACTION_REQUIRED',
   452→   452→              link: '/admin-centre/business-plan',
   453→   453→            })
   454→   454→          )
   455→   455→          return Promise.all(notificationPromises)
   456→   456→        }).catch(() => {})
   457→   457→
   458→   458→        // Send email to counselors (fire-and-forget)
   459→   459→        import('@/lib/email').then(({ sendBpSubmittedEmail }) => {
   460→   460→          db.user.findMany({
   461→   461→            where: { tenantId: userTenantId, role: 'COUNSELOR', isActive: true },
   462→   462→            select: { email: true, firstName: true },
   463→   463→          }).then((counselors) => {
   464→   464→            const beneficiaryFirstName = previousJourney?.user?.firstName || ''
   465→   465→            counselors.forEach((c) => {
   466→   466→              sendBpSubmittedEmail(beneficiaryFirstName, c.email, c.firstName || '').catch(() => {})

   498→   498→export async function POST(request: NextRequest) {
   499→   499→  try {
   500→   500→    const payload = await getAuth(request)
   501→   501→
   502→   502→    const body = await request.json()
   503→   503→    const parsed = bpActionSchema.safeParse(body)
   504→   504→
   505→   505→    if (!parsed.success) {
   506→   506→      return Errors.validation(
   507→   507→        parsed.error.issues.map((i) => ({
   508→   508→          field: i.path.join('.'),
   509→   509→          message: i.message,
   510→   510→        })),
   511→   511→      )
   512→   512→    }
   513→   513→
   514→   514→    switch (parsed.data.action) {
   515→   515→      case 'ai-suggest':
   516→   516→        return handleAiSuggest(parsed.data, payload.userId)
   517→   517→      case 'generate-from-parcours':
   518→   518→        return handleGenerateFromParcours(payload.userId)
   519→   519→      case 'sync-simulators':
   520→   520→        return handleSyncSimulators(payload.userId)
   521→   521→    }
   522→   522→  } catch (err) {
   523→   523→    if (err && typeof err === 'object' && 'code' in err) {
   524→   524→      const authErr = err as { code: string }
   525→   525→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   526→   526→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   527→   527→      }
   528→   528→    }
   529→   529→    return handleApiError(err)
   530→   530→  }

   862→   862→async function handleSyncSimulators(userId: string) {
   863→   863→  // 1. Fetch existing BP sections and metadata
   864→   864→  const journey = await db.creatorJourney.findUnique({
   865→   865→    where: { userId },
   866→   866→    select: { bpSections: true, bpSectionMeta: true },
   867→   867→  })
   868→   868→
   869→   869→  const existingSections = (journey?.bpSections as Record<string, unknown>) ?? {}
   870→   870→  const mergedSections = { ...existingSections }
   871→   871→  const sectionMeta = (journey?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   872→   872→
   873→   873→  // Track what was synced vs skipped (with source key for meta)
   874→   874→  const synced: { section: string; source: string; sourceKey?: DataSource }[] = []
   875→   875→  const skipped: { section: string; source: string; reason: string }[] = []
   876→   876→
   877→   877→  // Helper: only fill empty sections
   878→   878→  function fillSection(key: string, value: string, source: string, sourceKey?: DataSource) {
   879→   879→    const existing = mergedSections[key]
   880→   880→    const isEmpty = existing === null || existing === undefined ||
   881→   881→      (typeof existing === 'string' && existing.trim() === '')
   882→   882→
   883→   883→    if (isEmpty && value.trim()) {
   884→   884→      mergedSections[key] = value
   885→   885→      synced.push({ section: key, source, sourceKey })
   886→   886→      // Update section metadata with sync source
   887→   887→      if (sourceKey) {
   888→   888→        sectionMeta[key] = buildSectionMetaForSync(sectionMeta, key, value, sourceKey as 'marche' | 'juridique' | 'financier' | 'creasim' | 'parcours')
   889→   889→      }
   890→   890→    } else if (!isEmpty) {
   891→   891→      skipped.push({ section: key, source, reason: 'Déjà renseigné par l\'utilisateur' })
   892→   892→    }
   893→   893→  }
   894→   894→
   895→   895→  // 2. Fetch ALL simulator data in parallel
   896→   896→  const [
