   362тЖТ   362тЖТfunction readModuleData(moduleCode: string): Record<string, string> {
   363тЖТ   363тЖТ  const key = MODULE_STORAGE_KEYS[moduleCode]
   364тЖТ   364тЖТ  if (!key) return {}
   365тЖТ   365тЖТ
   366тЖТ   366тЖТ  try {
   367тЖТ   367тЖТ    if (typeof window === 'undefined') return {}
   368тЖТ   368тЖТ    const raw = localStorage.getItem(key)
   369тЖТ   369тЖТ    if (!raw) return {}
   370тЖТ   370тЖТ
   371тЖТ   371тЖТ    const data = JSON.parse(raw)
   372тЖТ   372тЖТ    if (!data || typeof data !== 'object') return {}
   373тЖТ   373тЖТ
   374тЖТ   374тЖТ    // Flatten the data for the AI context (max 6 entries, truncate long values)
   375тЖТ   375тЖТ    const flat: Record<string, string> = {}
   376тЖТ   376тЖТ    let count = 0
   377тЖТ   377тЖТ    for (const [k, v] of Object.entries(data)) {
   378тЖТ   378тЖТ      if (count >= 6) break
   379тЖТ   379тЖТ      if (v === null || v === undefined) continue
   380тЖТ   380тЖТ      const strVal = typeof v === 'string' ? v : JSON.stringify(v)
   381тЖТ   381тЖТ      flat[k] = strVal.length > 300 ? strVal.slice(0, 300) + '...' : strVal
   382тЖТ   382тЖТ      count++
   383тЖТ   383тЖТ    }
   384тЖТ   384тЖТ    return flat
   385тЖТ   385тЖТ  } catch {
   386тЖТ   386тЖТ    return {}
   387тЖТ   387тЖТ  }
   388тЖТ   388тЖТ}
