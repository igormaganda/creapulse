   200→        bpStatus: status,
   201→        bpScore,
   202→        filledSections,
   203→        totalSections,
   204→      },
   205→      'Business plan sauvegardé avec succès',
   206→    )
   207→  } catch (err) {
   208→    if (err && typeof err === 'object' && 'code' in err) {
   209→      const authErr = err as { code: string }
   210→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   211→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   212→      }
   213→    }
   214→    return handleApiError(err)
   215→  }
   216→}

   220→export async function POST(request: NextRequest) {
   221→  try {
   222→    const payload = await getAuth(request)
   223→
   224→    const body = await request.json()
   225→    const parsed = bpActionSchema.safeParse(body)
   226→
   227→    if (!parsed.success) {
   228→      return Errors.validation(
   229→        parsed.error.issues.map((i) => ({
   230→          field: i.path.join('.'),
   231→          message: i.message,
   232→        })),
   233→      )
   234→    }
   235→
   236→    switch (parsed.data.action) {
   237→      case 'ai-suggest':
   238→        return handleAiSuggest(parsed.data, payload.userId)
   239→      case 'generate-from-parcours':
   240→        return handleGenerateFromParcours(payload.userId)
   241→      case 'sync-simulators':
   242→        return handleSyncSimulators(payload.userId)
   243→    }
   244→  } catch (err) {
   245→    if (err && typeof err === 'object' && 'code' in err) {
   246→      const authErr = err as { code: string }
   247→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   248→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   249→      }
   250→    }
   251→    return handleApiError(err)
   252→  }
   253→}

   685→    const fmtEur = (n: number) => n.toLocaleString('fr-FR')
   686→    const years = [
   687→      { year: 1, rev: financialForecast.year1Revenue, exp: financialForecast.year1Expenses },
   688→      { year: 2, rev: financialForecast.year2Revenue, exp: financialForecast.year2Expenses },
   689→      { year: 3, rev: financialForecast.year3Revenue, exp: financialForecast.year3Expenses },
   690→    ]
   691→    for (const y of years) {
   692→      if (y.rev != null || y.exp != null) {
   693→        const rev = y.rev ?? 0
   694→        const exp = y.exp ?? 0
   695→        const result = rev - exp
   696→        plParts.push(`### Année ${y.year}\n- Chiffre d'affaires : ${fmtEur(rev)} €\n- Charges : ${fmtEur(exp)} €\n- **Résultat net** : ${fmtEur(result)} € ${result >= 0 ? '✅' : '⚠️'}`)
   697→      }
   698→    }
   699→    if (plParts.length > 0) {
   700→      fillSection('compte-resultat', `## Compte de résultat prévisionnel (3 ans)\n\n${plParts.join('\n\n')}`, 'Prévisions financières')
   701→    } else {
   702→      skipped.push({ section: 'compte-resultat', source: 'Prévisions financières', reason: 'Données insuffisantes' })
   703→    }
   704→
   705→    // Investments
   706→    if (financialForecast.initialInvestment) {
   707→      fillSection(
   708→        'investissements',
   709→        `## Investissements\n\n**Investissement initial** : ${financialForecast.initialInvestment.toLocaleString('fr-FR')} €\n\n*Cette donnée provient du module Prévisions Financières. Détaillez les postes d'investissement (matériel, logiciel, caution, etc.) pour compléter cette section.*`,
   710→        'Prévisions financières',
   711→      )
   712→    } else {
   713→      skipped.push({ section: 'investissements', source: 'Prévisions financières', reason: 'Aucun investissement renseigné' })
   714→    }
   715→  } else {
   716→    skipped.push({ section: 'financement', source: 'Prévisions financières', reason: 'Simulation non complétée' })
   717→    skipped.push({ section: 'compte-resultat', source: 'Prévisions financières', reason: 'Simulation non complétée' })
   718→    skipped.push({ section: 'investissements', source: 'Prévisions financières', reason: 'Simulation non complétée' })
   719→  }
   720→
   721→  // 5. Map creaSim → rentabilite
   722→  if (creaSim) {
   723→    const rentParts: string[] = []
   724→    const fmtEur = (n: number) => n.toLocaleString('fr-FR')
   725→    const fmtPct = (n: number) => `${n.toFixed(1)}%`
   726→
   727→    if (creaSim.monthlyRevenue) rentParts.push(`**CA mensuel** : ${fmtEur(creaSim.monthlyRevenue)} €`)
   728→    if (creaSim.grossMarginRate) rentParts.push(`**Marge brute** : ${fmtPct(creaSim.grossMarginRate)}`)
   729→    if (creaSim.netMarginRate) rentParts.push(`**Marge nette** : ${fmtPct(creaSim.netMarginRate)}`)
   730→    if (creaSim.monthlyBreakeven) rentParts.push(`**Seuil de rentabilité mensuel** : ${fmtEur(creaSim.monthlyBreakeven)} €`)
   731→    if (creaSim.breakevenMonths) rentParts.push(`**Point mort** : ${creaSim.breakevenMonths.toFixed(1)} mois`)
   732→
   733→    if (creaSim.profitability1Y != null) {
   734→      rentParts.push(`**Rentabilité Année 1** : ${fmtEur(creaSim.profitability1Y)} € ${creaSim.profitability1Y >= 0 ? '✅' : '⚠️'}`)
   735→    }
   736→    if (creaSim.profitability2Y != null) {
   737→      rentParts.push(`**Rentabilité Année 2** : ${fmtEur(creaSim.profitability2Y)} € ${creaSim.profitability2Y >= 0 ? '✅' : '⚠️'}`)
   738→    }
   739→    if (creaSim.profitability3Y != null) {
   740→      rentParts.push(`**Rentabilité Année 3** : ${fmtEur(creaSim.profitability3Y)} € ${creaSim.profitability3Y >= 0 ? '✅' : '⚠️'}`)
   741→    }
   742→
   743→    // Margin analysis
   744→    if (creaSim.averageSellingPrice && creaSim.unitCost) {
   745→      const marginPerUnit = creaSim.averageSellingPrice - creaSim.unitCost
   746→      rentParts.push(`**Marge unitaire** : ${fmtEur(marginPerUnit)} € (PV ${fmtEur(creaSim.averageSellingPrice)} € - Coût ${fmtEur(creaSim.unitCost)} €)`)
   747→    }
   748→    if (creaSim.targetMarginRate) {
   749→      rentParts.push(`**Objectif de marge** : ${fmtPct(creaSim.targetMarginRate)}`)
   750→      if (creaSim.grossMarginRate) {
   751→        const gap = creaSim.grossMarginRate - creaSim.targetMarginRate
   752→        rentParts.push(`**Écart par rapport à l'objectif** : ${gap >= 0 ? '+' : ''}${fmtPct(gap)} ${gap >= 0 ? '✅ Atteint' : '⚠️ Non atteint'}`)
   753→      }
   754→    }
   755→
   756→    if (rentParts.length > 0) {
   757→      fillSection('seuil-rentabilite', `## Analyse de rentabilité (CreaSim)\n\n${rentParts.join('\n\n')}`, 'CreaSim')
   758→    } else {
   759→      skipped.push({ section: 'seuil-rentabilite', source: 'CreaSim', reason: 'Données insuffisantes' })
   760→    }
   761→  } else {
   762→    skipped.push({ section: 'seuil-rentabilite', source: 'CreaSim', reason: 'Simulation non complétée' })
   763→  }
   764→
   765→  // 6. Map juridiqueAnalysis → statutJuridique
   766→  if (juridiqueAnalysis) {
   767→    const jurParts: string[] = []
   768→    if (juridiqueAnalysis.recommendedStatus) jurParts.push(`**Statut recommandé** : ${juridiqueAnalysis.recommendedStatus}`)
   769→    if (juridiqueAnalysis.fiscalRegime) jurParts.push(`**Régime fiscal** : ${juridiqueAnalysis.fiscalRegime}`)
   770→    if (juridiqueAnalysis.legalStructure) jurParts.push(`**Structure juridique** : ${juridiqueAnalysis.legalStructure}`)
   771→
   772→    const charges = juridiqueAnalysis.socialCharges as unknown[]
   773→    if (Array.isArray(charges) && charges.length > 0) {
   774→      const chargeItems = charges.map((c: unknown) => {
   775→        const charge = c as Record<string, string>
   776→        return `- ${charge.name || charge.type || 'Charge'} : ${charge.amount || charge.rate || ''}`
   777→      })
   778→      jurParts.push(`**Charges sociales** :\n${chargeItems.join('\n')}`)
   779→    }
   780→
   781→    if (jurParts.length > 0) {
   782→      fillSection('statut-juridique', `## Statut juridique\n\n${jurParts.join('\n\n')}`, 'Analyse juridique')
   783→    } else {
   784→      skipped.push({ section: 'statut-juridique', source: 'Analyse juridique', reason: 'Données insuffisantes' })
   785→    }
   786→  } else {
   787→    skipped.push({ section: 'statut-juridique', source: 'Analyse juridique', reason: 'Simulation non complétée' })
   788→  }
   789→
   790→  // 7. Map businessModelCanvas → businessModelCanvas
   791→  if (businessModelCanvas) {
   792→    const bmcParts: string[] = []
   793→    const bmcLabels: Record<string, string> = {
   794→      partenairesCles: 'Partenaires clés',
   795→      activitesCles: 'Activités clés',
   796→      ressourcesCles: 'Ressources clés',
   797→      propositionValeur: 'Proposition de valeur',
   798→      relationsClients: 'Relations clients',
   799→      canaux: 'Canaux de distribution',
   800→      segmentsClients: 'Segments de clientèle',
   801→      structureCouts: 'Structure des coûts',
   802→      sourcesRevenus: 'Sources de revenus',
   803→    }
   804→
   805→    let hasContent = false
   806→    for (const [field, label] of Object.entries(bmcLabels)) {
   807→      const value = businessModelCanvas[field as keyof typeof businessModelCanvas]
   808→      if (value && typeof value === 'string' && value.trim()) {
   809→        bmcParts.push(`### ${label}\n${value}`)
   810→        hasContent = true
   811→      }
   812→    }
   813→
   814→    // BMC has no matching section in the BP — data is available in the BMC module
   815→    skipped.push({ section: 'businessModelCanvas', source: 'Business Model Canvas', reason: 'Aucune section correspondante dans le BP' })
   816→  } else {
   817→    skipped.push({ section: 'businessModelCanvas', source: 'Business Model Canvas', reason: 'Simulation non complétée' })
   818→  }
   819→
   820→  // 8. Recalculate bpScore and save
   821→  const filledCount = Object.values(mergedSections).filter((v) =>
   822→    typeof v === 'string' && v.trim().length > 0 ||
   823→    typeof v === 'object' && v !== null && !Array.isArray(v) && Object.keys(v).length > 0 ||
   824→    Array.isArray(v) && (v as unknown[]).length > 0,
   825→  ).length
   826→  const totalSections = 24
   827→  const bpScore = Math.min(100, Math.round((filledCount / totalSections) * 100))
   828→
   829→  await db.creatorJourney.upsert({
   830→    where: { userId },
   831→    create: {
   832→      userId,
   833→      bpSections: mergedSections as Prisma.InputJsonValue,
   834→      bpScore,
   835→      bpGeneratedAt: new Date(),
   836→    },
   837→    update: {
   838→      bpSections: mergedSections as Prisma.InputJsonValue,
   839→      bpScore,
   840→      bpGeneratedAt: new Date(),
   841→    },
   842→  })
   843→
   844→  // Derive syncedSources from synced array
   845→  const syncedSources: Record<string, boolean> = {}
   846→  for (const item of synced) {
   847→    const source = item.source as string
   848→    if (source === 'Analyse de marché') syncedSources.marche = true
   849→    if (source === 'Prévisions financières' || source === 'CreaSim') syncedSources.financier = true
   850→    if (source === 'Analyse juridique') syncedSources.juridique = true
   851→    if (source === 'CreaSim') syncedSources.creasim = true
   852→  }
   853→  const summary = `${synced.length} sections synchronisées depuis ${[...new Set(synced.map(s => s.source))].join(', ')}`
   854→
   855→  return success(
   856→    {
   857→      sections: mergedSections,
   858→      synced,
   859→      skipped,
   860→      bpScore,
   861→      totalSections,
   862→      syncedSources,
   863→      summary,
   864→    },
   865→    `Synchronisation terminée : ${synced.length} sections ajoutées, ${skipped.length} ignorées`,
   866→  )
   867→}
