    70→// ─── Helpers ─────────────────────────────────

    72→function isFilled(value: unknown): boolean {
    73→  if (value === null || value === undefined) return false
    74→  if (typeof value === 'string') return value.trim().length > 0
    75→  if (Array.isArray(value)) return value.length > 0
    76→  if (typeof value === 'object') {
    77→    return Object.values(value).some((v) =>
    78→      typeof v === 'string' && v.trim().length > 0 ||
    79→      typeof v === 'number' && v !== 0 ||
    80→      Array.isArray(v) && v.length > 0
    81→    )
    82→  }
    83→  return false
    84→}
    85→
    86→function computeWordCount(value: unknown): number {
    87→  if (typeof value === 'string') return value.trim().split(/\s+/).filter(Boolean).length
    88→  return 0
    89→}

    93→function buildSectionMetaForManual(
    94→  existingMeta: BpSectionMeta,
    95→  sectionId: string,
    96→  content: unknown,
    97→): SectionMetaEntry {
    98→  const prev = existingMeta[sectionId]

   100→    source: 'manual' as DataSource,
   101→    lastModified: new Date().toISOString(),
   102→    manuallyEditedAt: new Date().toISOString(),
   103→    syncedAt: prev?.syncedAt ?? null,
   104→    wordCount: computeWordCount(content),
   105→    version: (prev?.version ?? 0) + 1,
   106→  }

   126→/** Auto-create a snapshot if last one is older than 5 minutes (debounce) */
   127→async function maybeAutoSnapshot(
   128→  userId: string,
   129→  journeyId: string,
   130→  tenantId: string,
   131→  bpSections: Record<string, unknown>,
   132→  bpProjectContext: Record<string, unknown> | null,
   133→  trigger: 'auto' | 'ai-generate' = 'auto',
   134→) {
   135→  try {
   136→    const lastSnapshot = await db.bpSnapshot.findFirst({
   137→      where: { userId },
   138→      orderBy: { createdAt: 'desc' },
   139→      select: { createdAt: true, version: true },
   140→    })
   141→    const fiveMinAgo = new Date(Date.now() - 5 * 60_000)
   142→    if (lastSnapshot && lastSnapshot.createdAt > fiveMinAgo) return
   143→
   144→    const sectionCount = Object.values(bpSections).filter(isFilled).length
   145→    const wordCount = Object.values(bpSections).reduce(
   146→      (sum: number, v) => sum + computeWordCount(v), 0,
   147→    )
   148→    const version: number = (lastSnapshot?.version ?? 0) + 1
   149→    const now = new Date()
   150→    const label = `V${version} — ${now.toLocaleDateString('fr-FR', { day: 'numeric', month: 'short', year: 'numeric' })} ${now.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' })}`
   151→
   152→    await db.bpSnapshot.create({
   153→      data: {
   154→        userId,
   155→        tenantId,
   156→        creatorJourneyId: journeyId,
   157→        bpSections: bpSections as Prisma.InputJsonValue,
   158→        bpProjectContext: (bpProjectContext ?? {}) as Prisma.InputJsonValue,
   159→        version,
   160→        label,
   161→        trigger,
   162→        sectionCount,
   163→        wordCount,
   164→      },
   165→    })
   166→  } catch (err) {
   167→    // Non-critical: log but don't fail the parent operation
   168→    console.error('[AutoSnapshot] Failed to create auto-snapshot:', err)
   169→  }
   170→}

   174→async function getAuth(request: NextRequest) {
   175→  const cookieToken = request.cookies.get('session')?.value
   176→  const headerToken = getTokenFromHeader(request)
   177→  const token = cookieToken || headerToken
   178→  if (!token) throw Object.assign(new Error('No session token found'), { code: 'UNAUTHORIZED' })
   179→  return verifyToken(token)
   180→}

   184→function parseJsonFromLLM(text: string): Record<string, string> | null {
   185→  // Try direct JSON parse first
   186→  try {
   187→    const parsed = JSON.parse(text)
   188→    if (typeof parsed === 'object' && parsed !== null) return parsed
   189→  } catch {}
   190→
   191→  // Try matching JSON block
   192→  const jsonMatch = text.match(/\{[\s\S]*\}/)
   193→  if (jsonMatch) {
   194→    try {
   195→      const parsed = JSON.parse(jsonMatch[0])
   196→      if (typeof parsed === 'object' && parsed !== null) return parsed
   197→    } catch {}
   198→  }
   199→
   200→  return null
   201→}

   269→export async function PUT(request: NextRequest) {
   270→  try {
   271→    const payload = await getAuth(request)
   272→    const enrollmentId = getEnrollmentIdFromRequest(request)
   273→    const body = await request.json()
   274→
   275→    // ── Single-section save: action=save-section ──
   276→    if (body.action === 'save-section') {
   277→      const { sectionId, content } = body
   278→      if (!sectionId || content === undefined) {
   279→        return Errors.validation([{ field: 'sectionId/content', message: 'sectionId et content sont requis' }])
   280→      }
   281→
   282→      const existingJourney = await db.creatorJourney.findUnique({
   283→        where: { userId: payload.userId },
   284→        select: {
   285→          id: true, bpSections: true, bpSectionMeta: true,
   286→          bpStatus: true, bpScore: true, userId: true,
   287→          user: { select: { tenantId: true } },
   288→        },
   289→      })
   290→
   291→      const existingSections = (existingJourney?.bpSections as Record<string, unknown>) ?? {}
   292→      const existingMeta = (existingJourney?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   293→
   294→      // Update the section
   295→      existingSections[sectionId] = content
   296→
   297→      // Update metadata for this section only
   298→      existingMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)
   299→
   300→      // Recalculate bpScore
   301→      const filledSections = Object.values(existingSections).filter(isFilled).length
   302→      const bpScore = Math.min(100, Math.round((filledSections / 24) * 100))
   303→
   304→      let status = existingJourney?.bpStatus ?? 'IN_PROGRESS'
   305→      if (filledSections === 0) status = 'NOT_STARTED'
   306→      else if (filledSections === 24) status = 'DRAFT'
   307→      else status = 'IN_PROGRESS'
   308→
   309→      const journey = await db.creatorJourney.upsert({
   310→        where: { userId: payload.userId },
   311→        create: {
   312→          userId: payload.userId,
   313→          bpSections: existingSections as Prisma.InputJsonValue,
   314→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   315→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   316→          bpScore,
   317→        },
   318→        update: {
   319→          bpSections: existingSections as Prisma.InputJsonValue,
   320→          bpSectionMeta: existingMeta as unknown as Prisma.InputJsonValue,
   321→          bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   322→          bpScore,
   323→        },
   324→      })
   325→
   326→      // ── Auto-snapshot: debounced to 5 min ──
   327→      const sectionTenantId = existingJourney?.user?.tenantId
   328→      if (sectionTenantId) {
   329→        maybeAutoSnapshot(payload.userId, journey.id, sectionTenantId, existingSections, null, 'auto')
   330→      }
   331→
   332→      return success(
   333→        {
   334→          id: journey.id,
   335→          sectionId,
   336→          meta: existingMeta[sectionId],
   337→          bpScore,
   338→          filledSections,
   339→          totalSections: 24,
   340→        },
   341→        'Section sauvegardée avec succès',
   342→      )
   343→    }
   344→
   345→    // ── Bulk save (existing logic) ──
   346→    const parsed = saveBpSchema.safeParse(body)
   347→
   348→    if (!parsed.success) {
   349→      return Errors.validation(
   350→        parsed.error.issues.map((i) => ({
   351→          field: i.path.join('.'),
   352→          message: i.message,
   353→        })),
   354→      )
   355→    }
   356→
   357→    const { sections, bpStatus } = parsed.data
   358→
   359→    // Read existing bpSectionMeta for metadata merge
   360→    const previousJourney = await db.creatorJourney.findUnique({
   361→      where: { userId: payload.userId },
   362→      select: {
   363→        bpSectionMeta: true, bpStatus: true, bpSections: true, userId: true,
   364→        user: { select: { tenantId: true, firstName: true, lastName: true } },
   365→      },
   366→    })
   367→    const existingMeta = (previousJourney?.bpSectionMeta as unknown as BpSectionMeta) ?? {}
   368→    const updatedMeta: BpSectionMeta = { ...existingMeta }
   369→    const previousStatus = previousJourney?.bpStatus ?? null
   370→    const userTenantId = previousJourney?.user?.tenantId
   371→
   372→    // Calculate completion & update metadata for all provided sections
   373→    const filledSections = sections
   374→      ? Object.values(sections).filter(isFilled).length
   375→      : 0
   376→    const totalSections = 24
   377→    const bpScore = Math.min(100, Math.round((filledSections / totalSections) * 100))
   378→
   379→    // Update bpSectionMeta for each non-empty section
   380→    if (sections) {
   381→      for (const [sectionId, content] of Object.entries(sections)) {
   382→        if (isFilled(content)) {
   383→          updatedMeta[sectionId] = buildSectionMetaForManual(existingMeta, sectionId, content)
   384→        }
   385→      }
   386→    }
   387→
   388→    // Determine status — preserve SUBMITTED when explicitly requested
   389→    let status: string = bpStatus ?? 'IN_PROGRESS'
   390→    if (status !== 'SUBMITTED') {
   391→      if (filledSections === 0) status = 'NOT_STARTED'
   392→      else if (filledSections === totalSections) status = 'DRAFT'
   393→      else status = 'IN_PROGRESS'
   394→    }
   395→
   396→    // Upsert CreatorJourney bp fields (now includes bpSectionMeta)
   397→    const journey = await db.creatorJourney.upsert({
   398→      where: { userId: payload.userId },
   399→      create: {
   400→        userId: payload.userId,
   401→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   402→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   403→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   404→        bpScore,
   405→      },
   406→      update: {
   407→        bpSections: (sections ?? {}) as Prisma.InputJsonValue,
   408→        bpSectionMeta: updatedMeta as unknown as Prisma.InputJsonValue,
   409→        bpStatus: status as 'NOT_STARTED' | 'IN_PROGRESS' | 'GENERATING' | 'DRAFT' | 'REVIEW' | 'VALIDATED' | 'EXPORTED',
   410→        bpScore,
   411→        bpGeneratedAt: new Date(),
   412→      },
   413→    })
   414→
   415→    // ── Auto-snapshot: debounced to 5 min ──
   416→    if (sections && Object.keys(sections).length > 0 && userTenantId) {
   417→      const projectContext = previousJourney ? {
   418→        projectTitle: (previousJourney as any).projectTitle ?? null,
   419→        projectSector: (previousJourney as any).projectSector ?? null,
   420→      } : null
   421→      maybeAutoSnapshot(payload.userId, journey.id, userTenantId, sections ?? {}, projectContext, 'auto')
   422→    }
   423→
   424→    // ── Fire-and-forget notifications on status change ──
   425→    if (previousStatus && previousStatus !== status) {
   426→      const statusLabel = status === 'NOT_STARTED' ? 'Non démarré'
   427→        : status === 'IN_PROGRESS' ? 'En cours'
   428→        : status === 'DRAFT' ? 'Brouillon'
   429→        : status === 'SUBMITTED' ? 'Soumis'
   430→        : status
   431→
   432→      createNotification({
   433→        userId: payload.userId,
   434→        title: 'Business plan mis à jour',
   435→        content: `Votre business plan a été mis à jour — statut : ${statusLabel}`,
   436→        type: 'INFO',
   437→        link: '/bureau/business-plan',
   438→      }).catch(() => {})
   439→
   440→      // If SUBMITTED, notify all COUNSELOR users in the same tenant
   441→      if (status === 'SUBMITTED' && userTenantId) {
   442→        const beneficiaryName = [previousJourney?.user?.firstName, previousJourney?.user?.lastName].filter(Boolean).join(' ') || 'Un bénéficiaire'
   443→
   444→        // Fetch counselors + send notifications (fire-and-forget)
   445→        db.user.findMany({
   446→          where: { tenantId: userTenantId, role: 'COUNSELOR', isActive: true },
   447→          select: { id: true, email: true, firstName: true },
   448→        }).then((counselors) => {
   449→          const notificationPromises = counselors.map((c) =>
   450→            createNotification({
   451→              userId: c.id,
   452→              title: 'Business plan à reviewer',
   453→              content: `${beneficiaryName} a soumis son business plan pour review`,
   454→              type: 'ACTION_REQUIRED',
   455→              link: '/admin-centre/business-plan',
   456→            })
   457→          )
   458→          return Promise.all(notificationPromises)
   459→        }).catch(() => {})
   460→
   461→        // Send email to counselors (fire-and-forget)
   462→        import('@/lib/email').then(({ sendBpSubmittedEmail }) => {
   463→          db.user.findMany({
   464→            where: { tenantId: userTenantId, role: 'COUNSELOR', isActive: true },
   465→            select: { email: true, firstName: true },
   466→          }).then((counselors) => {
   467→            const beneficiaryFirstName = previousJourney?.user?.firstName || ''
   468→            counselors.forEach((c) => {
   469→              sendBpSubmittedEmail(beneficiaryFirstName, c.email, c.firstName || '').catch(() => {})
   470→            })
   471→          }).catch(() => {})
   472→        }).catch(() => {})
   473→
   474→        log.info('BP submitted', { userId: payload.userId, beneficiaryName })
   475→      }
   476→    }
   477→
   478→    return success(
   479→      {
   480→        id: journey.id,
   481→        bpStatus: status,
   482→        bpScore,
   483→        filledSections,
   484→        totalSections,
   485→      },
   486→      'Business plan sauvegardé avec succès',
   487→    )
   488→  } catch (err) {
   489→    if (err && typeof err === 'object' && 'code' in err) {
   490→      const authErr = err as { code: string }
   491→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   492→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   493→      }
   494→    }
   495→    return handleApiError(err)
   496→  }
   497→}

   499→// ─── POST: Multi-action handler ──────────────

   501→export async function POST(request: NextRequest) {
   502→  try {
   503→    const payload = await getAuth(request)
   504→    const enrollmentId = getEnrollmentIdFromRequest(request)
   505→
   506→    const body = await request.json()
   507→    const parsed = bpActionSchema.safeParse(body)
   508→
   509→    if (!parsed.success) {
   510→      return Errors.validation(
   511→        parsed.error.issues.map((i) => ({
   512→          field: i.path.join('.'),
   513→          message: i.message,
   514→        })),
   515→      )
   516→    }
   517→
   518→    switch (parsed.data.action) {
   519→      case 'ai-suggest':
   520→        return handleAiSuggest(parsed.data, payload.userId, enrollmentId)
   521→      case 'generate-from-parcours':
   522→        return handleGenerateFromParcours(payload.userId, enrollmentId)
   523→      case 'sync-simulators':
   524→        return handleSyncSimulators(payload.userId, enrollmentId)
   525→    }
   526→  } catch (err) {
   527→    if (err && typeof err === 'object' && 'code' in err) {
   528→      const authErr = err as { code: string }
   529→      if (authErr.code === 'TOKEN_EXPIRED' || authErr.code === 'UNAUTHORIZED') {
   530→        return Errors.unauthorized('Session expirée — veuillez vous reconnecter')
   531→      }
   532→    }
   533→    return handleApiError(err)
   534→  }
   535→}

   539→async function handleAiSuggest(
   540→  data: z.infer<typeof aiSuggestSchema>,
   541→  userId: string,
   542→  enrollmentId: string | null,
   543→) {
   544→  const { sectionId, sectionTitle, existingContent, projectContext } = data
   545→
   546→  // Fetch project context if not provided
   547→  let context = projectContext || ''
   548→  if (!context) {
   549→    const journey = await db.creatorJourney.findUnique({
   550→      where: { userId: userId },
   551→      select: {
   552→        projectTitle: true,
   553→        projectSector: true,
   554→        projectStage: true,
   555→        targetAudience: true,
   556→        valueProposition: true,
   557→        projectDescription: true,
   558→      },
   559→    })
   560→    if (journey) {
   561→      const parts: string[] = []
   562→      if (journey.projectTitle) parts.push(`Projet: ${journey.projectTitle}`)
   563→      if (journey.projectSector) parts.push(`Secteur: ${journey.projectSector}`)
   564→      if (journey.projectStage) parts.push(`Stade: ${journey.projectStage}`)

   880→  const skipped: { section: string; source: string; reason: string }[] = []
   881→

   885→    const isEmpty = existing === null || existing === undefined ||
   886→      (typeof existing === 'string' && existing.trim() === '')
   887→
   888→    if (isEmpty && value.trim()) {
   889→      mergedSections[key] = value
   890→      synced.push({ section: key, source, sourceKey })

   892→      if (sourceKey) {
   893→        sectionMeta[key] = buildSectionMetaForSync(sectionMeta, key, value, sourceKey as 'marche' | 'juridique' | 'financier' | 'creasim' | 'parcours')
   894→      }
   895→    } else if (!isEmpty) {
   896→      skipped.push({ section: key, source, reason: 'Déjà renseigné par l\'utilisateur' })

  1009→        `## Investissements\n\n**Investissement initial** : ${financialForecast.initialInvestment.toLocaleString('fr-FR')} €\n\n*Cette donnée provient du module Prévisions Financières. Détaillez les postes d'investissement (matériel, logiciel, caution, etc.) pour compléter cette section.*`,
  1010→        'Prévisions financières',
  1011→        'financier',
  1012→      )
  1013→    } else {
  1014→      skipped.push({ section: 'investissements', source: 'Prévisions financières', reason: 'Aucun investissement renseigné' })
  1015→    }
  1016→  } else {
  1017→    skipped.push({ section: 'financement', source: 'Prévisions financières', reason: 'Simulation non complétée' })
  1018→    skipped.push({ section: 'compte-resultat', source: 'Prévisions financières', reason: 'Simulation non complétée' })
  1019→    skipped.push({ section: 'investissements', source: 'Prévisions financières', reason: 'Simulation non complétée' })
  1020→  }
  1021→
  1022→  // 5. Map creaSim → rentabilite
  1023→  if (creaSim) {
  1024→    const rentParts: string[] = []
  1025→    const fmtEur = (n: number) => n.toLocaleString('fr-FR')
  1026→    const fmtPct = (n: number) => `${n.toFixed(1)}%`
  1027→
  1028→    if (creaSim.monthlyRevenue) rentParts.push(`**CA mensuel** : ${fmtEur(creaSim.monthlyRevenue)} €`)
  1029→    if (creaSim.grossMarginRate) rentParts.push(`**Marge brute** : ${fmtPct(creaSim.grossMarginRate)}`)
  1030→    if (creaSim.netMarginRate) rentParts.push(`**Marge nette** : ${fmtPct(creaSim.netMarginRate)}`)
  1031→    if (creaSim.monthlyBreakeven) rentParts.push(`**Seuil de rentabilité mensuel** : ${fmtEur(creaSim.monthlyBreakeven)} €`)
  1032→    if (creaSim.breakevenMonths) rentParts.push(`**Point mort** : ${creaSim.breakevenMonths.toFixed(1)} mois`)
  1033→
  1034→    if (creaSim.profitability1Y != null) {
  1035→      rentParts.push(`**Rentabilité Année 1** : ${fmtEur(creaSim.profitability1Y)} € ${creaSim.profitability1Y >= 0 ? '✅' : '⚠️'}`)
  1036→    }
  1037→    if (creaSim.profitability2Y != null) {
  1038→      rentParts.push(`**Rentabilité Année 2** : ${fmtEur(creaSim.profitability2Y)} € ${creaSim.profitability2Y >= 0 ? '✅' : '⚠️'}`)
  1039→    }
  1040→    if (creaSim.profitability3Y != null) {
  1041→      rentParts.push(`**Rentabilité Année 3** : ${fmtEur(creaSim.profitability3Y)} € ${creaSim.profitability3Y >= 0 ? '✅' : '⚠️'}`)
  1042→    }
  1043→
  1044→    // Margin analysis
  1045→    if (creaSim.averageSellingPrice && creaSim.unitCost) {
  1046→      const marginPerUnit = creaSim.averageSellingPrice - creaSim.unitCost
  1047→      rentParts.push(`**Marge unitaire** : ${fmtEur(marginPerUnit)} € (PV ${fmtEur(creaSim.averageSellingPrice)} € - Coût ${fmtEur(creaSim.unitCost)} €)`)
  1048→    }
  1049→    if (creaSim.targetMarginRate) {
  1050→      rentParts.push(`**Objectif de marge** : ${fmtPct(creaSim.targetMarginRate)}`)
  1051→      if (creaSim.grossMarginRate) {
  1052→        const gap = creaSim.grossMarginRate - creaSim.targetMarginRate
  1053→        rentParts.push(`**Écart par rapport à l'objectif** : ${gap >= 0 ? '+' : ''}${fmtPct(gap)} ${gap >= 0 ? '✅ Atteint' : '⚠️ Non atteint'}`)
  1054→      }
  1055→    }
  1056→
  1057→    if (rentParts.length > 0) {
  1058→      fillSection('seuil-rentabilite', `## Analyse de rentabilité (CreaSim)\n\n${rentParts.join('\n\n')}`, 'CreaSim', 'creasim')
  1059→    } else {
  1060→      skipped.push({ section: 'seuil-rentabilite', source: 'CreaSim', reason: 'Données insuffisantes' })
  1061→    }
  1062→  } else {
  1063→    skipped.push({ section: 'seuil-rentabilite', source: 'CreaSim', reason: 'Simulation non complétée' })
  1064→  }
  1065→
  1066→  // 6. Map juridiqueAnalysis → statutJuridique
  1067→  if (juridiqueAnalysis) {
  1068→    const jurParts: string[] = []
  1069→    if (juridiqueAnalysis.recommendedStatus) jurParts.push(`**Statut recommandé** : ${juridiqueAnalysis.recommendedStatus}`)
  1070→    if (juridiqueAnalysis.fiscalRegime) jurParts.push(`**Régime fiscal** : ${juridiqueAnalysis.fiscalRegime}`)
  1071→    if (juridiqueAnalysis.legalStructure) jurParts.push(`**Structure juridique** : ${juridiqueAnalysis.legalStructure}`)
  1072→
  1073→    const charges = juridiqueAnalysis.socialCharges as unknown[]
  1074→    if (Array.isArray(charges) && charges.length > 0) {
  1075→      const chargeItems = charges.map((c: unknown) => {
  1076→        const charge = c as Record<string, string>
  1077→        return `- ${charge.name || charge.type || 'Charge'} : ${charge.amount || charge.rate || ''}`
  1078→      })
  1079→      jurParts.push(`**Charges sociales** :\n${chargeItems.join('\n')}`)
  1080→    }
  1081→
  1082→    if (jurParts.length > 0) {
  1083→      fillSection('statut-juridique', `## Statut juridique\n\n${jurParts.join('\n\n')}`, 'Analyse juridique', 'juridique')
  1084→    } else {
  1085→      skipped.push({ section: 'statut-juridique', source: 'Analyse juridique', reason: 'Données insuffisantes' })
  1086→    }
  1087→  } else {
  1088→    skipped.push({ section: 'statut-juridique', source: 'Analyse juridique', reason: 'Simulation non complétée' })
  1089→  }
  1090→
  1091→  // 7. Map businessModelCanvas → businessModelCanvas
  1092→  if (businessModelCanvas) {
  1093→    const bmcParts: string[] = []
  1094→    const bmcLabels: Record<string, string> = {
  1095→      partenairesCles: 'Partenaires clés',
  1096→      activitesCles: 'Activités clés',
  1097→      ressourcesCles: 'Ressources clés',
  1098→      propositionValeur: 'Proposition de valeur',
  1099→      relationsClients: 'Relations clients',
  1100→      canaux: 'Canaux de distribution',
  1101→      segmentsClients: 'Segments de clientèle',
  1102→      structureCouts: 'Structure des coûts',
  1103→      sourcesRevenus: 'Sources de revenus',
  1104→    }
  1105→
  1106→    let hasContent = false
  1107→    for (const [field, label] of Object.entries(bmcLabels)) {
  1108→      const value = businessModelCanvas[field as keyof typeof businessModelCanvas]
  1109→      if (value && typeof value === 'string' && value.trim()) {
  1110→        bmcParts.push(`### ${label}\n${value}`)
  1111→        hasContent = true
  1112→      }
  1113→    }
  1114→
  1115→    // BMC has no matching section in the BP — data is available in the BMC module
  1116→    skipped.push({ section: 'businessModelCanvas', source: 'Business Model Canvas', reason: 'Aucune section correspondante dans le BP' })
  1117→  } else {
  1118→    skipped.push({ section: 'businessModelCanvas', source: 'Business Model Canvas', reason: 'Simulation non complétée' })
  1119→  }
  1120→
  1121→  // 8. Recalculate bpScore and save (including bpSectionMeta)
  1122→  const filledCount = Object.values(mergedSections).filter(isFilled).length
  1123→  const totalSections = 24
  1124→  const bpScore = Math.min(100, Math.round((filledCount / totalSections) * 100))
  1125→
  1126→  await db.creatorJourney.upsert({
  1127→    where: { userId: userId },
  1128→    create: {
  1129→      userId,
  1130→      bpSections: mergedSections as Prisma.InputJsonValue,
  1131→      bpSectionMeta: sectionMeta as unknown as Prisma.InputJsonValue,
  1132→      bpScore,
  1133→      bpGeneratedAt: new Date(),
  1134→    },
  1135→    update: {
  1136→      bpSections: mergedSections as Prisma.InputJsonValue,
  1137→      bpSectionMeta: sectionMeta as unknown as Prisma.InputJsonValue,
  1138→      bpScore,
  1139→      bpGeneratedAt: new Date(),
  1140→    },
  1141→  })
  1142→
  1143→  // Derive syncedSources from synced array
  1144→  const syncedSources: Record<string, boolean> = {}
  1145→  for (const item of synced) {
  1146→    const source = item.source as string
  1147→    if (source === 'Analyse de marché') syncedSources.marche = true
  1148→    if (source === 'Prévisions financières' || source === 'CreaSim') syncedSources.financier = true
  1149→    if (source === 'Analyse juridique') syncedSources.juridique = true
  1150→    if (source === 'CreaSim') syncedSources.creasim = true
  1151→  }
  1152→  const summary = `${synced.length} sections synchronisées depuis ${[...new Set(synced.map(s => s.source))].join(', ')}`
  1153→
  1154→  return success(
  1155→    {
  1156→      sections: mergedSections,
  1157→      synced,
  1158→      skipped,
  1159→      bpScore,
  1160→      totalSections,
  1161→      syncedSources,
  1162→      summary,
  1163→    },
  1164→    `Synchronisation terminée : ${synced.length} sections ajoutées, ${skipped.length} ignorées`,
  1165→  )
  1166→}
